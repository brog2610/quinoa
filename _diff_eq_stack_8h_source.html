00077     //! \brief Function object for registering a differential equation into the
00078     //!   differential equation factory
00079     //! \details This functor is repeatedly called by MPL's cartesian_product,
00080     //!   sweeping all combinations of the differential equation policies. The
00081     //!   purpose of template template is to simplify client code as that will
00082     //!   not have to specify the template arguments of the template argument
00083     //!   (the policies of Eq), since we can figure it out here. See also
00084     //!   http://stackoverflow.com/a/214900
00085     //! \author J. Bakosi
00086     template< template< class, class > class Eq >
00087     struct registerDiffEq {
00088       //! Need to store the reference to factory we are registering into
00089       DiffEqFactory& factory;
00090       //! Need to store which differential equation we are registering
00091       const ctr::DiffEqType type;
00092       //! Constructor, also count number of unique equation types registered
00093       explicit registerDiffEq( DiffEqFactory& f,
00094                                ctr::DiffEqType t,
00095                                std::set< ctr::DiffEqType >& eqTypes ) :
00096         factory( f ), type( t ) { eqTypes.insert( t ); }
00097       //! \brief Function call operator called by mpl::cartesian_product for
00098       //!   each unique sequence of policy combinations
00099       template< typename U > void operator()( U ) {
00100         namespace mpl = boost::mpl;
00101         // Get Initialization policy: 1st type of mpl::vector U
00102         using InitPolicy = typename mpl::at< U, mpl::int_<0> >::type;
00103         // Get coefficients policy: 2nd type of mpl::vector U
00104         using CoeffPolicy = typename mpl::at< U, mpl::int_<1> >::type;
00105         // Build differential equation key
00106         ctr::DiffEqKey key{ type, InitPolicy::type(), CoeffPolicy::type() };
00107         // Register equation (with policies given by mpl::vector U) into factory
00108         tk::recordModelLate< DiffEq, Eq< InitPolicy, CoeffPolicy > >
00109                            ( factory, key, static_cast<ncomp_t>(0) );
00110       }
00111     };

00113     //! \brief Instantiate a differential equation
00114     //! \details The template argument, EqTag, is used to find the given
00115     //!   differential equation in the input deck, the hierarchical data filled
00116     //!   during control file parsing, containing user input.
00117     //! \param[in] eq The unique differential equation key whose object to
00118     //!   instantiate.
00119     //! \param[inout] cnt Counter, a std::map, that counts all instantiated
00120     //!   differential equations by type.
00121     //! \author J. Bakosi
00122     template< class EqTag >
00123     DiffEq createDiffEq( ctr::DiffEqType eq,
00124                          std::map< ctr::DiffEqType, ncomp_t >& cnt ) const {
00125       auto c = ++cnt[ eq ];   // count eqs
00126       --c;                    // used to index vectors starting with 0
00127       if ( g_inputdeck.get< tag::component, EqTag >()[c] ) {
00128         // re-create key and search for it
00129         ctr::DiffEqKey key{ eq,
00130           g_inputdeck.get< tag::param, EqTag, tag::initpolicy >()[c],
00131           g_inputdeck.get< tag::param, EqTag, tag::coeffpolicy >()[c] };
00132         const auto it = m_factory.find( key );
00133         Assert( it != end( m_factory ), "Can't find eq in factory" );
00134         return it->second( c );    // instantiate and return diff eq object
00135       } else Throw ( "Can't create DiffEq with zero independent variables" );
00136     }
