77  //! \brief Function object for registering a differential equation into the
78  //! differential equation factory
79  //! \details This functor is repeatedly called by MPL's cartesian_product,
80  //! sweeping all combinations of the differential equation policies. The
81  //! purpose of template template is to simplify client code as that will
82  //! not have to specify the template arguments of the template argument
83  //! (the policies of Eq), since we can figure it out here. See also
84  //! http://stackoverflow.com/a/214900
85  //! \author J. Bakosi
86  template< template< class, class > class Eq >
87  struct registerDiffEq {
88  //! Need to store the reference to factory we are registering into
89  DiffEqFactory& factory;
90  //! Need to store which differential equation we are registering
91  const ctr::DiffEqType type;
92  //! Constructor, also count number of unique equation types registered
93  explicit registerDiffEq( DiffEqFactory& f,
94  ctr::DiffEqType t,
95  std::set< ctr::DiffEqType >& eqTypes ) :
96  factory( f ), type( t ) { eqTypes.insert( t ); }
97  //! \brief Function call operator called by mpl::cartesian_product for
98  //! each unique sequence of policy combinations
99  template< typename U > void operator()( U ) {
100  namespace mpl = boost::mpl;
101  // Get Initialization policy: 1st type of mpl::vector U
102  using InitPolicy = typename mpl::at< U, mpl::int_<0> >::type;
103  // Get coefficients policy: 2nd type of mpl::vector U
104  using CoeffPolicy = typename mpl::at< U, mpl::int_<1> >::type;
105  // Build differential equation key
106  ctr::DiffEqKey key{ type, InitPolicy::type(), CoeffPolicy::type() };
107  // Register equation (with policies given by mpl::vector U) into factory
108  tk::recordModelLate< DiffEq, Eq< InitPolicy, CoeffPolicy > >
109  ( factory, key, static_cast<ncomp_t>(0) );
110  }
111  };

113  //! \brief Instantiate a differential equation
114  //! \details The template argument, EqTag, is used to find the given
115  //! differential equation in the input deck, the hierarchical data filled
116  //! during control file parsing, containing user input.
117  //! \param[in] eq The unique differential equation key whose object to
118  //! instantiate.
119  //! \param[inout] cnt Counter, a std::map, that counts all instantiated
120  //! differential equations by type.
121  //! \author J. Bakosi
122  template< class EqTag >
123  DiffEq createDiffEq( ctr::DiffEqType eq,
124  std::map< ctr::DiffEqType, ncomp_t >& cnt ) const {
125  auto c = ++cnt[ eq ]; // count eqs
126  --c; // used to index vectors starting with 0
127  if ( g_inputdeck.get< tag::component, EqTag >()[c] ) {
128  // re-create key and search for it
129  ctr::DiffEqKey key{ eq,
130  g_inputdeck.get< tag::param, EqTag, tag::initpolicy >()[c],
131  g_inputdeck.get< tag::param, EqTag, tag::coeffpolicy >()[c] };
132  const auto it = m_factory.find( key );
133  Assert( it != end( m_factory ), "Can't find eq in factory" );
134  return it->second( c ); // instantiate and return diff eq object
135  } else Throw ( "Can't create DiffEq with zero independent variables" );
136  }
