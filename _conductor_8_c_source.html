00134 void
00135 Conductor::timestamp(
00136   const std::vector< std::pair< std::string, tk::real > >& stamp,
00137   std::map< std::string, std::vector< tk::real > >& map,
00138   int& counter,
00139   int max,
00140   const std::string& of )
00141 //******************************************************************************
00142 // Collect and compute averages over time stamps contributed by chares
00143 //! \param[in] stamp Vector of time stamps contributed
00144 //! \param[inout] map Map in which to sore labels and vector of times
00145 //! \param[inout] counter Counter to use
00146 //! \param[in] max Max number of contributions to expect
00147 //! \param[in] of Of what the averages computed over
00148 //! \author J. Bakosi
00149 //******************************************************************************
00150 {
00151   // Collect time stamps with the same name. These come from different chares
00152   // performing their portion of some work.
00153   for (const auto& s : stamp) map[ s.first ].push_back( s.second );
00154 
00155   if (++counter == max) {       // if all expected have arrived
00156     // Compute average over chares and send to main proxy for printing
00157     mainProxy.timestamp(
00158       tk::average( map, ", avg of " + std::to_string(max) + " " + of ) );
00159     // Prepare for next round
00160     counter = 0;
00161     map.clear();
00162   }
00163 }
