00276   static_assert( d >= 1 && d <= 3 ,

00278   long int n = -1;
00279   std::vector< std::string > v;
00280   for (const auto& probability : pdfs) {
00281     if (probability.size() == d) ++n;
00282     if (n == idx) {
00283       for (const auto& term : probability)
