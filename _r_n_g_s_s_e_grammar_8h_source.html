00022   //!    by tags
00023   //! \author J. Bakosi
00024   template< class Stack, template < class > class use, class Option,
00025             typename field, typename sel, typename vec, typename tag,
00026             typename... tags >
00027   struct insert_seq : pegtl::action_base<
00028                         insert_seq< Stack, use, Option, field, sel, vec, tag,
00029                                     tags... > > {
00030     static void apply( const std::string& value, Stack& stack ) {
00031       ctr::RNGSSESeqLen opt;
00032       using EnumType = ctr::RNGSSESeqLen::EnumType;
00033       // get recently inserted key from <sel,vec>
00034       using key_type =
00035         typename Stack::template nT< sel >::template nT< vec >::value_type;
00036       const key_type& key = stack.template get< sel, vec >().back();
00037       // Error out if RNG does not support option specified
00038       if ( !ctr::RNG().supportsOpt( key, opt.value(value) ) ) {
00039         grm::Message< Stack, grm::ERROR, grm::MsgKey::UNSUPPORTED >
00040                     ( stack, value );
00041       }
00042       stack.template insert_opt< key_type, field, EnumType, tag, tags... >
00043                                ( key, opt.value(value) );
00044       // trigger error at compile-time if any of the expected option values
00045       // is not in the keywords pool of the grammar
00046       boost::mpl::for_each< typename Option::keywords >
00047                           ( tk::grm::is_keyword< use >() );
00048     }
00049   };
