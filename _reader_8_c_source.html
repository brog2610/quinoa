00023 Reader::Reader( const std::string& filename ) : m_filename( filename )
00024 //******************************************************************************
00025 //  Constructor: Acquire file handle
00026 //! \param[in] filename Name of file to open for reading
00027 //! \author J. Bakosi
00028 //******************************************************************************
00029 {
00030   // Make sure there is a filename
00031   Assert( !filename.empty(), "No filename specified" );
00032 
00033   // Check if file exists, throw exception if it does not
00034   m_inFile.open( filename, std::ifstream::in );
00035   ErrChk( m_inFile.good(), "Failed to open file: " + filename );
00036 
00037   // Attempt to read a character, throw if it fails
00038   // It is curious that on some systems opening a directory instead of a file
00039   // with the above ifstream::open() call does not set the failbit. Thus we get
00040   // here fine, so we try to read a character from it. If it is a directory or
00041   // an empty file the read will fail, so we throw. Read more at: http://
00042   // stackoverflow.com/questions/9591036/
00043   // ifstream-open-doesnt-set-error-bits-when-argument-is-a-directory.
00044   m_inFile.get();
00045   ErrChk( m_inFile.good(), "Failed to read from file: " + filename );
00046 
00047   // Close it
00048   m_inFile.close();
00049   ErrChk( !m_inFile.fail(), "Failed to close file: " + filename );
00050 
00051   // Re-open
00052   m_inFile.open( filename, std::ifstream::in );
00053   ErrChk( m_inFile.good(), "Failed to open file: " + filename );
00054 }
