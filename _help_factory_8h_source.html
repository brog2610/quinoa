00024 //! \brief Keyword information bundle
00025 //! \details This bundle contains the information that is used to display
00026 //!    on-screen help on all command-line arguments and control file keywords
00027 //!    for an exectuable. This struct is stored in a container that associates
00028 //!    keywords (used by a grammar and parser) to this struct. The container, an
00029 //!    runtime, std::map, is filled by the CmdLine and InputDeck objects'
00030 //!    constructors by one or more boost::mpl::for_each which loops through the
00031 //!    set of all keywords used in a grammar. The maps are stored in the CmdLine
00032 //!    and InputDeck objects (which are tagged tuples) and thus can be migrated
00033 //!    through the network, thus the Charm++ parck/unpack routines are defined.
00034 //! \see Info functor used to fill the std::maps
00035 //! \author J. Bakosi
00036 struct KeywordInfo {
00037   std::string shortDescription;           //!< Short description
00038   std::string longDescription;            //!< Long description
00039   boost::optional< std::string > alias;   //!< Keyword alias
00040   boost::optional< std::string > expt;    //!< Expected type description
00041   boost::optional< std::string > choices; //!< Expected choices descr.
00042 
00043   /** @name Pack/Unpack: Serialize KeywordInfo object for Charm++ */
00044   ///@{
00045   //! \brief Pack/Unpack serialize member function
00046   //! \param[inout] p Charm++'s PUP::er serializer object reference
00047   //! \author J. Bakosi
00048   void pup( PUP::er& p ) {
00049     p | shortDescription;
00050     p | longDescription;
00051     p | alias;
00052     p | expt;
00053     p | choices;
00054   }
00055   //! \brief Pack/Unpack serialize operator|
00056   //! \param[inout] p Charm++'s PUP::er serializer object reference
00057   //! \param[inout] info KeywordInfo object reference
00058   //! \author J. Bakosi
00059   friend void operator|( PUP::er& p, KeywordInfo& info ) { info.pup(p); }
00060   ///@}
