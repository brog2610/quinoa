39  //! Constructor: Quiet output by default, only stuff written to qstr shown.
40  //! Instantiate with str = std::cout for verbose output. Any member function
41  //! can be called by overriding the default stream via the template
42  //! argument, Style, a C-style enum. Note: By default, str == std::clog.
43  //! This is used to initialize str to a local stringstream into which all
44  //! verbose output goes by default, i.e., it will not be shown. This
45  //! solution is chosen instead of trickery with null-streams, as
46  //! boost:formatted output into null-streams caused invalid reads in
47  //! valgrind. This way quiet output (formatted or not) simply goes into a
48  //! local stringstream. In other words, the default argument to str,
49  //! std::clog, is only used to detect whether client code passed a default
50  //! argument or not: if it did not, the string stream is used for verbose
51  //! output, if it did, the specified stream is used for the verbose output.
52  //! \param[inout] str Verbose stream
53  //! \param[inout] qstr Quiet stream
54  //! \author J. Bakosi

592  //! \brief Clean up whitespaces and format a long string into multiple lines
593  //! \param[in] str String to format
594  //! \param[in] name String to insert before string to output
595  //! \param[in] indent String to use as identation
596  //! \param[in] width Width in characters to insert newlines for output
597  //! \author J. Bakosi
598  //! \see http://stackoverflow.com/a/6892562
599  //! \see http://stackoverflow.com/a/8362145
600  // TODO A line longer than 'width' will cause a hang!
601  std::string splitLines( std::string str,
602  std::string indent,
603  std::string name = "",
604  std::size_t width = 80 ) const {
605  // remove form feeds, line feeds, carriage returns, horizontal tabs,
606  // vertical tabs, see http://en.cppreference.com/w/cpp/string/byte/isspace
607  str.erase(
608  std::remove_if( str.begin(), str.end(),
609  []( char x ){ return std::isspace( x ) && x != ' '; } ),
610  str.end() );
611  // remove duplicate spaces
612  str.erase(
613  std::unique( str.begin(), str.end(),
614  []( char a, char b ){ return a == b && a == ' '; } ),
615  str.end() );
616  // format str to 'width'-character-long lines with indent
617  str.insert( 0, indent + name );
618  std::size_t currIndex = width - 1;
619  std::size_t sizeToElim;
620  while ( currIndex < str.length() ) {
621  const std::string whitespace = " ";
622  currIndex = str.find_last_of( whitespace, currIndex + 1 );
623  if ( currIndex == std::string::npos ) break;
624  currIndex = str.find_last_not_of( whitespace, currIndex );
625  if ( currIndex == std::string::npos ) break;
626  sizeToElim =
627  str.find_first_not_of( whitespace, currIndex + 1 ) - currIndex - 1;
628  str.replace( currIndex + 1, sizeToElim , "\n" + indent );
629  currIndex += width + indent.length() + 1;
630  }
631  return str;
632  }
