00028 //  Generate derived data structure, elements surrounding points
00029 //! \param[in] inpoel Inteconnectivity of points and elements. These are the
00030 //!   node ids of each element of an unstructured mesh. Example:
00031 //!   \code{.cpp}
00032 //!     std::vector< std::size_t > inpoel { 12, 14,  9, 11,
00033 //!                                         10, 14, 13, 12 };
00034 //!   \endcode
00035 //!   specifies two tetrahedra whose vertices (node ids) are { 12, 14, 9, 11 },
00036 //!   and { 10, 14, 13, 12 }.
00037 //! \param[in] nnpe Number of nodes per element
00038 //! \return Linked lists storing elements surrounding points
00039 //! \warning It is not okay to call this function with an empty container or a
00040 //!   non-positive number of nodes per element; it will throw an exception.
00041 //! \details The data generated here is stored in a linked list, more precisely,
00042 //!   two linked arrays (vectors), _esup1_ and _esup2_, where _esup2_ holds the
00043 //!   indices at which _esup1_ holds the element ids surrounding points. Looping
00044 //!   over all elements surrounding all points can then be accomplished by the
00045 //!   following loop:
00046 //!   \code{.cpp}
00047 //!     for (std::size_t p=0; p<npoin; ++p)
00048 //!       for (auto i=esup.second[p]+1; i<=esup.second[p+1]; ++i)
00049 //!          use element id esup.first[i]
00050 //!   \endcode
00051 //!     To find out the number of points, _npoin_, the mesh connectivity,
00052 //!     _inpoel_, can be queried:
00053 //!   \code{.cpp}
00054 //!     auto minmax = std::minmax_element( begin(inpoel), end(inpoel) );
00055 //!     Assert( *minmax.first == 0, "node ids should start from zero" );
00056 //!     auto npoin = *minmax.second + 1;
00057 //!   \endcode
00058 //! \note In principle, this function *should* work for any positive nnpe,
00059 //!   however, only nnpe = 4 (tetrahedra) and nnpe = 3 (triangles) are tested.
00060 //! \see Lohner, An Introduction to Applied CFD Techniques, Wiley, 2008
00061 //! \author J. Bakosi
00062 //******************************************************************************
00063 {
00064   Assert( !inpoel.empty(), "Attempt to call genEsup() on empty container" );
00065   Assert( nnpe > 0, "Attempt to call genEsup() with zero nodes per element" );
00066   Assert( inpoel.size()%nnpe == 0, "Size of inpoel must be divisible by nnpe" );
00067 
00068   // find out number of points in mesh connectivity
00069   auto minmax = std::minmax_element( begin(inpoel), end(inpoel) );
00070   Assert( *minmax.first == 0, "node ids should start from zero" );
00071   auto npoin = *minmax.second + 1;
00072 
00073   // allocate one of the linked lists storing elements surrounding points: esup2
00074   // fill with zeros
00075   std::vector< std::size_t > esup2( npoin+1, 0 );
00076 
00077   // element pass 1: count number of elements connected to each point
00078   for (auto n : inpoel) ++esup2[ n + 1 ];
00079 
00080   // storage/reshuffling pass 1: update storage counter and store
00081   // also find out the maximum size of esup1 (mesup)
00082   auto mesup = esup2[0]+1;
00083   for (std::size_t i=1; i<npoin+1; ++i) {
00084     esup2[i] += esup2[i-1];
00085     if (esup2[i]+1 > mesup) mesup = esup2[i]+1;
00086   }
00087 
00088   // now we know mesup, so allocate the other one of the linked lists storing
00089   // elements surrounding points: esup1
00090   std::vector< std::size_t > esup1( mesup );
00091 
00092   // store the elements in esup1
00093   std::size_t e = 0;
00094   for (auto n : inpoel) {
00095     auto j = esup2[n]+1;
00096     esup2[n] = j;
00097     esup1[j] = e/nnpe;
00098     ++e;
00099   }
00100 
00101   // storage/reshuffling pass 2
00102   for (auto i=npoin; i>0; --i) esup2[i] = esup2[i-1];
00103   esup2[0] = 0;
00104 
00105   // Return (move out) linked lists
00106   return std::make_pair( std::move(esup1), std::move(esup2) );

00585 //! \param[in] inpoel Inteconnectivity of points and elements. These are the

00592 //!   and { 10, 14, 13, 12 }.
00593 //! \param[in] nnpe Number of nodes per element
00594 //! \param[in] inpoed Edge connectivity as linear vector, see tk::genInpoed
00595 //! \return Linear vector storing all edge ids * 2 of all elements
00596 //! \warning It is not okay to call this function with an empty container for
00597 //!   inpoel or inpoed or a non-positive number of nodes per element; it will
00598 //!   throw an exception.
00599 //! \details The data generated here is stored in a linear vector with all
00600 //!   edge ids (as defined by inpoed) of all elements. The edge ids stored in
00601 //!   inedel can be directly used to index the vector inpoed. Because the
00602 //!   derived data structure generated here, inedel, is intended to be used in
00603 //!   conjunction with the linear vector inpoed and not with the linked lists
00604 //!   edsup1 and edsup2, this function takes inpoed as an argument. Accessing
00605 //!   the edges of element e using the edge of elements data structure, inedel,
00606 //!   generated here can be accomplished by
00607 //!   \code{.cpp}
00608 //!     for (std::size_t e=0; e<nelem; ++e) {
00609 //!       for (std::size_t i=0; i<nepe; ++i) {
00610 //!         use edge id inedel[e*nepe+i] of element e, or
00611 //!         use point ids p < q of edge id inedel[e*nepe+i] of element e as
00612 //!           p = inpoed[ inedel[e*nepe+i]*2 ]
00613 //!           q = inpoed[ inedel[e*nepe+i]*2+1 ]
00614 //!       }
00615 //!     }
00616 //!   \endcode
00617 //!   where _nepe_ denotes the number of edges per elements: 3 for triangles, 6
00618 //!   for tetrahedra. To find out the number of elements, _nelem_, the size of
00619 //!   the mesh connectivity vector, _inpoel_, can be devided by the number of
00620 //!   nodes per elements, _nnpe_:
00621 //!   \code{.cpp}
00622 //!     auto nelem = inpoel.size()/nnpe;
00623 //!   \endcode
00624 //! \note At first sight, this function seems to work for elements with more
00625 //!   vertices than that of tetrahedra. However, that is not the case since the
00626 //!   algorithm for nnpe > 4 would erronously identify any two combination of
00627 //!   vertices as a valid edge of an element. Since only triangles and
00628 //!   tetrahedra have no internal edges, this algorithm only works for triangle
00629 //!   and tetrahedra element connectivity.

00631 //! \author J. Bakosi
00632 //******************************************************************************
00633 {
00634   Assert( !inpoel.empty(), "Attempt to call genInedel() on empty container" );
00635   Assert( nnpe > 0, "Attempt to call genInedel() with zero nodes per element" );
00636   Assert( nnpe == 3 || nnpe == 4,
00637           "Attempt to call genInedel() with nodes per element, nnpe, that is "
00638           "neither 4 (tetrahedra) nor 3 (triangles)." );
00639   Assert( inpoel.size()%nnpe == 0, "Size of inpoel must be divisible by nnpe" );
00640   Assert( !inpoed.empty(), "Attempt to call genInedel() with empty inpoed" );
00641 
00642   // find out number of points in mesh connectivity
00643   auto minmax = std::minmax_element( begin(inpoel), end(inpoel) );
00644   Assert( *minmax.first == 0, "node ids should start from zero" );
00645   auto npoin = *minmax.second + 1;
00646 
00647   // First, generate index of star centers. This is necessary to avoid a
00648   // brute-force search for point ids of edges when searching for element edges.
00649   // Note that this is the same as edsup2, generated by genEdsup(). However,
00650   // because the derived data structure generated here, inedel, is intended to
00651   // be used in conjunction with the linear vector inpoed and not with the
00652   // linked lists edsup1 and edsup2, this function takes inpoed as an argument,
00653   // and so edsup2 is temporarily generated here to avoid a brute-force search.
00654

00656   // storing only the end-point id, q, of point ids p < q
00657   std::map< std::size_t, std::vector< std::size_t > > star;
00658 
00659   // generate stars from inpoed; starting with zero, every even is a star
00660   // center, every odd is a spike
00661   for (std::size_t i=0; i<inpoed.size()/2; ++i)
00662     star[ inpoed[i*2] ].push_back( inpoed[i*2+1] );
00663 
00664   // store index of star centers in vector; assume non-center points of each
00665   // star have already been sorted
00666   std::vector< std::size_t > edsup2( 1, 0 );
00667   for (auto p : star) edsup2.push_back( edsup2.back() + p.second.size() );
00668   // fill up index array with the last index for points with no new edges
00669   for (std::size_t i=0; i<npoin-star.size(); ++i)
00670     edsup2.push_back( edsup2.back() );
00671   star.clear();
00672 
00673   // Second, generate edges of elements
00674 
00675   auto nelem = inpoel.size()/nnpe;
00676 
00677   // map associating elem id with vector of edge ids
00678   std::map< std::size_t, std::vector< std::size_t > > edges;
00679 
00680   // generate map of elements associated to edge ids

00683       auto p = inpoel[e*nnpe+n];

00686             if (inpoed[(i-1)*2+1] == inpoel[e*nnpe+j])

00688     }

00691   std::vector< std::size_t > inedel;
00692 
00693   // store edge ids of elements in linear vector
00694   for (auto e : edges) for (auto p : e.second) inedel.push_back( p );

00696   // Return (move out) vector
00697   return inedel;
