00132 //! \brief Zoltan query function query function which format we will supply the
00133 //!   hypergraph
00134 //! \details A hypergraph can be supplied to the Zoltan library in one of two
00135 //!   compressed storage formats. Although hypergraphs are often used to
00136 //!   represent the structure of sparse matrices, the Zoltan/PHG terminology is
00137 //!   purely in terms of vertices and hyperedges (not rows or columns). The two
00138 //!   compressed formats are analogous to CRS and CCS for matrices.
00139 //!
00140 //!   In compressed hyperedge format (ZOLTAN_COMPRESSED_EDGE) a list of global
00141 //!   hyperedge IDs is provided. Then a single list of the hypergraph pins, is
00142 //!   provided. A pin is the connection between a vertex and a hyperedge
00143 //!   (corresponds to a nonzero in a sparse matrix). Pins do not have separate
00144 //!   IDs but are rather identified by the global ID of the vertex containing
00145 //!   the pin, and implicitly also by the hyperedge ID.
00146 //!
00147 //!   The other format is compressed vertex (ZOLTAN_COMPRESSED_VERTEX). In this
00148 //!   format a list of vertex global IDs is provided. Then a list of pins
00149 //!   ordered by vertex and then by hyperedge is provided. The pin ID in this
00150 //!   case is the global ID of the hyperedge in which the pin appears. In both
00151 //!   formats, an array must be provided pointing to the start in the list of
00152 //!   pins where each hyperedge or vertex begins.
00153 //!
00154 //!   The purpose of this query function is to tell Zoltan in which format we
00155 //!   will supply the hypergraph, how many vertices and hyperedges there will
00156 //!   be, and how many pins. The actual hypergraph is supplied with a query
00157 //!   function of the type ZOLTAN_HG_CS_FN_TYPE.
00158 //!
00159 //!   This query function is required by all applications using the hypergraph
00160 //!   methods of Zoltan (unless they are using the graph-based functions with
00161 //!   hypergraph code instead).
00162 //! \param[inout] data Pointer to user-defined data
00163 //! \param[inout] num_lists Upon return, the number of vertices (if using
00164 //!   compressed vertex storage) or hyperedges (if using compressed hyperedge
00165 //!   storage) that will be supplied to Zoltan by the MPI rank.
00166 //! \param[inout] num_nonzeros Upon return, the number of pins (connections
00167 //!   between vertices and hyperedges) that will be supplied to Zoltan by the
00168 //!   MPI rank.
00169 //! \param[inout] format Upon return, the format in which we will provide the
00170 //!   hypergraph to Zoltan. The options are ZOLTAN_COMPRESSED_EDGE and
00171 //!   ZOLTAN_COMPRESSED_VERTEX.
00172 //! \param[inout] ierr Error code
00173 //! \author J. Bakosi
00174 //******************************************************************************
00175 {
00176   HGRAPH_DATA *hg = (HGRAPH_DATA *)data;
00177   *ierr = ZOLTAN_OK;
00178 
00179   *num_lists = hg->numMyHEdges;
00180   *num_nonzeros = hg->numAllNbors;
00181 
00182   // We will provide a compressed hyperedge (row) format. The alternative is
00183   // compressed vertex (column) format: ZOLTAN_COMPRESSED_VERTEX.
00184   *format = ZOLTAN_COMPRESSED_EDGE;
00185 }

00224 //!   vertex global IDs are expected to be returned in edgeGID and hyperedge
00225 //!   global IDs are expected to be returned in vtxGID.
00226 //! \param[inout] edgeGID Upon return, a list of num_edges global IDs.
00227 //! \param[inout] vtxPtr Upon return, this array contains num_edges integers

00229 //!   ZOLTAN_COMPRESSED_EDGE) or vertex j (if format is
00230 //!   ZOLTAN_COMPRESSED_VERTEX) is vtxPtr[j+1]-vtxPtr[j]. If the format is
00231 //!   ZOLTAN_COMPRESSED_EDGE, vtxPtr[j]*sizeGID is the index into the array
00232 //!   vtxGID where edge j's pins (vertices belonging to edge j) begin; if the
00233 //!   format is ZOLTAN_COMPRESSED_VERTEX, vtxPtr[j]*sizeGID is the index into
00234 //!   the array vtxGID where vertex j's pins (edges to which vertex j belongs)
00235 //!   begin. Array indices begin at zero.
00236 //! \param[inout] vtxGID Upon return, a list of num_pins global IDs. This is the
00237 //!   list of the pins contained in the hyperedges or vertices listed in vtxGID.
00238 //! \param[inout] ierr Error code
00239 //! \author J. Bakosi
00240 //******************************************************************************
00241 {
00242   HGRAPH_DATA *hg = (HGRAPH_DATA *)data;
00243   *ierr = ZOLTAN_OK;
00244 
00245   if ( (num_edges != hg->numMyHEdges) ||
00246        (num_nonzeros != hg->numAllNbors) ||
00247        (format != ZOLTAN_COMPRESSED_EDGE) )
00248   {
00249     *ierr = ZOLTAN_FATAL;
00250     return;
00251   }
00252 
00253   for (int i=0; i<num_edges; ++i) {
00254     edgeGID[i] = hg->edgeGID[i];
00255     vtxPtr[i] = hg->nborIndex[i];
00256   }
00257 
00258   for (int i=0; i<num_nonzeros; ++i) vtxGID[i] = hg->nborGID[i];
00259 }

00262 createHyperGraph( tk::UnsMesh& graph, HGRAPH_DATA& hg )
00263 //******************************************************************************
00264 //  Create hypergraph data structure on MPI rank zero
00265 //! \param[in] graph Unstructured mesh graph object reference
00266 //! \param[inout] hg Hypergraph data structure to fill
00267 //! \return The number of hyperedges in graph, the new->old mesh point map
00268 //! \warning This function must not be called on MPI ranks other than zero.
00269 //! \author J. Bakosi

00271 {
00272   // Get number of points from graph.
00273   const auto npoin = graph.size();
00274 
00275   // Create hypergraph data structure based on mesh graph
00276   hg.numMyVertices = static_cast< int >( npoin );
00277   hg.numMyHEdges = hg.numMyVertices;
00278   hg.vtxGID = (ZOLTAN_ID_PTR)
00279     malloc(sizeof(ZOLTAN_ID_TYPE) * static_cast<std::size_t>(hg.numMyVertices));
00280   hg.edgeGID = (ZOLTAN_ID_PTR)
00281     malloc(sizeof(ZOLTAN_ID_TYPE) * static_cast<std::size_t>(hg.numMyHEdges));
00282   hg.nborIndex = (int*)
00283     malloc(sizeof(int) * static_cast<std::size_t>(hg.numMyHEdges + 1));
00284 
00285   // Assign global point ids
00286   for (std::size_t i=0; i<static_cast<std::size_t>(hg.numMyVertices); ++i)
00287     hg.vtxGID[i] = static_cast< ZOLTAN_ID_TYPE >( i );
00288 
00289   // Get tetrahedron mesh graph connectivity
00290   auto& inpoel = graph.tetinpoel();
00291 
00292   // Generate (connectivity graph) points surrounding points of graph
00293   auto psup = tk::genPsup( inpoel, 4, tk::genEsup( inpoel, 4 ) );
00294 
00295   // Renumber mesh points for better data locality
00296   std::vector< std::size_t > map, invmap;
00297   std::tie( map, invmap ) = tk::renumber( psup );
00298   // Remap element connectivity
00299   tk::remap( inpoel, map );
00300   // Re-generate points surrounding points
00301   psup = tk::genPsup( inpoel, 4, tk::genEsup( inpoel, 4 ) );
00302 
00303   // Allocate data to store the hypergraph ids. The total number of vertices or
00304   // neighbors in all the hyperedges of the hypergraph, nhedge = all points
00305   // surrounding points + number of points, since psup does not store the
00306   // connection to the own point, i.e., in matrix parlance, the main-diagonal.
00307   // In other words, here we need the number of edges in the graph, independent
00308   // of direction.
00309   auto nhedge = psup.first.size() - 1 + npoin;
00310   hg.numAllNbors = static_cast< int >( nhedge );
00311   hg.nborGID = (ZOLTAN_ID_PTR)malloc(sizeof(ZOLTAN_ID_TYPE) * nhedge);
00312 
00313   // Fill up hypergraph edge ids and their indices
00314   hg.nborIndex[0] = 0;
00315   for (std::size_t p=0; p<npoin; ++p) {
00316     hg.edgeGID[p] = static_cast< ZOLTAN_ID_TYPE >( p+1 );
00317     // put in own point id, i.e., main diagonal
00318     hg.nborGID[ hg.nborIndex[p] ] = static_cast< ZOLTAN_ID_TYPE >( p );
00319     int j = 1;
00320     // put in neighbor point ids, i.e., off-diagonals
00321     for (auto i=psup.second[p]+1; i<=psup.second[p+1]; ++i, ++j) {
00322       hg.nborGID[ hg.nborIndex[p] + j ] =
00323         static_cast< ZOLTAN_ID_TYPE >( psup.first[i] );
00324     }
00325     hg.nborIndex[p+1] = hg.nborIndex[p] + j;
00326   }
00327 
00328   return { nhedge, invmap };
00329 }
