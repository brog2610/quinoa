79 //! \brief Zoltan query function query function to fill two (three if weights
80 //! are used) arrays with information about the objects assigned to an MPI
81 //! rank
82 //! \details This Zoltan query function fills two (three if weights are used)
83 //! arrays with information about the objects assigned to an MPI rank. Both
84 //! arrays are allocated (and subsequently freed) by Zoltan; their size is
85 //! determined by a call to a ZOLTAN_NUM_OBJ_FN query function
86 //! (get_number_of_vertices) to get the array size. For many algorithms,
87 //! either a ZOLTAN_OBJ_LIST_FN query function or a
88 //! ZOLTAN_FIRST_OBJ_FN/ZOLTAN_NEXT_OBJ_FN query-function pair must be
89 //! registered; however, both query options need not be provided. The
90 //! ZOLTAN_OBJ_LIST_FN is preferred for efficiency.
91 //! \param[inout] data Pointer to user-defined data
92 //! \param[in] sizeGID The number of array entries used to describe a single
93 //! global ID. This value is the maximum value over all MPI ranks of the
94 //! parameter NUM_GID_ENTRIES.
95 //! \param[in] sizeLID The number of array entries used to describe a single
96 //! local ID. This value is the maximum value over all MPI ranks of the
97 //! parameter NUM_LID_ENTRIES. (It should be zero if local ids are not used.)
98 //! \param[inout] globalID Upon return, an array of unique global IDs for all
99 //! objects assigned to the MPI rank.
100 //! \param[inout] localID Upon return, an array of local IDs, the meaning of
101 //! which can be determined by the application, for all objects assigned to
102 //! the MPI rank (optional).
103 //! \param[in] wgt_dim The number of weights associated with an object
104 //! (typically 1), or 0 if weights are not requested. This value is set
105 //! through the parameter OBJ_WEIGHT_DIM.
106 //! \param[inout] obj_wgts Upon return, an array of object weights. Weights for
107 //! object i are stored in obj_wgts[(i-1)*wgt_dim:i*wgt_dim-1]. If wgt_dim=0,
108 //! the return value of obj_wgts is undefined and may be NULL.
109 //! \param[inout] ierr Error code
110 //! \return The number of objects that are assigned to the MPI rank
111 //! \author J. Bakosi
112 //******************************************************************************
113 {
114  HGRAPH_DATA *hg = (HGRAPH_DATA*)data;
115  *ierr = ZOLTAN_OK;
116 
117  // Return the IDs of our vertices, but no weights. Zoltan will assume equally
118  // weighted vertices.
119  for (int i=0; i<hg->numMyVertices; ++i) {
120  globalID[i] = hg->vtxGID[i];
121  localID[i] = static_cast< ZOLTAN_ID_TYPE >( i );
122  }
123 }

131 //******************************************************************************
132 //! \brief Zoltan query function query function which format we will supply the
133 //! hypergraph
134 //! \details A hypergraph can be supplied to the Zoltan library in one of two
135 //! compressed storage formats. Although hypergraphs are often used to
136 //! represent the structure of sparse matrices, the Zoltan/PHG terminology is
137 //! purely in terms of vertices and hyperedges (not rows or columns). The two
138 //! compressed formats are analogous to CRS and CCS for matrices.
139 //!
140 //! In compressed hyperedge format (ZOLTAN_COMPRESSED_EDGE) a list of global
141 //! hyperedge IDs is provided. Then a single list of the hypergraph pins, is
142 //! provided. A pin is the connection between a vertex and a hyperedge
143 //! (corresponds to a nonzero in a sparse matrix). Pins do not have separate
144 //! IDs but are rather identified by the global ID of the vertex containing
145 //! the pin, and implicitly also by the hyperedge ID.
146 //!
147 //! The other format is compressed vertex (ZOLTAN_COMPRESSED_VERTEX). In this
148 //! format a list of vertex global IDs is provided. Then a list of pins
149 //! ordered by vertex and then by hyperedge is provided. The pin ID in this
150 //! case is the global ID of the hyperedge in which the pin appears. In both
151 //! formats, an array must be provided pointing to the start in the list of
152 //! pins where each hyperedge or vertex begins.
153 //!
154 //! The purpose of this query function is to tell Zoltan in which format we
155 //! will supply the hypergraph, how many vertices and hyperedges there will
156 //! be, and how many pins. The actual hypergraph is supplied with a query
157 //! function of the type ZOLTAN_HG_CS_FN_TYPE.
158 //!
159 //! This query function is required by all applications using the hypergraph
160 //! methods of Zoltan (unless they are using the graph-based functions with
161 //! hypergraph code instead).
162 //! \param[inout] data Pointer to user-defined data
163 //! \param[inout] num_lists Upon return, the number of vertices (if using
164 //! compressed vertex storage) or hyperedges (if using compressed hyperedge
165 //! storage) that will be supplied to Zoltan by the MPI rank.
166 //! \param[inout] num_nonzeros Upon return, the number of pins (connections
167 //! between vertices and hyperedges) that will be supplied to Zoltan by the
168 //! MPI rank.
169 //! \param[inout] format Upon return, the format in which we will provide the
170 //! hypergraph to Zoltan. The options are ZOLTAN_COMPRESSED_EDGE and
171 //! ZOLTAN_COMPRESSED_VERTEX.
172 //! \param[inout] ierr Error code
173 //! \author J. Bakosi
174 //******************************************************************************
175 {
176  HGRAPH_DATA *hg = (HGRAPH_DATA *)data;
177  *ierr = ZOLTAN_OK;
178 
179  *num_lists = hg->numMyHEdges;
180  *num_nonzeros = hg->numAllNbors;
181 
182  // We will provide a compressed hyperedge (row) format. The alternative is
183  // compressed vertex (column) format: ZOLTAN_COMPRESSED_VERTEX.
184  *format = ZOLTAN_COMPRESSED_EDGE;
185 }

188 get_hypergraph( void *data,

197 //******************************************************************************
198 //! \brief Zoltan query function that returns a hypergraph in a compressed
199 //! storage (CS) format.
200 //! \details The size and format of the data to be returned must have been
201 //! supplied to Zoltan using a ZOLTAN_HG_SIZE_CS_FN_TYPE function.
202 //!
203 //! When a hypergraph is distributed across multiple MPI ranks, Zoltan expects
204 //! that all ranks share a consistent global numbering scheme for hyperedges
205 //! and vertices. Also, no two ranks should return the same pin (matrix
206 //! non-zero) in this query function. (Pin ownership is unique.)
207 //!
208 //! This query function is required by all applications using the hypergraph
209 //! methods of Zoltan (unless they are using the graph-based functions with
210 //! hypergraph code instead).
211 //!
212 //! \param[inout] data Pointer to user-defined data
213 //! \param[in] sizeGID The number of array entries used to describe a single
214 //! global ID. This value is the maximum value over all ranks of the parameter
215 //! NUM_GID_ENTRIES.
216 //! \param[in] num_edges The number of global IDs that is expected to appear on
217 //! return in edgeGID. This may correspond to either vertices or
218 //! (hyper-)edges.
219 //! \param[in] num_nonzeros The number of pins that is expected to appear on
220 //! return in vtxGID.
221 //! \param[in] format If format is ZOLTAN_COMPRESSED_EDGE, Zoltan expects that
222 //! hyperedge global IDs will be returned in edgeGID, and that vertex global
223 //! IDs will be returned in vtxGID. If it is ZOLTAN_COMPRESSED_VERTEX, then
224 //! vertex global IDs are expected to be returned in edgeGID and hyperedge
225 //! global IDs are expected to be returned in vtxGID.
226 //! \param[inout] edgeGID Upon return, a list of num_edges global IDs.
227 //! \param[inout] vtxPtr Upon return, this array contains num_edges integers
228 //! such that the number of pins specified for hyperedge j (if format is
229 //! ZOLTAN_COMPRESSED_EDGE) or vertex j (if format is
230 //! ZOLTAN_COMPRESSED_VERTEX) is vtxPtr[j+1]-vtxPtr[j]. If the format is
231 //! ZOLTAN_COMPRESSED_EDGE, vtxPtr[j]*sizeGID is the index into the array
232 //! vtxGID where edge j's pins (vertices belonging to edge j) begin; if the
233 //! format is ZOLTAN_COMPRESSED_VERTEX, vtxPtr[j]*sizeGID is the index into
234 //! the array vtxGID where vertex j's pins (edges to which vertex j belongs)
235 //! begin. Array indices begin at zero.
236 //! \param[inout] vtxGID Upon return, a list of num_pins global IDs. This is the
237 //! list of the pins contained in the hyperedges or vertices listed in vtxGID.
238 //! \param[inout] ierr Error code
239 //! \author J. Bakosi
240 //******************************************************************************
241 {
242  HGRAPH_DATA *hg = (HGRAPH_DATA *)data;
243  *ierr = ZOLTAN_OK;
244 
245  if ( (num_edges != hg->numMyHEdges) ||
246  (num_nonzeros != hg->numAllNbors) ||
247  (format != ZOLTAN_COMPRESSED_EDGE) )
248  {
249  *ierr = ZOLTAN_FATAL;
250  return;
251  }
252 
253  for (int i=0; i<num_edges; ++i) {
254  edgeGID[i] = hg->edgeGID[i];
255  vtxPtr[i] = hg->nborIndex[i];
256  }
257 
258  for (int i=0; i<num_nonzeros; ++i) vtxGID[i] = hg->nborGID[i];
259 }

261 static std::pair< std::size_t, std::vector< std::size_t > >
262 createHyperGraph( tk::UnsMesh& graph, HGRAPH_DATA& hg )
263 //******************************************************************************
264 // Create hypergraph data structure on MPI rank zero
265 //! \param[in] graph Unstructured mesh graph object reference
266 //! \param[inout] hg Hypergraph data structure to fill
267 //! \return The number of hyperedges in graph, the new->old mesh point map
268 //! \warning This function must not be called on MPI ranks other than zero.
269 //! \author J. Bakosi
270 //******************************************************************************
271 {
272  // Get number of points from graph.
273  const auto npoin = graph.size();
274 
275  // Create hypergraph data structure based on mesh graph
276  hg.numMyVertices = static_cast< int >( npoin );
277  hg.numMyHEdges = hg.numMyVertices;
278  hg.vtxGID = (ZOLTAN_ID_PTR)
279  malloc(sizeof(ZOLTAN_ID_TYPE) * static_cast<std::size_t>(hg.numMyVertices));
280  hg.edgeGID = (ZOLTAN_ID_PTR)
281  malloc(sizeof(ZOLTAN_ID_TYPE) * static_cast<std::size_t>(hg.numMyHEdges));
282  hg.nborIndex = (int*)
283  malloc(sizeof(int) * static_cast<std::size_t>(hg.numMyHEdges + 1));
284 
285  // Assign global point ids
286  for (std::size_t i=0; i<static_cast<std::size_t>(hg.numMyVertices); ++i)
287  hg.vtxGID[i] = static_cast< ZOLTAN_ID_TYPE >( i );
288 
289  // Get tetrahedron mesh graph connectivity
290  auto& inpoel = graph.tetinpoel();
291 
292  // Generate (connectivity graph) points surrounding points of graph
293  auto psup = tk::genPsup( inpoel, 4, tk::genEsup( inpoel, 4 ) );
294 
295  // Renumber mesh points for better data locality
296  std::vector< std::size_t > map, invmap;
297  std::tie( map, invmap ) = tk::renumber( psup );
298  // Remap element connectivity
299  tk::remap( inpoel, map );
300  // Re-generate points surrounding points
301  psup = tk::genPsup( inpoel, 4, tk::genEsup( inpoel, 4 ) );
302 
303  // Allocate data to store the hypergraph ids. The total number of vertices or
304  // neighbors in all the hyperedges of the hypergraph, nhedge = all points
305  // surrounding points + number of points, since psup does not store the
306  // connection to the own point, i.e., in matrix parlance, the main-diagonal.
307  // In other words, here we need the number of edges in the graph, independent
308  // of direction.
309  auto nhedge = psup.first.size() - 1 + npoin;
310  hg.numAllNbors = static_cast< int >( nhedge );
311  hg.nborGID = (ZOLTAN_ID_PTR)malloc(sizeof(ZOLTAN_ID_TYPE) * nhedge);
312 
313  // Fill up hypergraph edge ids and their indices
314  hg.nborIndex[0] = 0;
315  for (std::size_t p=0; p<npoin; ++p) {
316  hg.edgeGID[p] = static_cast< ZOLTAN_ID_TYPE >( p+1 );
317  // put in own point id, i.e., main diagonal
318  hg.nborGID[ hg.nborIndex[p] ] = static_cast< ZOLTAN_ID_TYPE >( p );
319  int j = 1;
320  // put in neighbor point ids, i.e., off-diagonals
321  for (auto i=psup.second[p]+1; i<=psup.second[p+1]; ++i, ++j) {
322  hg.nborGID[ hg.nborIndex[p] + j ] =
323  static_cast< ZOLTAN_ID_TYPE >( psup.first[i] );
324  }
325  hg.nborIndex[p+1] = hg.nborIndex[p] + j;
326  }
327 
328  return { nhedge, invmap };
329 }

366 partitionMesh( tk::UnsMesh& graph,
367  uint64_t npart,
368  const tk::Print& print )
369 //******************************************************************************
370 // Partition mesh graph using Zoltan's hypergraph algorithm in serial
371 //! \param[in] graph Unstructured mesh graph object reference
372 //! \param[in] npart Number of desired graph partitions
373 //! \param[in] print Pretty printer
374 //! \return Array of chare ownership IDs mapping graph points to concurrent
375 //! async chares, and new->old mesh point id map (new: renumbered, old: as in
376 //! mesh file)
377 //! \details This function uses Zoltan to partition the mesh graph in serial. It
378 //! assumes the mesh graph only exists on MPI rank 0.
379 //! \author J. Bakosi
380 //******************************************************************************
381 {
382  // Initialize the Zoltan library
383  float ver = 0.0;
384  ErrChkMPI( Zoltan_Initialize( 0, nullptr, &ver ) == ZOLTAN_OK,
385  "Zoltan could not be initialized" );
386 
387  // Create Zoltan data structure
388  struct Zoltan_Struct *zz;
389  zz = Zoltan_Create( MPI_COMM_WORLD );
390  AssertMPI( zz != nullptr, "Failed to create Zoltan data structure" );
391 
392  int peid;
393  MPI_Comm_rank( MPI_COMM_WORLD, &peid );
394 
395  // Set Zoltan parameters
396  Zoltan_Set_Param( zz, "DEBUG_LEVEL", "0" );
397  Zoltan_Set_Param( zz, "LB_METHOD", "HYPERGRAPH" );
398  Zoltan_Set_Param( zz, "LB_APPROACH", "PARTITION" );
399  Zoltan_Set_Param( zz, "HYPERGRAPH_PACKAGE", "PHG" );
400  Zoltan_Set_Param( zz, "NUM_GID_ENTRIES", "1" );
401  Zoltan_Set_Param( zz, "NUM_LID_ENTRIES", "1" );
402  Zoltan_Set_Param( zz, "OBJ_WEIGHT_DIM", "0" );
403  Zoltan_Set_Param( zz, "EDGE_WEIGHT_DIM", "0" );
404  Zoltan_Set_Param( zz, "RETURN_LISTS", "PART" );
405  //Zoltan_Set_Param( zz, "PHG_EDGE_SIZE_THRESHOLD", "1.0" );
406  //Zoltan_Set_Param( zz, "REMAP", "0" );
407  Zoltan_Set_Param( zz, "NUM_GLOBAL_PARTS", std::to_string(npart).c_str() );
408 
409  HGRAPH_DATA hg;
410  std::size_t nhedge = 0;
411  std::vector< std::size_t > invmap;
412  if (peid == 0)
413  std::tie( nhedge, invmap ) = createHyperGraph( graph, hg );
414  else
415  nhedge = emptyHyperGraph( hg );
416 
417  // Set Zoltan query functions
418  Zoltan_Set_Num_Obj_Fn( zz, get_number_of_vertices, &hg );
419  Zoltan_Set_Obj_List_Fn( zz, get_vertex_list, &hg );
420  Zoltan_Set_HG_Size_CS_Fn( zz, get_hypergraph_size, &hg );
421  Zoltan_Set_HG_CS_Fn( zz, get_hypergraph, &hg );
422 
423  // Perform partitioning using Zoltan
424  int changes, numGidEntries, numLidEntries, numImport, numExport;
425  ZOLTAN_ID_PTR importGlobalGids, importLocalGids, exportGlobalGids,
426  exportLocalGids;
427  int *importProcs, *importToPart, *exportProcs, *exportToPart;
428  int rc =
429  Zoltan_LB_Partition(
430  zz, // Input hypergraph data structure (remaining: output)
431  &changes, // 1 if partitioning was changed, 0 otherwise
432  &numGidEntries, // Number of integers used for a global ID
433  &numLidEntries, // Number of integers used for a local ID
434  &numImport, // Number of vertices to be sent to me
435  &importGlobalGids, // Global IDs of vertices to be sent to me
436  &importLocalGids, // Local IDs of vertices to be sent to me
437  &importProcs, // Process rank for source of each incoming vertex
438  &importToPart, // New partition for each incoming vertex
439  &numExport, // Number of vertices I must send to other processes
440  &exportGlobalGids, // Global IDs of the vertices I must send
441  &exportLocalGids, // Local IDs of the vertices I must send
442  &exportProcs, // Process to which I send each of the vertices
443  &exportToPart ); // Partition to which each vertex will belong
444 
445  // Destructor lambda
446  auto destruct = [&]() {
447  destroyHyperGraph( hg, nhedge ); // destroy hypergraph data structure
448  Zoltan_Destroy( &zz ); // destroy Zoltan data structure
449  };
450 
451  if (rc != ZOLTAN_OK) {
452  destruct();
453  Throw( "Zoltan_LB_Partition failed" );
454  }
455 
456  // Copy over array of chare IDs corresponding to the ownership of all points
457  // in the mesh graph, i.e., the coloring or chare ids for all mesh nodes
458  std::vector< std::size_t > chare;
459  for (int p=0; p<numExport; ++p )
460  chare.push_back( static_cast< std::size_t >( exportToPart[p] ) );
461 
462  std::size_t nchare = 1;
463  if (peid == 0) {
464  // Find out the number of partitions created by Zoltan
465  auto minmax = std::minmax_element( begin(chare), end(chare) );
466  nchare = *minmax.second - *minmax.first + 1;
467 
468  ErrChk( npart == nchare,
469  "The number of parts returned from the graph partitioner (" +
470  std::to_string(nchare) + ") does not equal the number of work units "
471  "computed (" + std::to_string(npart) + ") based on the degree of "
472  "virtualization desired. This is an indication of a too large "
473  "overdecomposition. Solution 1: decrease the virtualization to a lower "
474  "value using the command-line arrgument '-u'. Solution 2: decrease the "
475  "number processing elements (PEs) using the charmrun command-line "
476  "argument '+pN' where N is the number of PEs, which implicitly increases "
477  "the size (and thus decreases the number) of work units." );
478  }
479 
480 // if (peid == 0) {
481 // std::cout << "\nchp: ";
482 // for (auto i : chare) std::cout << i << " ";
483 // std::cout << '\n';
484 // }
485 
486  // Free the arrays allocated by Zoltan_LB_Partition
487  Zoltan_LB_Free_Part( &importGlobalGids, &importLocalGids, &importProcs,
488  &importToPart );
489  Zoltan_LB_Free_Part( &exportGlobalGids, &exportLocalGids, &exportProcs,
490  &exportToPart);
491 
492  // Free hypergraph and Zoltan data structure
493  destruct();
494 
495  return { chare, invmap };
496 }
