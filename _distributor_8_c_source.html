00233 //******************************************************************************
00234 // Wait for all integrators to finish accumulation of the ordinary moments
00235 //! \param[in] ord Partially accumulated ordinary moments contributed by caller
00236 //! \author J. Bakosi
00237 //******************************************************************************
00238 {
00239   // Increase number of integrators completing the accumulation of the ordinary
00240   // moments
00241   ++m_count.get< tag::ordinary >();
00242 
00243   // Add contribution from PE to total sums, i.e., u[i] += v[i] for all i
00244   for (std::size_t i=0; i<m_ordinary.size(); ++i) m_ordinary[i] += ord[i];
00245 
00246   // Wait for all integrators completing accumulation of ordinary moments
00247   if (m_count.get< tag::ordinary >() == m_count.get< tag::chare >()) {
00248     // Finish computing moments, i.e., divide sums by the number of samples
00249     for (auto& m : m_ordinary)
00250       m /= static_cast<tk::real>( g_inputdeck.get< tag::discr, tag::npar >() );
00251     // Activate SDAG trigger signaling that ordinary moments have been estimated
00252     estimateOrdDone();
00253   }

00583 void
00584 Distributor::evaluateTime()
00585 //******************************************************************************
00586 // Evaluate time step, compute new time step size, decide if it is time to quit
00587 //! \author J. Bakosi
00588 //******************************************************************************
00589 {
00590   // Increase number of iterations taken
00591   ++m_it;
00592 
00593   // Compute size of next time step
00594   const auto dt = computedt();
00595 
00596   // Advance physical time
00597   m_t += dt;
00598 
00599   // Get physical time at which to terminate
00600   const auto term = g_inputdeck.get< tag::discr, tag::term >();
00601 
00602   // Truncate the size of last time step
00603   if (m_t > term) m_t = term;
00604 
00605   // Echo one-liner info on time step
00606   report();
00607 
00608   // Finish if either max iterations or max time reached 
00609   if ( std::fabs(m_t - term) > std::numeric_limits< tk::real >::epsilon() &&
00610        m_it < g_inputdeck.get< tag::discr, tag::nstep >() ) {
00611 
00612     // Update map of statistical moments
00613     std::size_t ord = 0;
00614     std::size_t cen = 0;
00615     for (const auto& product : g_inputdeck.get< tag::stat >())
00616       if (tk::ctr::ordinary( product ))
00617         m_moments[ product ] = m_ordinary[ ord++ ];
00618       else
00619         m_moments[ product ] = m_central[ cen++ ];
00620 
00621     // Zero statistics counters and accumulators
00622     m_count.get< tag::ordinary >() = m_count.get< tag::central >() = 0;
00623     std::fill( begin(m_ordinary), end(m_ordinary), 0.0 );
00624     std::fill( begin(m_central), end(m_central), 0.0 );
00625 
00626     // Re-activate SDAG-wait for estimation of ordinary statistics for next step
00627     wait4ord();
00628     // Re-activate SDAG-wait for estimation of central moments for next step
00629     wait4cen();
00630 
00631     // Selectively re-activate SDAG-wait for estimation of PDFs for next step
00632     if ( !(m_it % g_inputdeck.get< tag::interval, tag::pdf >()) ) {
00633       // Zero PDF counters and accumulators
00634       m_count.get< tag::ordpdf >() = m_count.get< tag::cenpdf >() = 0;
00635       for (auto& p : m_ordupdf) p.zero();
00636       for (auto& p : m_ordbpdf) p.zero();
00637       for (auto& p : m_ordtpdf) p.zero();
00638       for (auto& p : m_cenupdf) p.zero();
00639       for (auto& p : m_cenbpdf) p.zero();
00640       for (auto& p : m_centpdf) p.zero();
00641       wait4pdf();
00642     }
00643 
00644     // Continue with next time step with all integrators
00645     for (auto& p : m_proxy) p.advance( dt, m_it, m_moments );
00646 
00647   } else {
00648 
00649     // Normal finish, print out reason
00650     const auto term = g_inputdeck.get< tag::discr, tag::term >();
00651     const auto nstep = g_inputdeck.get< tag::discr, tag::nstep >();
00652     m_print.endsubsection();
00653     if (m_it >= g_inputdeck.get< tag::discr, tag::nstep >())
00654        m_print.note( "Normal finish, maximum number of iterations reached: " +
00655                      std::to_string( nstep ) );
00656      else 
00657        m_print.note( "Normal finish, maximum time reached: " +
00658                      std::to_string( term ) );
00659     // Quit
00660     mainProxy.finalize();
00661 
00662   }
00663 }
