00130 void
00131 Exception::echoTrace() noexcept
00132 //******************************************************************************
00133 //  Demangle and echo call trace
00134 //! \details Exception safety: no-throw guarantee: this member function never
00135 //!   throws exceptions. Credit goes to Timo Bingmann at http://idlebox.net,
00136 //!   published under the WTFPL v2.0. For more information see
00137 //!   * http://panthema.net/2008/0901-stacktrace-demangled
00138 //!   * http://panthema.net/2008/0901-stacktrace-demangled/cxa_demangle.html
00139 //!   * http://gcc.gnu.org/onlinedocs/libstdc++/latest-doxygen
00140 //! \author T. Bingmann, J. Bakosi
00141 //******************************************************************************
00142 {
00143   // Allocate string which will be filled with the demangled function name
00144   size_t funcnamesize = 256;
00145   char* funcname = (char*)malloc(funcnamesize);
00146 
00147   // Iterate over the returned symbol lines. skip the first two, these are the
00148   // addresses of Exception::saveTrace() and the Exception constructor
00149   for (int i=2; i<m_addrLength; ++i) {
00150     char *begin_name = 0, *begin_offset = 0, *end_offset = 0;
00151 
00152     // Find parentheses and +address offset surrounding the mangled name:
00153     // ./module(function+0x15c) [0x8048a6d]
00154     for (char *p = m_symbolList[i]; *p; ++p) {
00155       if (*p == '(')
00156         begin_name = p;
00157       else if (*p == '+')
00158         begin_offset = p;
00159       else if (*p == ')' && begin_offset) {
00160         end_offset = p;
00161         break;
00162       }
00163     }
00164 
00165     if (begin_name && begin_offset && end_offset && begin_name < begin_offset) {
00166       *begin_name++ = '\0';
00167       *begin_offset++ = '\0';
00168       *end_offset = '\0';
00169 
00170       // Mangled name is now in [begin_name, begin_offset) and caller
00171       // offset in [begin_offset, end_offset). now apply __cxa_demangle()
00172       int status;
00173       char* ret = abi::__cxa_demangle(begin_name,
00174                                       funcname, &funcnamesize, &status);
00175 
00176       if (status == 0) {
00177         funcname = ret; // use possibly realloc()-ed string
00178         printf(">>> %s : %s+%s\n", m_symbolList[i], funcname, begin_offset);
00179       } else {
00180         // Demangling failed. Output function name as a C function with no
00181         // arguments
00182         printf(">>> %s : %s()+%s\n", m_symbolList[i], begin_name, begin_offset);
00183       }
00184     } else {
00185       // Couldn't parse the line? Print the whole line
00186       printf(">>> %s\n", m_symbolList[i]);
00187     }
00188   }
00189 
00190   free(funcname);
00191 }
