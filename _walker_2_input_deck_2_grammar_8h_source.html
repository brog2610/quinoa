00104   //! \brief Do general error checking on the differential equation block
00105   //! \details This is error checking that all equation types must satisfy.
00106   //! \author J. Bakosi
00107   template< class eq >
00108   struct check_eq : pegtl::action_base< check_eq< eq > > {
00109     static void apply( const std::string& value, Stack& stack ) {
00110 
00111       // Error out if no dependent variable has been selected
00112       const auto& depvar = stack.get< tag::param, eq, tag::depvar >();
00113       if (depvar.empty() || depvar.size() != neq.get< eq >())
00114         tk::grm::Message< Stack, tk::grm::ERROR, tk::grm::MsgKey::NODEPVAR >
00115                         ( stack, value );
00116 
00117       // Error out if no number of components has been selected
00118       const auto& ncomp = stack.get< tag::component, eq >();
00119       if (ncomp.empty() || ncomp.size() != neq.get< eq >())
00120         tk::grm::Message< Stack, tk::grm::ERROR, tk::grm::MsgKey::NONCOMP >
00121                         ( stack, value );
00122 
00123       // Error out if no RNG has been selected
00124       const auto& rng = stack.get< tag::param, eq, tag::rng >();
00125       if (rng.empty() || rng.size() != neq.get< eq >())
00126         tk::grm::Message< Stack, tk::grm::ERROR, tk::grm::MsgKey::NORNG >
00127                         ( stack, value );
00128 
00129       // Error out if no initialization policy has been selected
00130       const auto& init = stack.get< tag::param, eq, tag::initpolicy >();
00131       if (init.empty() || init.size() != neq.get< eq >())
00132         tk::grm::Message< Stack, tk::grm::ERROR, tk::grm::MsgKey::NOINIT >
00133                         ( stack, value );
00134 
00135       // Error out if no coefficients policy has been selected
00136       const auto& coeff = stack.get< tag::param, eq, tag::coeffpolicy >();
00137       if (coeff.empty() || coeff.size() != neq.get< eq >())
00138         tk::grm::Message< Stack, tk::grm::ERROR, tk::grm::MsgKey::NOCOEFF >
00139                         ( stack, value );
00140     }
00141   };

00143   //! \brief Do error checking on the selected initialization policy
00144   //! \author J. Bakosi
00145   template< class eq >
00146   struct check_init : pegtl::action_base< check_init< eq > > {
00147     static void apply( const std::string& value, Stack& stack ) {
00148       const auto& init = stack.get< tag::param, eq, tag::initpolicy >();
00149       // Error checks for joint delta initpolicy
00150       if (init.size() == neq.get< eq >() &&
00151           init.back() == ctr::InitPolicyType::JOINTDELTA) {
00152         // Make sure there was an icdelta...end block with at least a single
00153         // spike...end block
00154         const auto& spike = stack.template get< tag::param, eq, tag::spike >();
00155         if (!spike.empty() && spike.back().empty())
00156           tk::grm::Message< Stack, tk::grm::ERROR, tk::grm::MsgKey::NODELTA >
00157                           ( stack, value );
00158       }
00159       // Error checks for joint beta initpolicy
00160       if (init.size() == neq.get< eq >() &&
00161           init.back() == ctr::InitPolicyType::JOINTBETA) {
00162         // Make sure there was an icbeta...end block with at least a single
00163         // betapdf...end block
00164         const auto& betapdf =
00165           stack.template get< tag::param, eq, tag::betapdf >();
00166         if (!betapdf.empty() && betapdf.back().empty())
00167           tk::grm::Message< Stack, tk::grm::ERROR, tk::grm::MsgKey::NOBETA >
00168                           ( stack, value );
00169       }
00170 
00171     }
00172   };
