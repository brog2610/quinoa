50  //! \brief Add a statistical test to battery
51  //! \details Note that adding a test to the battery does not invoke the test
52  //! constructor, it only records the information on how to call the test
53  //! constructor in the future. That is it binds the constructor arguments
54  //! to the constructor call and records the the information so only a
55  //! function call "()" is necessary to instantiate it.
56  //! \param[in] proxy Charm++ host proxy to which the test calls back to
57  //! \param[in] tests Vector of test constructors to add tests to
58  //! \param[in] r RNG ID enum
59  //! \param[in] gen Raw function pointer to TestU01 statistical test
60  //! \param[in] names Vector of statisical test names (can be more than one
61  //! associated with a given test, since a test can contain more than one
62  //! statistical test evaluation, yielding multiple p-values)
63  //! \param[in] xargs Extra arguments to test-run
64  template< class TestType, class Proxy, typename... Ts >
65  void add( Proxy& proxy,
66  std::vector< std::function< StatTest() > >& tests,
67  tk::ctr::RNGType r,
68  unif01_Gen* const gen,
69  std::vector< std::string >&& names,
70  Ts&&... xargs ) const
71  {
72  using Model = TestType;
73  using Host = StatTest;
74  using Props = typename TestType::Props;
75  tests.emplace_back(
76  std::bind( boost::value_factory< Host >(),
77  std::function< Model() >(),
78  std::forward< Props >(
79  Props( proxy, r, std::move(names), gen,
80  std::forward<Ts>(xargs)... ) ) ) );
81  }
