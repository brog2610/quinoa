00050     //! \brief Add a statistical test to battery
00051     //! \details Note that adding a test to the battery does not invoke the test
00052     //!   constructor, it only records the information on how to call the test
00053     //!   constructor in the future. That is it binds the constructor arguments
00054     //!   to the constructor call and records the the information so only a
00055     //!   function call "()" is necessary to instantiate it.
00056     //! \param[in] proxy Charm++ host proxy to which the test calls back to
00057     //! \param[in] tests Vector of test constructors to add tests to
00058     //! \param[in] r RNG ID enum
00059     //! \param[in] gen Raw function pointer to TestU01 statistical test
00060     //! \param[in] names Vector of statisical test names (can be more than one
00061     //!   associated with a given test, since a test can contain more than one
00062     //!   statistical test evaluation, yielding multiple p-values)
00063     //! \param[in] xargs Extra arguments to test-run
00064     template< class TestType, class Proxy, typename... Ts >
00065     void add( Proxy& proxy,
00066               std::vector< std::function< StatTest() > >& tests,
00067               tk::ctr::RNGType r,
00068               unif01_Gen* const gen,
00069               std::vector< std::string >&& names,
00070               Ts&&... xargs ) const
00071     {
00072       using Model = TestType;
00073       using Host = StatTest;
00074       using Props = typename TestType::Props;
00075       tests.emplace_back(
00076         std::bind( boost::value_factory< Host >(),
00077                    std::function< Model() >(),
00078                    std::forward< Props >(
00079                      Props( proxy, r, std::move(names), gen,
00080                             std::forward<Ts>(xargs)... ) ) ) );
00081     }
