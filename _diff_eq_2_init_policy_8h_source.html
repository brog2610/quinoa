00111     using ncomp_t = kw::ncomp::info::expect::type;
00112 
00113     const auto& spike = deck.template get< tag::param, eq, tag::spike >().at(e);
00114 
00115     // use only the first ncomp spikes if there are more than the equation is
00116     // configured for
00117     const ncomp_t size = std::min( ncomp, spike.size() );
00118 
00119     for (ncomp_t c=0; c<size; ++c) {
00120       const auto& sc = spike[c];        // vector of spikes for component c
00121 
00122       ncomp_t i = 0;
00123       for (ncomp_t s=0; s<sc.size(); s+=2) {
00124         // compute number of samples to be set at relative probability height
00125         const auto npar =
00126           static_cast< ncomp_t >(
00127             static_cast< tk::real >( particles.npar() ) * sc[s+1] );
00128         // assign sample values

00130         i += npar;
00131       }
