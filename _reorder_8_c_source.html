00022 shiftToZero( std::vector< std::size_t >& inpoel )
00023 //******************************************************************************
00024 //  Shift node IDs to start with zero in element connectivity
00025 //! \param[inout] inpoel Inteconnectivity of points and elements
00026 //! \details This function implements a simple reordering of the node ids of the
00027 //!   element connectivity in inpoel by shifting the node ids so that the
00028 //!   smallest is zero.
00029 //! \note It is okay to call this function with an empty container; it will
00030 //!    simply return without throwing an exception.
00031 //! \author J. Bakosi
00032 //******************************************************************************
00033 {
00034   if (inpoel.empty()) return;
00035 
00036   // find smallest node id
00037   auto minId = *std::min_element( begin(inpoel), end(inpoel) );
00038 
00039   // shift node ids to start from zero
00040   for (auto& n : inpoel) n -= minId;
00041 }

00044 remap( std::vector< std::size_t >& id, const std::vector< std::size_t >& newid )
00045 //******************************************************************************
00046 //  Reorder mesh points given a new order, i.e., index map
00047 //! \param[inout] id Vector of point ids
00048 //! \param[in] newid Array of indices creating a new order
00049 //! \details This function implements a simple reordering (or remapping) of the
00050 //!   node ids of the vector passed in using the vector newid. Thus the vector
00051 //!   in newid is thought of as a mapping between the array index to its value.
00052 //!   The function overwrites every value, n, of vector id with newid[n].
00053 //! \note It is okay to call this function with either of the containers empty;
00054 //!   it will simply return without throwing an exception.
00055 //! \author J. Bakosi
00056 //******************************************************************************
00057 {
00058   if (id.empty() || newid.empty()) return;
00059 
00060   Assert( *max_element( begin(id), end(id) ) < newid.size(),
00061           "attempt to index out of node id bounds using newid" );
00062 
00063   // remap node ids in vector id
00064   for (auto& n : id) n = newid[n];
00065 }

00083   hpoin[0] = 0;

00086   while (num < npoin) {

00089     std::vector< int > kpoin( npoin, -1 );

00092       ++i;

00094       for (auto j=psup.second[P]+1; j<=psup.second[P+1]; ++j) {
00095         auto q = psup.first[j];
00096         if (lpoin[q] != 1) {    // consider points not yet counted
00097           mapvec[q] = num++;
00098           kpoin[cnt] = static_cast< int >( q ); // register point as counted
00099           lpoin[q] = 1;                         // register the point as counted
00100           ++cnt;
