49 //! Integrator Charm++ chare used to advance differential equations in time
50 template< class Proxy >
51 class Integrator : public CBase_Integrator< Proxy > {
52 
53  public:
54  //! Constructor
55  //! \param[in] proxy Host proxy to call back to (here: Distributor)
56  //! \param[in] npar Number of particles this integrator advances
57  //! \param[in] dt Size of time step
58  //! \param[in] it Iteration count
59  //! \param[in] moments Map of statistical moments
60  explicit Integrator( Proxy& proxy,
61  uint64_t npar,
62  tk::real dt,
63  uint64_t it,
64  const std::map< tk::ctr::Product, tk::real >& moments )
65  : m_proxy( proxy ),
66  m_particles( npar, g_inputdeck.get< tag::component >().nprop() ),
67  m_stat( m_particles,
68  g_inputdeck.get< tag::component >().offsetmap(
69  g_inputdeck.depvars() ),
70  g_inputdeck.get< tag::stat >(),
71  g_inputdeck.get< tag::pdf >(),
72  g_inputdeck.get< tag::discr, tag::binsize >() )
73  {
74  ic(); // set initial conditions for all equations
75  advance( dt, it, moments ); // start time stepping all equations
76  }
77 
78  //! Set initial conditions
79  void ic() {
80  for (const auto& eq : g_diffeqs) eq.initialize( CkMyPe(), m_particles );
81  m_proxy.init(); // signal to host that initialization is complete
82  }
83 
84  //! Advance all particles owned by this integrator
85  //! \param[in] dt Size of time step
86  //! \param[in] it Iteration count
87  //! \param[in] moments Map of statistical moments
88  void advance( tk::real dt,
89  uint64_t it,
90  const std::map< tk::ctr::Product, tk::real >& moments )
91  {
92  //! Advance all equations one step in time
93  if (it < g_inputdeck.get< tag::discr, tag::nstep >())
94  for (const auto& e : g_diffeqs)
95  e.advance( m_particles, CkMyPe(), dt, moments );
96  // Accumulate sums for ordinary moments (every time step)
97  accumulateOrd();
98  // Accumulate sums for ordinary PDFs at select times
99  if ( !(it % g_inputdeck.get< tag::interval, tag::pdf >()) )
100  accumulateOrdPDF();
101  }
102 
103  // Accumulate sums for ordinary moments
104  void accumulateOrd() {
105  // Accumulate partial sums for ordinary moments
106  m_stat.accumulateOrd();
107  // Send accumulated ordinary moments to host for estimation
108  m_proxy.estimateOrd( m_stat.ord() );
109  }
110 
111  // Accumulate sums for central moments
112  //! \param[in] ord Estimated ordinary moments (collected from all PEs)
113  void accumulateCen( const std::vector< tk::real >& ord ) {
114  // Accumulate partial sums for central moments
115  m_stat.accumulateCen( ord );
116  // Send accumulated central moments to host for estimation
117  m_proxy.estimateCen( m_stat.ctr() );
118  }
119 
120  // Accumulate sums for ordinary PDFs
121  void accumulateOrdPDF() {
122  // Accumulate partial sums for ordinary PDFs
123  m_stat.accumulateOrdPDF();
124  // Send accumulated ordinary PDFs to host for estimation
125  m_proxy.estimateOrdPDF( m_stat.oupdf(), m_stat.obpdf(), m_stat.otpdf() );
126  }
127 
128  // Accumulate sums for central PDFs
129  //! \param[in] ord Estimated ordinary moments (collected from all PEs)
130  void accumulateCenPDF( const std::vector< tk::real >& ord ) {
131  // Accumulate partial sums for central PDFs
132  m_stat.accumulateCenPDF( ord );

134  m_proxy.estimateCenPDF( m_stat.cupdf(), m_stat.cbpdf(), m_stat.ctpdf() );
135  }
136

141 };
