00365 //! semantics) with Charm++ children. Not in Factory_common so Charm++ can find
00366 //! it. The struct below uses runtime polymorphism without client-side

00368 //! inivisble to client-code. The struct exclusively deals with ownership

00370 //! https://github.com/sean-parent/sean-parent.github.com/wiki/
00371 //! Papers-and-Presentations
00372 struct VBase {
00373   //! Constructor taking a function pointer to a constructor of an object
00374   //! modeling Concept (see below). Passing std::function allows late
00375   //! execution of the constructor of T, i.e., at some future time, and thus
00376   //! usage from a factory. Note that the value of the first function
00377   //! argument, std::function<T()>, is not used here, but its constructor
00378   //! type, T, is used to enable the compiler to deduce the model constructor
00379   //! type, used to create its Charm proxy, defined by T::Proxy. The actual
00380   //! constructor of T is not called here but at some future time by the
00381   //! Charm++ runtime system, here only an asynchrounous ckNew() is called,
00382   //! i.e., a message (or request) for a future call to T's constructor. This
00383   //! overload is only enabled for Charm++ chare objects defining typedef
00384   //! 'Proxy', which must define the Charm++ proxy. All optional constructor
00385   //! arguments are forwarded to ckNew() and thus to T's constructor. If
00386   //! it was somehow possible to obtain all bound arguments' types and values
00387   //! from an already-bound std::function, we could use those instead of
00388   //! having to explicitly forward the model constructor arguments via this
00389   //! host constructor. See also tk::recordCharmModel().
00390   template< typename T, typename... ConstrArgs,
00391     typename std::enable_if< tk::HasTypedefProxy<T>::value, int >::type = 0 >
00392   explicit VBase( std::function<T()> c, ConstrArgs... args ) :
00393     self( tk::make_unique< Model< typename T::Proxy > >
00394          (std::move(T::Proxy::ckNew(std::forward<ConstrArgs>(args)...))) ) {
00395     Assert( c == nullptr, "std::function argument to VBase Charm "
00396                           "constructor must be nullptr" );
00397   }
00398   //! Copy assignment
00399   VBase& operator=( const VBase& x )
00400   { VBase tmp(x); *this = std::move(tmp); return *this; }
00401   //! Copy constructor
00402   VBase( const VBase& x ) : self( x.self->copy() ) {}
00403   //! Move assignment
00404   VBase& operator=( VBase&& ) noexcept = default;
00405   //! Move constructor
00406   VBase( VBase&& ) noexcept = default;
00407   //! Concept is a pure virtual base struct specifying the requirements of
00408   //! polymorphic objects deriving from it
00409   struct Concept {
00410     virtual ~Concept() = default;
00411     virtual Concept* copy() const = 0;
00412   };

00414   //! the virtual functions required by Concept
00415   template< typename T >
00416   struct Model : Concept {
00417     Model( T x ) : data( std::move(x) ) {}
00418     Concept* copy() const { return new Model( *this ); }
00419     T data;
00420   };
