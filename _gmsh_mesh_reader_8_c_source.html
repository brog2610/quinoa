00029 void
00030 GmshMeshReader::readMesh( UnsMesh& mesh )
00031 //******************************************************************************
00032 //  Public interface for read a Gmsh mesh from file
00033 //! \param[in] mesh Unstructured mesh object
00034 //! \author J. Bakosi
00035 //******************************************************************************
00036 {
00037   // Read in mandatory "$MeshFormat" section
00038   readMeshFormat();
00039 
00040   // Keep reading in sections until end of file. These sections can be in
00041   // arbitrary order, hence a while loop.
00042   while ( !m_inFile.eof() ) {
00043     std::string s;
00044     getline( m_inFile, s );
00045     if ( s == "$Nodes" )
00046       readNodes( mesh );
00047     else if ( s == "$Elements" )
00048       readElements( mesh );
00049     else if ( s == "$PhysicalNames" )
00050       readPhysicalNames();
00051   }
00052 
00053   // Clear failbit triggered by eof, so close() won't throw a false FAILED_CLOSE
00054   m_inFile.clear();
00055 }

00059 //******************************************************************************
00060 //  Read mandatory "$MeshFormat--$EndMeshFormat" section
00061 //! \author J. Bakosi
00062 //******************************************************************************
00063 {
00064   using tk::operator<<;
00065   std::string s;
00066 
00067   // Read in beginning of header: $MeshFormat
00068   getline( m_inFile, s );
00069   ErrChk( s == "$MeshFormat",
00070           std::string("Unsupported mesh format '") + s + "' in file " +
00071           m_filename );
00072 
00073   // Read in "version-number file-type data-size"
00074   int type;
00075   m_inFile >> m_version >> type >> m_datasize;
00076   if (type == 0 )
00077     m_type = GmshFileType::ASCII;
00078   else if (type == 1 )
00079     m_type = GmshFileType::BINARY;
00080 
00081   ErrChk( ( fabs(m_version-2.2) < std::numeric_limits<tk::real>::epsilon() ||
00082             fabs(m_version-2.0) < std::numeric_limits<tk::real>::epsilon() ),
00083             std::string("Unsupported mesh version '") << m_version <<
00084             "' in file " << m_filename );
00085 
00086   ErrChk( ( m_type == GmshFileType::ASCII || m_type == GmshFileType::BINARY ),
00087             std::string("Unsupported mesh type '") << m_type << "' in file " <<
00088             m_filename );
00089 
00090   ErrChk( m_datasize == sizeof(tk::real),
00091           std::string("Unsupported mesh datasize '") << m_datasize <<
00092           "' in file " << m_filename );
00093 
00094   getline( m_inFile, s );  // finish reading the line
00095 
00096   // if file is binary, binary-read in binary "one"
00097   if ( isBinary() ) {
00098     int one;
00099     m_inFile.read( reinterpret_cast<char*>(&one), sizeof(int) );
00100     ErrChk( one == 1, "Endianness does not match in file " + m_filename );
00101     getline( m_inFile, s );  // finish reading the line
00102   }
00103 
00104   // Read in end of header: $EndMeshFormat
00105   getline( m_inFile, s );
00106   ErrChk( s == "$EndMeshFormat",
00107           "'$EndMeshFormat' keyword is missing in file " + m_filename );
00108 }

00113 //  Read "$Nodes--$EndNodes" section
00114 //! \param[in] mesh Unstructured mesh object
00115 //! \author J. Bakosi
00116 //******************************************************************************
00117 {
00118   // Read in number of nodes in this node set
00119   std::size_t nnode;
00120   m_inFile >> nnode;
00121   ErrChk( nnode > 0,
00122           "Number of nodes must be greater than zero in file " + m_filename  );
00123   std::string s;
00124   if (isBinary()) getline( m_inFile, s );  // finish reading the line
00125 
00126   // Read in node ids and coordinates: node-number x-coord y-coord z-coord
00127   for ( std::size_t i=0; i<nnode; ++i ) {
00128     int id;
00129     std::array< tk::real, 3 > coord;
00130 
00131     if (isASCII()) {
00132       m_inFile >> id >> coord[0] >> coord[1] >> coord[2];
00133     } else {
00134       m_inFile.read( reinterpret_cast<char*>(&id), sizeof(int) );
00135       m_inFile.read( reinterpret_cast<char*>(coord.data()), 3*sizeof(double) );
00136     }
00137 
00138     mesh.x().push_back( coord[0] );
00139     mesh.y().push_back( coord[1] );
00140     mesh.z().push_back( coord[2] );
00141   }
00142   getline( m_inFile, s );  // finish reading the last line
00143 
00144   // Read in end of header: $EndNodes
