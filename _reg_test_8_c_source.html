00062 //! \brief Charm++ main chare for the regression test suite executable, regtest.
00063 //! \details Note that this object should not be in a namespace.
00064 class Main : public CBase_Main {
00065 
00066   public:
00067     //! \brief Constructor
00068     //! \details The main chare constructor is the main entry point of the
00069     //!   program, called by the Charm++ runtime system. The constructor does
00070     //!   basic initialization steps, e.g., parser the command-line, prints out
00071     //!   some useful information to screen (in verbose mode), and instantiates
00072     //!   a driver. Since Charm++ is fully asynchronous, the constructor
00073     //!   usually spawns asynchronous objects and immediately exits. Thus in the
00074     //!   body of the main chare constructor we fire up an 'execute' chare,
00075     //!   which then calls back to Main::execute(). Finishing the main chare
00076     //!   constructor the Charm++ runtime system then starts the
00077     //!   network-migration of all global-scope data (if any). The execute chare
00078     //!   calling back to Main::execute() signals the end of the migration of
00079     //!   the global-scope data. Then we are ready to execute the driver. Since
00080     //!   the regression test suite is parallel and asynchronous, its driver
00081     //!   fires up additional Charm++ chare objects which then call back to
00082     //!   Main::finalize() at some point in the future when all work has been
00083     //!   finished. finalize() then exits by calling Charm++'s CkExit(),
00084     //!   shutting down the runtime system.
00085     //! \see http://charm.cs.illinois.edu/manuals/html/charm++/manual.html
00086     Main( CkArgMsg* msg )
00087     try :
00088       // Parse command line into m_cmdline using default simple pretty printer
00089       m_cmdParser( msg->argc, msg->argv, tk::Print(), m_cmdline ),
00090       // Create pretty printer initializing output streams based on command line
00091       m_print( m_cmdline.get< tag::verbose >() ? std::cout : std::clog ),
00092       // Create RegTest driver
00093       m_driver( tk::Main< regtest::RegTestDriver >
00094                         ( msg->argc, msg->argv,
00095                           m_cmdline,
00096                           tk::HeaderType::REGTEST,
00097                           REGTEST_EXECUTABLE,
00098                           m_print ) ),
00099       m_timer(1)  // Start new timer measuring the serial+Charm++ runtime
00100     {
00101       delete msg;
00102       mainProxy = thisProxy;
00103       // Fire up an asynchronous execute object, which when created at some
00104       // future point in time will call back to this->execute(). This is
00105       // necessary so that this->execute() can access already migrated
00106       // global-scope data.
00107       CProxy_execute::ckNew();
00108       // Start new timer measuring the migration of global-scope data
00109       m_timer.emplace_back();
00110     } catch (...) { tk::processExceptionCharm(); }
