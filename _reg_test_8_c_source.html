62 //! \brief Charm++ main chare for the regression test suite executable, regtest.
63 //! \details Note that this object should not be in a namespace.
64 class Main : public CBase_Main {
65 
66  public:
67  //! \brief Constructor
68  //! \details The main chare constructor is the main entry point of the
69  //! program, called by the Charm++ runtime system. The constructor does
70  //! basic initialization steps, e.g., parser the command-line, prints out
71  //! some useful information to screen (in verbose mode), and instantiates
72  //! a driver. Since Charm++ is fully asynchronous, the constructor
73  //! usually spawns asynchronous objects and immediately exits. Thus in the
74  //! body of the main chare constructor we fire up an 'execute' chare,
75  //! which then calls back to Main::execute(). Finishing the main chare
76  //! constructor the Charm++ runtime system then starts the
77  //! network-migration of all global-scope data (if any). The execute chare
78  //! calling back to Main::execute() signals the end of the migration of
79  //! the global-scope data. Then we are ready to execute the driver. Since
80  //! the regression test suite is parallel and asynchronous, its driver
81  //! fires up additional Charm++ chare objects which then call back to
82  //! Main::finalize() at some point in the future when all work has been
83  //! finished. finalize() then exits by calling Charm++'s CkExit(),
84  //! shutting down the runtime system.
85  //! \see http://charm.cs.illinois.edu/manuals/html/charm++/manual.html
86  Main( CkArgMsg* msg )
87  try :
88  // Parse command line into m_cmdline using default simple pretty printer
89  m_cmdParser( msg->argc, msg->argv, tk::Print(), m_cmdline ),
90  // Create pretty printer initializing output streams based on command line
91  m_print( m_cmdline.get< tag::verbose >() ? std::cout : std::clog ),
92  // Create RegTest driver
93  m_driver( tk::Main< regtest::RegTestDriver >
94  ( msg->argc, msg->argv,
95  m_cmdline,
96  tk::HeaderType::REGTEST,
97  REGTEST_EXECUTABLE,
98  m_print ) ),
99  m_timer(1) // Start new timer measuring the serial+Charm++ runtime
100  {
101  delete msg;
102  mainProxy = thisProxy;
103  // Fire up an asynchronous execute object, which when created at some
104  // future point in time will call back to this->execute(). This is
105  // necessary so that this->execute() can access already migrated
106  // global-scope data.
107  CProxy_execute::ckNew();
108  // Start new timer measuring the migration of global-scope data
109  m_timer.emplace_back();
110  } catch (...) { tk::processExceptionCharm(); }
111

115  "Migration of global-scope data + fire up all tests",
116  m_timer[1].hms() );
