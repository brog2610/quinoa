00051   // Underline errors and warnings
00052   std::string underline( m_string.size(), ' ' );
00053   for (const auto& e : messages) {
00054 
00055     // decide if error or warning
00056     char underchar = ' ';
00057     if (e.find( "Error" ) != std::string::npos) { err = true; underchar = '^'; }
00058     else if (e.find( "Warning" ) != std::string::npos) underchar = '~';
00059 
00060     // underline error and warning
00061     if (underchar == '^' || underchar == '~') {
00062       auto sloc = e.find( "at 1," );
00063       if (sloc != std::string::npos) {  // if we have location info
00064         // skip "at 1,"
00065         sloc += 5;
00066         // find a dot starting from after "at 1,"
00067         const auto eloc = e.find_first_of( '.', sloc );
00068         // compute location of error by extracting it from error message
00069         const std::size_t errloc = std::stoul( e.substr( sloc, eloc-sloc ) ) - 1;
00070         // find beginning of erroneous argument
00071         sloc = m_string.rfind( ' ', errloc-1 );
00072         // special-handle the first argument with no space in front of it
00073         if (sloc == std::string::npos) sloc = 0; else ++sloc;
00074         // underline error and warning differently
00075         for (std::size_t i=sloc; i<errloc; ++i) underline[i] = underchar;
00076       }
00077     }
00078   }
00079

00081   // and warnings, and exit
