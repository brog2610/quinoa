26  //! \brief Constructor
27  //! \param[in] nthreads Initialize RNG using this many independent streams
28  //! \param[in] brng Index of the basic generator to initialize the stream
29  //! \param[in] seed RNG seed
30  //! \param[in] uniform_method MKL ID of the method to use for uniform RNGs
31  //! \param[in] gaussian_method MKL ID of the method to use for Gaussian RNGs
32  //! \param[in] beta_method MKL ID of the method to use for beta RNGs
33  explicit MKLRNG( int nthreads,
34  int brng,
35  unsigned int seed,
36  int uniform_method,
37  int gaussian_method,
38  int beta_method ) :
39  m_brng( brng ),
40  m_seed( seed ),
41  m_uniform_method( uniform_method ),
42  m_gaussian_method( gaussian_method ),
43  m_beta_method( beta_method ),
44  m_nthreads( nthreads )
45  {
46  Assert( nthreads > 0, "Need at least one thread" );
47  // Allocate array of stream-pointers for threads
48  m_stream = tk::make_unique< VSLStreamStatePtr[] >(
49  static_cast<std::size_t>(nthreads) );
50  // Initialize thread-streams for block-splitting. These MKL VSL functions
51  // dynamically allocate memory, so these calls being in a constructor are
52  // a potential memory leak hazard in the presence of exceptions. However,
53  // thankfully, the MKL functions below only emit warnings if they
54  // encounter errors and always continue. As a result, the constructor
55  // finishes, the MKLRNG object gets created, so the destructor will also
56  // get called when leaving scope.
57  for (int i=0; i<nthreads; ++i) {
58  auto I = static_cast< std::size_t >( i );
59  vslNewStream( &m_stream[ I ], brng, seed );
60  vslLeapfrogStream( m_stream[ I ], i, nthreads );
61  }
62  }
63
