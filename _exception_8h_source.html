00026 //! \brief Throw macro that always throws an exception
00027 //! \details Throw Exception with arguments passed in. Add source filename,
00028 //!   function name, and line number where exception occurred. This macro
00029 //!   facilitates a throw of Exception that is somehwat cleaner at the point
00030 //!   of invocation than a direct throw of Exception, as it hides the
00031 //!   file:func:line arguments. Whenever is possible, it should be used via the
00032 //!   Assert and ErrChk macros defined below.
00033 //! \author J. Bakosi
00034 #define Throw(...) \
00035    throw tk::Exception(__VA_ARGS__, __FILE__, __PRETTY_FUNCTION__, __LINE__)
00036 
00037 //! \brief Assert macro that only throws an exception if expr fails.
00038 //! \details If NDEBUG is defined (e.g. cmake's RELEASE or OPTIMIZED mode), do
00039 //!    nothing, expr is not evaluated. If NDEBUG is not defined, evaluate expr.
00040 //!    If expr is true, do nothing. If expr is false, throw Exception with
00041 //!    arguments passed in. The behavior is similar to libc's assert macro, but
00042 //!    throwing an Exception instead will also generate a nice call-trace and
00043 //!    will attempt to free memory. This macro should be used to detect
00044 //!    programmer errors.
00045 //! \author J. Bakosi
00046 #ifdef NDEBUG
00047 #  define Assert(expr, ...) (static_cast<void>(0))
00048 #else  // NDEBUG
00049 #  define Assert(expr, ...) \
00050    ((expr) ? static_cast<void>(0) : Throw(__VA_ARGS__))
00051 #endif // NDEBUG
00052 
00053 //! \brief ErrChk macro that only throws an exception if expr fails.
00054 //! \details The behavior of this macro is the same whether NDEBUG is defined or
00055 //!    not: expr is always evaluated. If expr is true, do nothing. If expr is
00056 //!    false, throw Exception with arguments passed in. This macro should be
00057 //!    used to detect user or runtime errors.
00058 //! \author J. Bakosi
00059 #define ErrChk(expr, ...) \
00060    ((expr) ? static_cast<void>(0) : Throw(__VA_ARGS__))
