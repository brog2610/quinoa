00238 //******************************************************************************
00239 //! Construct array of chare ownership IDs mapping mesh elements to chares
00240 //! \param[in] nchare Number chares points are partitioned into
00241 //! \param[in] chp Chares of points: array of chare ownership IDs mapping graph
00242 //!   points to Charm++ chares. Size: number of points in mesh graph.
00243 //! \param[inout] tetinpoel Mesh tetrahedron element connectivity
00244 //! \param[inout] esup Derived data structure, storing elements surrounding
00245 //!   points in mesh
00246 //! \details This function constructs the vector che, 'chares of elements'. This
00247 //!   is the equivalent of chp, 'chares of points', but stores the chare ids of
00248 //!   mesh elements. The element ownership is computed based on the point
00249 //!   ownership. If all points of an element are owned by a chare, then that
00250 //!   chare owns the element. If not all points of an element are owned by the
00251 //!   a chare, than the chare with lower chare id gets to own the element.
00252 //! \author J. Bakosi
00253 //******************************************************************************
00254 {
00255   // Lambda to find out if all 4 points of tetrahedron e are owned by the same
00256   // chare that owns point p; if so, return true, if not, return the lowest of
00257   // the owner chare ids of the points of the tetrahedron (chosen as the owner
00258   // of e)
00259   auto own = [ &chp, &tetinpoel ]( std::size_t e, std::size_t p )
00260            -> std::pair< bool, std::size_t >
00261   {
00262     std::vector< bool > op;
00263     std::set< std::size_t > owner;
00264     for (std::size_t n=0; n<4; ++n)
00265       if (chp[ tetinpoel[e*4+n] ] == chp[p])
00266         op.push_back( true );
00267       else
00268         owner.insert( chp[ tetinpoel[e*4+n] ] );
00269     if (op.size() == 4)
00270       return { true, 0 };
00271     else
00272       return { false, std::min( chp[p], *owner.begin() ) };
00273   };
00274 
00275   // Construct array of chare ownership IDs mapping mesh elements to chares
00276   std::vector< std::size_t > che( tetinpoel.size()/4 );
00277   for (std::size_t p=0; p<chp.size(); ++p)  // for all mesh points
00278     for (auto i=esup.second[p]+1; i<=esup.second[p+1]; ++i) {
00279       auto e = esup.first[i];
00280       auto o = own(e,p);
00281       if (o.first)
00282         che[e] = chp[p];
00283       else
00284         che[e] = o.second;
00285     }
00286 
00287 //   std::cout << "che: ";
00288 //   for (auto o : che) std::cout << o << " ";
00289 //   std::cout << '\n';
00290 
00291   Assert( nchare > *std::max_element( begin(che), end(che) ),
00292           "Elements assigned to more than the number chares" );
00293 
00294   // This check should always be done, as it can result from incorrect user
00295   // input compared to the mesh size and not due to programmer error.
00296   auto minmax = std::minmax_element( begin(che), end(che) );
00297   ErrChk( *minmax.first == 0 && *minmax.second == nchare-1, std::move(over) );
00298 
00299   return che;
00300 }
