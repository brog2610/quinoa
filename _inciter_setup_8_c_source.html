232 static std::vector< std::size_t >
233 chElem( std::size_t nchare,
234  const std::vector< std::size_t >& chp,
235  const std::vector< std::size_t >& tetinpoel,
236  const std::pair< std::vector< std::size_t >,
237  std::vector< std::size_t > >& esup )
238 //******************************************************************************
239 //! Construct array of chare ownership IDs mapping mesh elements to chares
240 //! \param[in] nchare Number chares points are partitioned into
241 //! \param[in] chp Chares of points: array of chare ownership IDs mapping graph
242 //! points to Charm++ chares. Size: number of points in mesh graph.
243 //! \param[inout] tetinpoel Mesh tetrahedron element connectivity
244 //! \param[inout] esup Derived data structure, storing elements surrounding
245 //! points in mesh
246 //! \details This function constructs the vector che, 'chares of elements'. This
247 //! is the equivalent of chp, 'chares of points', but stores the chare ids of
248 //! mesh elements. The element ownership is computed based on the point
249 //! ownership. If all points of an element are owned by a chare, then that
250 //! chare owns the element. If not all points of an element are owned by the
251 //! a chare, than the chare with lower chare id gets to own the element.
252 //! \author J. Bakosi
253 //******************************************************************************
254 {
255  // Lambda to find out if all 4 points of tetrahedron e are owned by the same
256  // chare that owns point p; if so, return true, if not, return the lowest of
257  // the owner chare ids of the points of the tetrahedron (chosen as the owner
258  // of e)
259  auto own = [ &chp, &tetinpoel ]( std::size_t e, std::size_t p )
260  -> std::pair< bool, std::size_t >
261  {
262  std::vector< bool > op;
263  std::set< std::size_t > owner;
264  for (std::size_t n=0; n<4; ++n)
265  if (chp[ tetinpoel[e*4+n] ] == chp[p])
266  op.push_back( true );
267  else
268  owner.insert( chp[ tetinpoel[e*4+n] ] );
269  if (op.size() == 4)
270  return { true, 0 };
271  else
272  return { false, std::min( chp[p], *owner.begin() ) };
273  };
274 
275  // Construct array of chare ownership IDs mapping mesh elements to chares
276  std::vector< std::size_t > che( tetinpoel.size()/4 );
277  for (std::size_t p=0; p<chp.size(); ++p) // for all mesh points
278  for (auto i=esup.second[p]+1; i<=esup.second[p+1]; ++i) {
279  auto e = esup.first[i];
280  auto o = own(e,p);
281  if (o.first)
282  che[e] = chp[p];
283  else
284  che[e] = o.second;
285  }
286 
287 // std::cout << "che: ";
288 // for (auto o : che) std::cout << o << " ";
289 // std::cout << '\n';
290 
291  Assert( nchare > *std::max_element( begin(che), end(che) ),
292  "Elements assigned to more than the number chares" );
293 
294  // This check should always be done, as it can result from incorrect user
295  // input compared to the mesh size and not due to programmer error.
296  auto minmax = std::minmax_element( begin(che), end(che) );
297  ErrChk( *minmax.first == 0 && *minmax.second == nchare-1, std::move(over) );
298 
299  return che;
300 }

315  std::vector< std::map< std::size_t, std::vector< std::size_t > > >& ecomm )
316 //******************************************************************************
317 //! Assign mesh partitions to Charm++ chares
318 //! \param[in] graph Unstructured mesh graph object reference
319 //! \param[in] part Array of chare ownership IDs mapping graph points to
320 //! Charm++ chares, and new->old mesh point id map (new: renumbered, old: as
321 //! in mesh file)
322 //! \param[inout] inputdeck Input deck object filled during parsing user input
323 //! \param[inout] npoin Total number of points in mesh
324 //! \param[inout] point Global mesh point ids owned by each chare
325 //! \param[inout] element Global mesh element ids owned by each chare
326 //! \param[inout] meshfilemap Index map between renumbered mesh point ids and
327 //! those in mesh file
328 //! \param[inout] tetinpoel Mesh tetrahedron element connectivity
329 //! \param[inout] esup Derived data structure, storing elements surrounding
330 //! points in mesh
331 //! \param[inout] pcomm Vector of point-based export maps for all chares (empty
332 //! if nchares=1)
333 //! \param[inout] ecomm Vector of elem-based export maps for all chares (empty
334 //! if nchares=1)
335 //! \note This function only operates on MPI rank 0, since that is the only rank
336 //! where the input argument 'part' is expected to be non-empty.
337 //! \author J. Bakosi
338 //******************************************************************************
339 {
340  int peid;
341  MPI_Comm_rank( MPI_COMM_WORLD, &peid );
342 
343  if (peid == 0) {
344 
345  const auto& chp = part.first;
346 
347  Assert( chp.size() == graph.size(),
348  "Size of ownership array must equal the number of graph nodes" );
349 
350  // Return total number points
351  npoin = chp.size();
352 
353  // Find out number of chares desired
354  auto minmax = std::minmax_element( begin(chp), end(chp) );
355  auto nchare = *minmax.second - *minmax.first + 1;
356 
357  // Construct and return global mesh point ids for each chare
358  point = poinOwner( nchare, chp );
359 
360  // Return index map between renumbered mesh points and those in mesh file
361  meshfilemap = part.second;
362 
363  // Return element connectivity
364  tetinpoel = graph.tetinpoel();
365 
366  // Generate and return elements surrounding points
367  esup = tk::genEsup( tetinpoel, 4 );
368 
369  //! Construct array of chare ownership IDs mapping mesh elements to chares
370  const auto che = chElem( nchare, chp, tetinpoel, esup );
371 
372  // Construct and return global mesh point ids for each chare
373  element = elemOwner( nchare, che );
374 
375  // If graph not partitioned, quit leaving communication maps empty
376  if (nchare == 1) return;
377 
378  // Compute and return point-based communication maps
379  pcomm = tk::poinCommMaps( graph, chp, tetinpoel, nchare, std::move(over) );
380 
381  // Compute and return element-based communication maps
382  ecomm = tk::elemCommMaps( chp, tetinpoel, element, nchare );
383  }

454 void
455 prepareMesh(
456  const ctr::CmdLine& cmdline,
457  const tk::Print& print,
458  const ctr::InputDeck& inputdeck,
459  std::vector< std::pair< std::string, tk::Timer::Watch > >& timestamp,
460  std::size_t& npoin,
461  std::vector< std::vector< std::size_t > >& point,
462  std::vector< std::vector< std::size_t > >& element,
463  std::vector< std::size_t >& meshfilemap,
464  std::vector< std::size_t >& tetinpoel,
465  std::pair< std::vector< std::size_t >, std::vector< std::size_t > >& esup,
466  std::vector< std::map< std::size_t, std::vector< std::size_t > > >& pcomm,
467  std::vector< std::map< std::size_t, std::vector< std::size_t > > >& ecomm )
468 //******************************************************************************
469 //! Prepare computational mesh
470 //! \param[in] cmdline Command-line stack
471 //! \param[in] print Pretty printer
472 //! \param[in] inputdeck Input deck with inser input
473 //! \param[inout] timestamp Time stamps in h:m:s format
474 //! \param[inout] npoin Total number of points in mesh
475 //! \param[inout] point Global mesh point ids owned by each chare
476 //! \param[inout] element Global mesh element ids owned by each chare
477 //! \param[inout] meshfilemap Index map between renumbered mesh point ids and
478 //! those in mesh file
479 //! \param[inout] tetinpoel Mesh tetrahedron element connectivity
480 //! \param[inout] esup Derived data structure, storing elements surrounding
481 //! points in mesh
482 //! \param[inout] pcomm Vector of point-based export maps for all chares (empty
483 //! if nchares=1)
484 //! \param[inout] ecomm Vector of elem-based export maps for all chares (empty
485 //! if nchares=1)
486 //! \author J. Bakosi
487 //******************************************************************************
488 {
489  // The load is taken to be proportional to the number of elements of the
490  // mesh which is proportional to the number of points as well as the number
491  // of unique edges in the mesh. For a typical mesh of tetrahedra
492  // nelem = 5.5*npoin
493  // nedge = 7*npoin
494  // npsup = 14*npoin,
495  // where
496  // * nelem - number of elements,
497  // * npoin - number of points,
498  // * nedge - number of unique edges,
499  // * npsup - number of points surrounding points, which is the same as the
500  // number of (non-unique) edges surrounding points. See also Lohner, An
501  // Introduction to Applied CFD Techniques, Wiley, 2008.
502  uint64_t load;
503  int load_tag = 1;
504 
505  // Create empty unstructured mesh object (will only load connectivity, so we
506  // call it a graph instead of a mesh)
507  tk::UnsMesh graph;
508 
509  int peid, numpes;
510  MPI_Status status;
511  MPI_Comm_rank( MPI_COMM_WORLD, &peid );
512  MPI_Comm_size( MPI_COMM_WORLD, &numpes );
513 
514  // Read mesh graph from file only on MPI rank 0 and distribute load size
515  if (peid == 0) {
516  tk::Timer timer;
517  tk::ExodusIIMeshReader er( cmdline.get< tag::io, tag::input >() );
518  er.readGraph( graph );
519  timestamp.emplace_back( "Read mesh graph from file", timer.hms() );
520  load = graph.tetinpoel().size()/4;
521  for (int i=1; i<numpes; ++i)
522  MPI_Send( &load, 1, MPI_UNSIGNED, i, load_tag, MPI_COMM_WORLD );
523  } else {
524  MPI_Recv( &load, 1, MPI_UNSIGNED, 0, load_tag, MPI_COMM_WORLD, &status );
525  }
526 
527  // Compute load distribution given total work (load) and user-specified
528  // virtualization
529  uint64_t chunksize, remainder;
530  const auto nchare =
531  tk::linearLoadDistributor( cmdline.get< tag::virtualization >(),
532  load, numpes, chunksize, remainder );
533 
534  // Print out info on mesh and load distribution
535  meshinfo( print, graph, load, chunksize, remainder, nchare,
536  cmdline.get< tag::virtualization >() );
537 
538  // Partition graph using Zoltan and assign mesh partitions to Charm++ chares
539  tk::Timer t;
540  assignMesh( graph, tk::zoltan::partitionMesh(graph, nchare, print),
541  inputdeck, npoin, point, element, meshfilemap, tetinpoel, esup,
542  pcomm, ecomm );
543  timestamp.emplace_back( "Partition mesh & assign to Charm++ chares",
544  t.hms() );
545 }
