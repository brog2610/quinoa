00059 void
00060 Statistics::setupOrdinary( const ctr::OffsetMap& offset,
00061                            const std::vector< ctr::Product >& stat )
00062 //******************************************************************************
00063 //  Prepare for computing ordinary moments
00064 //! \param[in] offset Map of offsets in memory to address variable fields
00065 //! \param[in] stat List of requested statistical moments
00066 //! \author J. Bakosi
00067 //******************************************************************************
00068 {
00069   for (const auto& product : stat)
00070     if (ordinary(product)) {
00071 
00072       m_instOrd.emplace_back( std::vector< const tk::real* >() );
00073 
00074       int i = 0;
00075       for (const auto& term : product) {
00076         auto o = offset.find( term.var );
00077         Assert( o != end( offset ), "No such depvar" );
00078         // Put in starting address of instantaneous variable
00079         m_instOrd.back().push_back( m_particles.cptr( term.field, o->second ) );
00080         // Collect all means of estimated statistics in a linear vector; this
00081         // will be used to find means for fluctuations. Thus only collect single
00082         // terms, i.e., <Y1>, <Y2>, etc., but not <Y1Y2>, etc.
00083         if (i==0) m_ordTerm.push_back( term );
00084         ++i;
00085       }
00086 
00087       // Increase number of ordinary moments by one
00088       m_ordinary.push_back( 0.0 );
00089       // Count up orindary moments
00090       ++m_nord;
00091     }
00092 
00093   // Put in a zero as the last ordinary moment. This will be used as the center
00094   // about which central moments are computed. If this is not needed, e.g.,
00095   // because there is no central moments or not central PDFs are requested, this
00096   // is small, unused, and harmless.
00097   if (m_nord) {
00098     // Storage for all the required ordinary moments
00099     // +1 for 0 as center for ordinary moments in computing central moments
00100     m_ordinary.resize( m_nord + 1 );
00101     // Put in zero as center for ordinary moments in central products
00102     m_ordinary[ m_nord ] = 0.0;
00103   }
00104 }

00244 void
00245 Statistics::accumulateOrd()
00246 //******************************************************************************
00247 //  Accumulate (i.e., only do the sum for) ordinary moments
00248 //! \author J. Bakosi
00249 //******************************************************************************
00250 {
00251   if (m_nord) {
00252     // Zero ordinary moment accumulators
00253     std::fill( begin(m_ordinary), end(m_ordinary), 0.0 );
00254 
00255     // Accumulate sum for ordinary moments. This is a partial sum, so no
00256     // division by the number of samples.
00257     const auto npar = m_particles.npar();
00258     for (auto p=decltype(npar){0}; p<npar; ++p) {
00259       for (std::size_t i=0; i<m_nord; ++i) {
00260        auto prod = m_particles.cvar( m_instOrd[i][0], p );
00261         const auto s = m_instOrd[i].size();
00262         for (auto j=decltype(s){1}; j<s; ++j) {
00263           prod *= m_particles.cvar( m_instOrd[i][j], p );
00264         }
00265         m_ordinary[i] += prod;
00266       }
00267     }
00268   }
00269 }

00271 void
00272 Statistics::accumulateCen( const std::vector< tk::real >& ord )
00273 //******************************************************************************
00274 //  Accumulate (i.e., only do the sum for) central moments
00275 //! \details The ordinary moments container, m_ordinary, is overwritten here
00276 //!   with the argument ord, because each of multiple Statistics class objects
00277 //!   (residing on different PEs) only collect their partial sums when
00278 //!   accumulateOrd() is run. By the time the accumulation of the central
00279 //!   moments is started, the ordinary moments have been collected from all
00280 //!   PEs and thus are the same to be passed here on all PEs. For example
00281 //!   client-code, see walker::Distributor.
00282 //! \param[in] ord Ordinary moments
00283 //! \author J. Bakosi
00284 //******************************************************************************
00285 {
00286   if (m_ncen) {
00287     // Overwrite ordinary moments by those computed across all PEs
00288     for (std::size_t i=0; i<ord.size(); ++i) m_ordinary[i] = ord[i];
00289 
00290     // Zero central moment accumulators
00291     std::fill( begin(m_central), end(m_central), 0.0 );
00292 
00293     // Accumulate sum for central moments. This is a partial sum, so no division
00294     // by the number of samples.
00295     const auto npar = m_particles.npar();
00296     for (auto p=decltype(npar){0}; p<npar; ++p) {
00297       for (std::size_t i=0; i<m_ncen; ++i) {
00298         auto prod = m_particles.cvar( m_instCen[i][0], p ) - *(m_ctr[i][0]);
00299         const auto s = m_instCen[i].size();
00300         for (auto j=decltype(s){1}; j<s; ++j) {
00301           prod *= m_particles.cvar( m_instCen[i][j], p ) - *(m_ctr[i][j]);
00302         }
00303         m_central[i] += prod;
00304       }
00305     }
00306   }

00310 Statistics::accumulateOrdPDF()
00311 //******************************************************************************
00312 //  Accumulate (i.e., only do the sum for) ordinary PDFs
00313 //! \author J. Bakosi
00314 //******************************************************************************
00315 {
00316   if (!m_ordupdf.empty() || !m_ordbpdf.empty() || !m_ordtpdf.empty()) {
00317     // Zero PDF accumulators
00318     for (auto& pdf : m_ordupdf) pdf.zero();
00319     for (auto& pdf : m_ordbpdf) pdf.zero();
00320     for (auto& pdf : m_ordtpdf) pdf.zero();
00321 
00322     // Accumulate partial sum for PDFs
00323     const auto npar = m_particles.npar();
00324     for (auto p=decltype(npar){0}; p<npar; ++p) {
00325       std::size_t i = 0;
00326       // Accumulate partial sum for univariate PDFs
00327       for (auto& pdf : m_ordupdf) {
00328         pdf.add( m_particles.cvar( m_instOrdUniPDF[i++][0], p ) );
00329       }
00330       // Accumulate partial sum for bivariate PDFs
00331       i = 0;
00332       for (auto& pdf : m_ordbpdf) {
00333         const auto inst = m_instOrdBiPDF[i++];
00334         pdf.add( {{ m_particles.cvar( inst[0], p ),
00335                     m_particles.cvar( inst[1], p ) }} );
00336       }
00337       // Accumulate partial sum for trivariate PDFs
00338       i = 0;
00339       for (auto& pdf : m_ordtpdf) {
00340         const auto inst = m_instOrdTriPDF[i++];
00341         pdf.add( {{ m_particles.cvar( inst[0], p ),
00342                     m_particles.cvar( inst[1], p ),
00343                     m_particles.cvar( inst[2], p ) }} );
00344       }
00345     }
00346   }
