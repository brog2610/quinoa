00028 //! Register class into factory with given key. This is used to register a
00029 //! derived-class object's constructor (deriving from some base class) to a
00030 //! factory. The factory itself is a std::map< key, std::function< Child*() > >,
00031 //! i.e., an associative container, associating some key to a std::function
00032 //! object holding a pointer of Child's base class constructor. The constructor
00033 //! and its bound arguments are stored via boost::factory, which, in this
00034 //! use-case, yields the correct function object of type Base constructor pointer
00035 //! and thus facilitates runtime polymorphism. This function works in conjunction
00036 //! with boost::factory, i.e., uses reference semantics (works with storing
00037 //! pointers of objects). For a simple example on how to use this function, see
00038 //! UnitTest/tests/Base/Factory.h.
00039 //! \param[in] f Factory to register to (std::map with value using reference
00040 //!   semantics)
00041 //! \param[in] key Key used to identify the entry in the factory
00042 //! \param[in] args Variable number of arguments to pass to the constructor
00043 //!   being registered. Note that the constructor arguments are only bound to
00044 //!   the constructor and stored in the factory (an std::map with value using
00045 //!   reference semantics). The object is not instantiated here, i.e., the
00046 //!   constructor is not called here. The object can be instantiated by function
00047 //!   instantiate. \see instantiate
00048 //! \author J. Bakosi
00049 template< class C, class Key, class Factory, typename... ConstructorArgs >
00050 void record( Factory& f, const Key& key, ConstructorArgs&&... args ) {
00051   f.emplace( key,
00052              std::bind( boost::factory< C* >(),
00053                         std::forward< ConstructorArgs >( args )... ) );
00054 }

00056 //! Instantiate object from factory. Factory must have a mapped_value which must
00057 //! have a result_type ptr, e.g., std::map< Key, std::function< Obj*() > >. This
00058 //! wrapper function can be used to instantiate an derived-class object from a
00059 //! factory, repeatedly filled with wrapper function 'record' above. The
00060 //! factory, as described in the documentation of 'record', stores base class
00061 //! pointers in an associative container, thereby facilitating runtime
00062 //! polymorphism and a simple lookup-and-instantiate-style object creation. The
00063 //! object instantiated is of type Child class. This function works in
00064 //! conjunction with boost::factory, i.e., uses reference semantics (works with
00065 //! storing pointers of objects). For a simple example on how to
00066 //! use this function, see UnitTest/tests/Base/Factory.h.
00067 //! \param[in] f Factory to instantiate object from (std::map with value using
00068 //!   reference semantics)
00069 //! \param[in] key Key used to identify the object to instantiate from factory
00070 //! \return std::unique_ptr pointing to the object instantiated from factory
00071 //! \see record
00072 //! \author J. Bakosi
00073 template< class Factory, class Key,
00074           class Obj = typename std::remove_pointer<
00075                         typename Factory::mapped_type::result_type >::type >
00076 std::unique_ptr< Obj > instantiate( const Factory& f, const Key& key ) {
00077   const auto it = f.find( key );
00078   Assert( it != end( f ), "No such object registered in factory" );
00079   return std::unique_ptr< Obj >( it->second() );
00080 }

00082 //! Register "model" class of "host" into factory with given key. This wrapper
00083 //! can be used to in a similar manner to 'record', but uses
00084 //! boost::value_factory to bind the model object constructor to its arguments
00085 //! and place it in the associative container storing host class objects. The
00086 //! container is thus of type std::map< key, std::function< T() > >, i.e.,
00087 //! associating a key to a function holding a constructor (and not its
00088 //! pointer). Runtime polymorphism here is realized entirely within the "base"
00089 //! class. See walker::DiffEq in DiffEq/DiffEq.h for an example and more
00090 //! information on runtime polymorphism without client-side inheritance. As a
00091 //! result, this wrapper works with factories that use value semantics, as opposed
00092 //! to 'record' and instantiate which work with reference semantics factories.
00093 //! In order to differentiate between runtime polymorphic classes using
00094 //! reference semantics, consistent with classes realizing runtime polymorphism
00095 //! without client-side inheritance, we call Host as the "Base" class and Model
00096 //! as the "derived" (or child) class. This wrapper function works in
00097 //! conjunction with boost::value_factory, i.e., uses value semantics (works
00098 //! with storing objects instead of object pointers). For a simple example on
00099 //! how to use this function, see UnitTest/tests/Base/Factory.h.
00100 //! \param[in] f Factory to register to (std::map with value using value
00101 //!   semantics)
00102 //! \param[in] key Key used to identify the entry in the factory
00103 //! \param[in] args Variable number of arguments to pass to the constructor
00104 //!   being registered. Note that the constructor arguments are only bound to
00105 //!   the constructor and stored in the factory (an std::map with value using
00106 //!   value semantics). The object is not instantiated here, i.e., the
00107 //!   constructor is not called here. The object can be instantiated by simply
00108 //!   calling the function call operator () on the mapped value. For an example,
00109 //!   RNGStack::selected() in RNG/RNGStack.C.
00110 //! \author J. Bakosi
00111 template< class Host, class ModelConstructor, class Factory, class Key,
00112           typename... ModelConstrArgs >
00113 void recordModel( Factory& f, const Key& key, ModelConstrArgs&&... args ) {
00114   // Bind model constructor to its arguments
00115   std::function< ModelConstructor() > c =
00116     std::bind( boost::value_factory< ModelConstructor >(),
00117                std::forward< ModelConstrArgs >( args )... );
00118   // Bind host to std::function of model constructor and place in factory
00119   f.emplace( key, std::bind( boost::value_factory< Host >(), std::move(c) ) );
00120 }

00122 //! Register model class of host into factory with given key using late binding.
00123 //! This variant of 'record' is very similar to 'recordModel', but registers a
00124 //! model class constructor to a factory with late binding of the constructor
00125 //! argument. Late binding allows specifying the constructor argument at the
00126 //! time when the object is instantiated instead of at the time when it is
00127 //! registered. This has all the benefits of using a factory and allows passing
00128 //! information into the model object only when it is available. The late bind
00129 //! is facilitated via boost::bind instead of std::bind) using a placeholder,
00130 //! _1, which stands for the first argument (bound later, i.e., not here). The
00131 //! value of the model constructor argument is then not used here, only its
00132 //! type, used to perform the late binding. The binding happens to both the
00133 //! model constructor via std::function (passed to the host constructor) as well
00134 //! as explicitly to the host constructor. Prescribing late binding to the model
00135 //! constructor ensures that the compiler requires the argument to the model
00136 //! constructor, i.e., ensures that the host constructor is required to pass the
00137 //! argument to the model constructor. Prescribing late binding to the host
00138 //! constructor puts in the actual request that an argument (with the correct
00139 //! type) must be passed to the host constructor at instantiate time, which then
00140 //! will forward it to the model constructor. See also, for example,
00141 //! walker::DiffEq's corresponding constructor. An example of client-side code
00142 //! is in walker::DiffEqStack::registerDiffEq for registration into factory, and
00143 //! DiffEqStack::createDiffEq for instantiation late-passing the argument.
00144 //! \param[in] f Factory to register to (std::map with value using value
00145 //!   semantics)
00146 //! \param[in] key Key used to identify the entry in the factory
00147 //! \warning Only works with a single constructor argument
00148 //! \author J. Bakosi
00149 template< class Host, class ModelConstructor, class Factory, class Key,
00150           typename ModelConstrArg >
00151 void recordModelLate( Factory& f, const Key& key, ModelConstrArg ) {
00152   // Prescribe late binding the model constructor to its single argument
00153   std::function< ModelConstructor(ModelConstrArg) > c =
00154     boost::bind( boost::value_factory< ModelConstructor >(), _1 );
00155   // Bind host to std::function of model constructor and place in factory and
00156   // also explicitly bind single model constructor argument to host constructor
00157   f.emplace( key,
00158     boost::bind( boost::value_factory< Host >(), std::move(c), _1 ) );
00159 }

00161 //! Register Charm++ model class of host into factory with given key. We bind a
00162 //! host constructor to its arguments of which the first one is a std::function
00163 //! holding a model constructor type (modeling, i.e., used polymorhically with
00164 //! host), followed by an optional number of others (possibly zero) with
00165 //! arbitrary types. Note that the model constructor is a nullptr (default-
00166 //! constructed) and only used to forward its type to the call site inside
00167 //! std::function. The host constructor function is then placed into the
00168 //! factory. This is because Charm++ chares do not explicitly invoke
00169 //! constructors, only call ckNew() on their proxy, which requires all
00170 //! constructor arguments to be present and forwarded to the actual constructor
00171 //! that is only called at a later point in time. This can then be used by those
00172 //! constructors of hosts that invoke the model constructors' proxies' ckNew()
00173 //! and ignore the std::function. See, e.g., rngtest::Battery() and the
00174 //! associated unit tests in UnitTest/tests/Base/Factory.h.
00175 //! \param[in] f Factory to register to (std::map with value using value
00176 //!   semantics)
00177 //! \param[in] key Key used to identify the entry in the factory
00178 //! \param[in] args Variable number of arguments to pass to the constructor
00179 //!   being registered.

00182           typename... ModelConstrArgs >
00183 void recordCharmModel( Factory& f, const Key& key, ModelConstrArgs&&... args ) {
