47  //! Data access dispatch. Public interface to particle data access. Use it
48  //! as ParProps( p, c, o ), where p is the particle index, c is component
49  //! index specifying the scalar equation within a system of equations, and o
50  //! is the offset specifying the position at which the system resides among
51  //! other systems.
52  //! \param[in] particle Particle index
53  //! \param[in] component Component index, i.e., position of a scalar within
54  //! a system
55  //! \param[in] offset System offset specifying the position of the system of
56  //! equations among other systems
57  //! \return Reference to particle data of type tk::real
58  //! \author J. Bakosi
59  inline tk::real&
60  operator()( ncomp_t particle, ncomp_t component, ncomp_t offset )
61  const noexcept {
62  return access( particle, component, offset, int2type< Layout >() );
63  }

65  //! Const ptr to physical variable access dispatch. Public interface to
66  //! physical variable access. cptr() and cvar() are intended to be used
67  //! together in case component and offset would be expensive to compute for
68  //! data access via the function call operator. In essence, cptr() returns
69  //! part of the address known based on component and offset and intended to
70  //! be used in a setup phase. Then cvar() takes this partial address and
71  //! finishes the address calculation given the particle id. The following
72  //! two data accesses are equivalent (modulo constness):
73  //! * real& value = operator()( par, comp, offs );
74  //! * const real* p = cptr( comp, offs );
75  //! const real& value = cvar( p, par );
76  //! \param[in] component Component index, i.e., position of a scalar within
77  //! a system
78  //! \param[in] offset System offset specifying the position of the system of
79  //! equations among other systems
80  //! \return Pointer to particle data of type tk::real for use with cvar()
81  //! \see Check out Statistics::setupOrdinary() and
82  //! Statistics::accumulateOrd() in Statistics/Statistics.C to see cptr()
83  //! and cvar() in action.
84  //! \author J. Bakosi
85  inline const tk::real*
86  cptr( ncomp_t component, ncomp_t offset ) const noexcept
87  { return cptr( component, offset, int2type< Layout >() ); }
