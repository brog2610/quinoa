88 //******************************************************************************
89 // Initialize linear system
90 //! \author J. Bakosi
91 //******************************************************************************
92 {
93  // Set initial conditions
94  ic();
95 
96  // If the desired max number of time steps is zero or the desired max time is
97  // less than the initial time step size, finish righ away.
98  if ( g_inputdeck.get< tag::discr, tag::nstep >() == 0 ||
99  g_inputdeck.get< tag::discr, tag::t0 >() >
100  g_inputdeck.get< tag::discr, tag::term >() ||
101  g_inputdeck.get< tag::discr, tag::term >() < dt ) {
102 
103  // Send time stamps to the host
104  m_hostproxy.arrTimestamp( m_timestamp );
105 
106  // Tell the Charm++ runtime system to call back to Conductor::finish(). The
107  // reduction is done via creating a callback that invokes the typed
108  // reduction client, where m_hostproxy is the proxy on which the reduction
109  // target method, evaluateTime(), is called upon completion of the
110  // reduction.
111  contribute(
112  CkCallback( CkReductionTarget( Conductor, finish ), m_hostproxy ) );
113 
114  } else {
115 
116  // Compute left-hand side of PDE
117  lhs();
118  // Advance PDE in time (start at stage 0)
119  advance( 0, dt, m_it, m_t );
120  // Send time stamps to the host
121  m_hostproxy.arrTimestamp( m_timestamp );
122 
123  }

143 void
144 Performer::lhs()
145 //******************************************************************************
146 // Compute left-hand side of PDE
147 //! \author J. Bakosi
148 //******************************************************************************
149 {
150  tk::Timer t;
151 
152  const auto& x = m_coord[0];
153  const auto& y = m_coord[1];
154  const auto& z = m_coord[2];
155 
156  // Sparse matrix: global mesh point row and column ids, and nonzero value
157  std::map< std::size_t, std::map< std::size_t, tk::real > > lhs;
158 
159  for (std::size_t e=0; e<m_inpoel.size()/4; ++e) {
160  const auto a = m_inpoel[e*4+0];
161  const auto b = m_inpoel[e*4+1];
162  const auto c = m_inpoel[e*4+2];
163  const auto d = m_inpoel[e*4+3];
164  std::array< tk::real, 3 > ba{{ x[b]-x[a], y[b]-y[a], z[b]-z[a] }},
165  ca{{ x[c]-x[a], y[c]-y[a], z[c]-z[a] }},
166  da{{ x[d]-x[a], y[d]-y[a], z[d]-z[a] }};
167  const auto J = tk::triple( ba, ca, da ) / 120.0;
168 
169  const auto A = m_gid[a];
170  const auto B = m_gid[b];
171  const auto C = m_gid[c];
172  const auto D = m_gid[d];
173  auto& lA = lhs[A];
174  lA[A] += 2.0*J;
175  lA[B] += J;
176  lA[C] += J;
177  lA[D] += J;
178  auto& lB = lhs[B];
179  lB[A] += J;
180  lB[B] += 2.0*J;
181  lB[C] += J;
182  lB[D] += J;
183  auto& lC = lhs[C];
184  lC[A] += J;
185  lC[B] += J;
186  lC[C] += 2.0*J;
187  lC[D] += J;
188  auto& lD = lhs[D];
189  lD[A] += J;
190  lD[B] += J;
191  lD[C] += J;
192  lD[D] += 2.0*J;
193  }
194 
195  m_timestamp.emplace_back( "Compute left-hand side matrix", t.dsec() );
196 
197  m_lsmproxy.ckLocalBranch()->charelhs( thisIndex, lhs );
198 }

200 void
201 Performer::rhs( tk::real mult,
202  tk::real dt,
203  const std::map< std::size_t, tk::real >& sol,
204  std::map< std::size_t, tk::real >& newrhs )
205 //******************************************************************************
206 // Compute right-hand side of PDE
207 //! \param[in] mult Multiplier differentiating the different stages in
208 //! multi-stage time stepping
209 //! \param[in] dt Size of time step
210 //! \param[in] sol Solution vector at current stage
211 //! \param[inout] newrhs Right-hand side vector computed
212 //! \author J. Bakosi
213 //******************************************************************************
214 {
215  tk::Timer t;
216 
217  const auto& x = m_coord[0];
218  const auto& y = m_coord[1];
219  const auto& z = m_coord[2];
220 
221  const tk::real U0 = 0.5;
222  const tk::real LAMBDA = 5.0e-4;
223  const tk::real D = 10.0;
224 
225  newrhs.clear();
226 
227  for (std::size_t e=0; e<m_inpoel.size()/4; ++e) {
228  const auto a = m_inpoel[e*4+0];
229  const auto b = m_inpoel[e*4+1];
230  const auto c = m_inpoel[e*4+2];
231  const auto d = m_inpoel[e*4+3];
232 
233  // compute element Jacobi determinant
234  std::array< tk::real, 3 > ba{{ x[b]-x[a], y[b]-y[a], z[b]-z[a] }},
235  ca{{ x[c]-x[a], y[c]-y[a], z[c]-z[a] }},
236  da{{ x[d]-x[a], y[d]-y[a], z[d]-z[a] }};
237  const auto J = tk::triple( ba, ca, da );
238 
239  // construct tetrahedron element-level matrices
240 
241  // consistent mass
242  std::array< std::array< tk::real, 4 >, 4 > mass; // nnode*nnode [4][4]
243  // diagonal
244  mass[0][0] = mass[1][1] = mass[2][2] = mass[3][3] = J/60.0;
245  // off-diagonal
246  mass[0][1] = mass[0][2] = mass[0][3] =
247  mass[1][0] = mass[1][2] = mass[1][3] =
248  mass[2][0] = mass[2][1] = mass[2][3] =
249  mass[3][0] = mass[3][1] = mass[3][2] = J/120.0;
250 
251  // prescribed shear velocity
252  std::array< std::array< tk::real, 4 >, 3 > vel; // ndim*nnode [3][4]
253  vel[0][0] = U0 + LAMBDA*y[a]; vel[1][0] = 0.0; vel[2][0] = 0.0;
254  vel[0][1] = U0 + LAMBDA*y[b]; vel[1][1] = 0.0; vel[2][1] = 0.0;
255  vel[0][2] = U0 + LAMBDA*y[c]; vel[1][2] = 0.0; vel[2][2] = 0.0;
256  vel[0][3] = U0 + LAMBDA*y[d]; vel[1][3] = 0.0; vel[2][3] = 0.0;
257 
258  // shape function derivatives
259  std::array< std::array< tk::real, 3 >, 4 > grad; // nnode*ndim [4][3]
260  grad[1] = tk::crossdiv( ca, da, J );
261  grad[2] = tk::crossdiv( da, ba, J );
262  grad[3] = tk::crossdiv( ba, ca, J );
263  for (std::size_t i=0; i<3; ++i)
264  grad[0][i] = -grad[1][i]-grad[2][i]-grad[3][i];
265 
266  // solution at nodes at time n
267  std::array< tk::real, 4 > u {{ tk::ref( m_u, m_gid[a] ),
268  tk::ref( m_u, m_gid[b] ),
269  tk::ref( m_u, m_gid[c] ),
270  tk::ref( m_u, m_gid[d] ) }};
271  // solution at nodes at time n (at stage 0) and n+1/2 (at stage 1)
272  std::array< tk::real, 4 > s {{ tk::ref( sol, m_gid[a] ),
273  tk::ref( sol, m_gid[b] ),
274  tk::ref( sol, m_gid[c] ),
275  tk::ref( sol, m_gid[d] ) }};
276  // pointers to rhs at nodes
277  std::array< tk::real*, 4 > r {{ &newrhs[ m_gid[a] ],
278  &newrhs[ m_gid[b] ],
279  &newrhs[ m_gid[c] ],
280  &newrhs[ m_gid[d] ] }};
281 
282  // add mass contribution to rhs
283  for (std::size_t i=0; i<4; ++i)
284  for (std::size_t j=0; j<4; ++j)
285  *r[i] += mass[i][j] * u[j];
286 
287  // add advection contribution to rhs
288  for (std::size_t i=0; i<4; ++i)
289  for (std::size_t j=0; j<4; ++j)
290  for (std::size_t k=0; k<3; ++k)
291  for (std::size_t l=0; l<4; ++l)
292  *r[i] -= mult * dt * mass[i][j] * vel[k][j] * grad[l][k] * s[l];
293 
294  // add diffusion contribution to rhs
295  for (std::size_t i=0; i<4; ++i)
296  for (std::size_t j=0; j<4; ++j)
297  for (std::size_t k=0; k<3; ++k)
298  *r[i] -= mult * dt * D * J/6.0 * grad[i][k] * grad[j][k] * s[j];
299  }
300 
301  m_timestamp.emplace_back( "Compute right-hand side vector", t.dsec() );
302 
303  m_lsmproxy.ckLocalBranch()->charerhs( thisIndex, newrhs );
304 }

307 Performer::initIds( const std::vector< std::size_t >& gelem )
308 //******************************************************************************
309 //! Initialize local->global, global->local node ids, element connectivity
310 //! \param[in] gelem Set of unique owned global element ids
311 //! \author J. Bakosi
312 //******************************************************************************
313 {
314  tk::Timer t;
315 
316  // Build unique global node ids of owned elements
317  for (auto e : gelem) {
318  m_gid.push_back( g_tetinpoel[e*4] );
319  m_gid.push_back( g_tetinpoel[e*4+1] );
320  m_gid.push_back( g_tetinpoel[e*4+2] );
321  m_gid.push_back( g_tetinpoel[e*4+3] );
322  }
323  tk::unique( m_gid );
324 
325  // Assign local node ids to global node ids
326  const auto lnode = assignLid( m_gid );
327 
328  // Generate element connectivity for owned elements using local point ids
329  for (auto e : gelem) {
330  m_inpoel.push_back( tk::val( lnode, g_tetinpoel[e*4] ) );
331  m_inpoel.push_back( tk::val( lnode, g_tetinpoel[e*4+1] ) );
332  m_inpoel.push_back( tk::val( lnode, g_tetinpoel[e*4+2] ) );
333  m_inpoel.push_back( tk::val( lnode, g_tetinpoel[e*4+3] ) );
334  }
335 
336  // Send off number of columns per row to linear system merger
337  m_lsmproxy.ckLocalBranch()->charerow( thisProxy, thisIndex, m_gid );
338 
339  m_timestamp.emplace_back( "Initialize mesh point ids, element connectivity",
340  t.dsec() );

491 void
492 Performer::advance( uint8_t stage, tk::real dt, uint64_t it, tk::real t )
493 //******************************************************************************
494 // Advance equations to next stage in multi-stage time stepping
495 //! \param[in] stage Stage in multi-stage time stepping
496 //! \param[in] dt Size of time step
497 //! \param[in] it Iteration count
498 //! \param[in] t Physical time
499 //! \author J. Bakosi
500 //******************************************************************************
501 {
502  // Re-enable SDAG waits for rebuilding the right-hand side vector only
503  m_lsmproxy.ckLocalBranch()->enable_wait4rhs();
504 
505  // Update local copy of time step stage
506  m_stage = stage;
507 
508  // Advance stage in multi-stage time stepping by updating the rhs
509  if (m_stage < 1) {
510 
511  rhs( 0.5, dt, m_u, m_uf );
512 
513  } else {
514 
515  // Update local copy of physical time and iteration count at the final stage
516  m_t = t;
517  m_it = it;
518 
519  rhs( 1.0, dt, m_uf, m_un );
520 
521  }
522 }

524 void
525 Performer::updateSolution( const std::map< std::size_t, tk::real >& sol )
526 //******************************************************************************
527 // Update solution vector
528 //! \author J. Bakosi
529 //******************************************************************************
530 {
531  // Lambda to tell the Charm++ runtime system to call back to
532  // Conductor::evaluateTime() once all Performer chares have received the
533  // update. The reduction is done via creating a callback that invokes the
534  // typed reduction client, where m_hostproxy is the proxy on which the
535  // reduction target method, evaluateTime(), is called upon completion of the
536  // reduction.
537  auto evaluateTime = [this](){
538  contribute(
539  CkCallback( CkReductionTarget( Conductor, evaluateTime ), m_hostproxy ) );
540  };
541 
542  // Receive update of solution vector
543  for (const auto& s : sol) m_ur[ s.first ] = s.second;
544 
545  // If all contributions we own have been received, continue by updating a
546  // different solution vector depending on time step stage
547  if (m_ur.size() == m_gid.size()) {
548  if (m_stage < 1) {
549  m_uf = std::move( m_ur );
550  } else {
551  m_u = std::move( m_ur );
552  if (!((m_it+1) % g_inputdeck.get< tag::interval, tag::field >()))
553  writeFields( m_t + g_inputdeck.get< tag::discr, tag::dt >() );
554  }
555  evaluateTime();
556  }
557 }
