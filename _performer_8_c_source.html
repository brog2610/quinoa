140 void
141 Performer::lhs()
142 //******************************************************************************
143 // Compute left-hand side of PDE
144 //! \author J. Bakosi
145 //******************************************************************************
146 {
147  // Generate points surrounding points
148  const auto psup = tk::genPsup( m_inpoel, 4, tk::genEsup(m_inpoel,4) );
149 
150  Assert( psup.second.size()-1 == m_gid.size(),
151  "Number of mesh points and number of global IDs unequal" );
152 
153  // Sparse matrix storing the nonzero matrix values at rows and columns given
154  // by psup. The format is similar to compressed row storage, but the diagonal
155  // and off-diagonal data are stored in separate vectors. For the off-diagonal
156  // data the local row and column indices, at which values are nonzero, are
157  // stored by psup (psup1 and _psup2, where psup2 holds the indices at which
158  // psup1 holds the point ids surrounding points, see also tk::genPsup()). Note
159  // that the number of mesh points (our chunk) npoin = psup.second.size()-1.
160  std::vector< tk::real > lhsd( psup.second.size()-1, 0.0 );
161  std::vector< tk::real > lhso( psup.first.size(), 0.0 );
162 
163  // Lambda to compute the sparse matrix vector index for row and column
164  // indices. Used only for off-diagonal entries.
165  auto spidx = [ &psup ]( std::size_t r, std::size_t c ) -> std::size_t {
166  Assert( r != c, "Only for computing the off-diagonal indices" );
167  for (auto i=psup.second[r]+1; i<=psup.second[r+1]; ++i)
168  if (c == psup.first[i]) return i;
169  Throw( "Cannot find row, column: " + std::to_string(r) + ',' +
170  std::to_string(c) + " in sparse matrix" );
171  };
172 
173  const auto& x = m_coord[0];
174  const auto& y = m_coord[1];
175  const auto& z = m_coord[2];
176 
177  for (std::size_t e=0; e<m_inpoel.size()/4; ++e) {
178  const auto a = m_inpoel[e*4+0];
179  const auto b = m_inpoel[e*4+1];
180  const auto c = m_inpoel[e*4+2];
181  const auto d = m_inpoel[e*4+3];
182  std::array< tk::real, 3 > ba{{ x[b]-x[a], y[b]-y[a], z[b]-z[a] }},
183  ca{{ x[c]-x[a], y[c]-y[a], z[c]-z[a] }},
184  da{{ x[d]-x[a], y[d]-y[a], z[d]-z[a] }};
185  const auto J = tk::triple( ba, ca, da ) / 120.0;
186 
187  lhsd[ a ] += 2.0*J;
188  lhsd[ b ] += 2.0*J;
189  lhsd[ c ] += 2.0*J;
190  lhsd[ d ] += 2.0*J;
191 
192  lhso[ spidx(a,b) ] += J;
193  lhso[ spidx(a,c) ] += J;
194  lhso[ spidx(a,d) ] += J;
195 
196  lhso[ spidx(b,a) ] += J;
197  lhso[ spidx(b,c) ] += J;
198  lhso[ spidx(b,d) ] += J;
199 
200  lhso[ spidx(c,a) ] += J;
201  lhso[ spidx(c,b) ] += J;
202  lhso[ spidx(c,d) ] += J;
203 
204  lhso[ spidx(d,a) ] += J;
205  lhso[ spidx(d,b) ] += J;
206  lhso[ spidx(d,c) ] += J;
207  }
208 
209  m_linsysmerger.ckLocalBranch()->charelhs( thisIndex, m_gid, psup, lhsd,
210  lhso );
211 }

213 void
214 Performer::rhs( tk::real mult,
215  tk::real dt,
216  const std::vector< tk::real >& sol,
217  std::vector< tk::real >& rhs )
218 //******************************************************************************
219 // Compute right-hand side of PDE
220 //! \param[in] mult Multiplier differentiating the different stages in
221 //! multi-stage time stepping
222 //! \param[in] dt Size of time step
223 //! \param[in] sol Solution vector at current stage
224 //! \param[inout] rhs Right-hand side vector computed
225 //! \author J. Bakosi
226 //******************************************************************************
227 {
228  const auto& x = m_coord[0];
229  const auto& y = m_coord[1];
230  const auto& z = m_coord[2];
231 
232  const tk::real U0 = 0.5;
233  const tk::real LAMBDA = 5.0e-4;
234  const tk::real D = 10.0;
235 
236  std::fill( begin(rhs), end(rhs), 0.0 );
237 
238  for (std::size_t e=0; e<m_inpoel.size()/4; ++e) {
239  const auto a = m_inpoel[e*4+0];
240  const auto b = m_inpoel[e*4+1];
241  const auto c = m_inpoel[e*4+2];
242  const auto d = m_inpoel[e*4+3];
243 
244  // compute element Jacobi determinant
245  const std::array< tk::real, 3 > ba{{ x[b]-x[a], y[b]-y[a], z[b]-z[a] }},
246  ca{{ x[c]-x[a], y[c]-y[a], z[c]-z[a] }},
247  da{{ x[d]-x[a], y[d]-y[a], z[d]-z[a] }};
248  const auto J = tk::triple( ba, ca, da );
249 
250  // construct tetrahedron element-level matrices
251 
252  // consistent mass
253  std::array< std::array< tk::real, 4 >, 4 > mass; // nnode*nnode [4][4]
254  // diagonal
255  mass[0][0] = mass[1][1] = mass[2][2] = mass[3][3] = J/60.0;
256  // off-diagonal
257  mass[0][1] = mass[0][2] = mass[0][3] =
258  mass[1][0] = mass[1][2] = mass[1][3] =
259  mass[2][0] = mass[2][1] = mass[2][3] =
260  mass[3][0] = mass[3][1] = mass[3][2] = J/120.0;
261 
262  // prescribed shear velocity
263  std::array< std::array< tk::real, 4 >, 3 > vel; // ndim*nnode [3][4]
264  vel[0][0] = U0 + LAMBDA*y[a]; vel[1][0] = 0.0; vel[2][0] = 0.0;
265  vel[0][1] = U0 + LAMBDA*y[b]; vel[1][1] = 0.0; vel[2][1] = 0.0;
266  vel[0][2] = U0 + LAMBDA*y[c]; vel[1][2] = 0.0; vel[2][2] = 0.0;
267  vel[0][3] = U0 + LAMBDA*y[d]; vel[1][3] = 0.0; vel[2][3] = 0.0;
268 
269  // shape function derivatives
270  std::array< std::array< tk::real, 3 >, 4 > grad; // nnode*ndim [4][3]
271  grad[1] = tk::crossdiv( ca, da, J );
272  grad[2] = tk::crossdiv( da, ba, J );
273  grad[3] = tk::crossdiv( ba, ca, J );
274  for (std::size_t i=0; i<3; ++i)
275  grad[0][i] = -grad[1][i]-grad[2][i]-grad[3][i];
276 
277  // solution at nodes at time n
278  const std::array< tk::real, 4 > u{{ m_u[a], m_u[b], m_u[c], m_u[d] }};
279  // solution at nodes at time n (at stage 0) and n+1/2 (at stage 1)
280  const std::array< tk::real, 4 > s{{ sol[a], sol[b], sol[c], sol[d] }};
281  // pointers to rhs at nodes
282  std::array< tk::real*, 4 > r{{ &rhs[a], &rhs[b], &rhs[c], &rhs[d] }};
283 
284  // add mass contribution to rhs
285  for (std::size_t i=0; i<4; ++i)
286  for (std::size_t j=0; j<4; ++j)
287  *r[i] += mass[i][j] * u[j];
288 
289  // add advection contribution to rhs
290  for (std::size_t i=0; i<4; ++i)
291  for (std::size_t j=0; j<4; ++j)
292  for (std::size_t k=0; k<3; ++k)
293  for (std::size_t l=0; l<4; ++l)
294  *r[i] -= mult * dt * mass[i][j] * vel[k][j] * grad[l][k] * s[l];
295 
296  // add diffusion contribution to rhs
297  for (std::size_t i=0; i<4; ++i)
298  for (std::size_t j=0; j<4; ++j)
299  for (std::size_t k=0; k<3; ++k)
300  *r[i] -= mult * dt * D * J/6.0 * grad[i][k] * grad[j][k] * s[j];
301  }
302 
303  m_linsysmerger.ckLocalBranch()->charerhs( thisIndex, m_gid, rhs );
304 }

423 Performer::advance( uint8_t stage, tk::real dt, uint64_t it, tk::real t )
424 //******************************************************************************
425 // Advance equations to next stage in multi-stage time stepping
426 //! \param[in] stage Stage in multi-stage time stepping
427 //! \param[in] dt Size of time step
428 //! \param[in] it Iteration count
429 //! \param[in] t Physical time
430 //! \author J. Bakosi
431 //******************************************************************************
432 {
433  // Update local copy of time step stage
434  m_stage = stage;
435 
436  // Advance stage in multi-stage time stepping by updating the rhs
437  if (m_stage < 1) {
438 
439  rhs( 0.5, dt, m_u, m_uf );
440 
441  } else {
442 
443  // Update local copy of physical time and iteration count at the final stage
444  m_t = t;
445  m_it = it;
446 
447  rhs( 1.0, dt, m_uf, m_un );
448 
449  }

452 void
453 Performer::updateSolution( const std::vector< std::size_t >& gid,
454  const std::vector< tk::real >& sol )
455 //******************************************************************************
456 // Update solution vector
457 //! \param[in] gid Global row indices of the vector updated
458 //! \param[in] sol Portion of the unknown/solution vector updated
459 //! \author J. Bakosi
460 //******************************************************************************
461 {
462  Assert( gid.size() == sol.size(),
463  "Size of solution and row ID vectors must equal" );
464 
465  // Receive update of solution vector
466  for (std::size_t i=0; i<gid.size(); ++i)
467  m_un[ tk::val_find(m_lid,gid[i]) ] = sol[i];
468 
469  // Count number of solution nodes updated
470  m_nsol += gid.size();
471 
472  // If all contributions we own have been received, continue by updating a
473  // different solution vector depending on time step stage
474  if (m_nsol == m_gid.size()) {
475 
476  if (m_stage < 1) {
477  m_uf = m_un;
478  } else {
479  m_u = m_un;
480  if (!((m_it+1) % g_inputdeck.get< tag::interval, tag::field >()))
481  writeFields( m_t + g_inputdeck.get< tag::discr, tag::dt >() );
482  }
483 
484  // Prepare for next time step stage
485  m_nsol = 0;
486  // Tell the Charm++ runtime system to call back to Conductor::evaluateTime()
487  // once all Performer chares have received the update. The reduction is done
488  // via creating a callback that invokes the typed reduction client, where
489  // m_conductor is the proxy on which the reduction target method,
490  // evaluateTime(), is called upon completion of the reduction.
491  contribute(
492  CkCallback( CkReductionTarget( Conductor, evaluateTime ), m_conductor ) );
493 
494  }
495 }
