00287 //! \param[in] gelem Set of unique owned global element ids

00296     m_gid.push_back( g_tetinpoel[e*4+1] );

00298     m_gid.push_back( g_tetinpoel[e*4+3] );

00300   tk::unique( m_gid );
00301 
00302   // Assign local node ids to global node ids
00303   const auto lnode = assignLid( m_gid );
00304 
00305   // Generate element connectivity for owned elements using local point ids
00306   for (auto e : gelem) {
00307     m_inpoel.push_back( tk::lid( lnode, g_tetinpoel[e*4] ) );
00308     m_inpoel.push_back( tk::lid( lnode, g_tetinpoel[e*4+1] ) );
00309     m_inpoel.push_back( tk::lid( lnode, g_tetinpoel[e*4+2] ) );
00310     m_inpoel.push_back( tk::lid( lnode, g_tetinpoel[e*4+3] ) );
00311   }
00312 
00313   // Send off number of columns per row to linear system merger
00314   m_lsmproxy.ckLocalBranch()->charerow( thisProxy, thisIndex, m_gid );
