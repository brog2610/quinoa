135 void
136 Performer::lhs()
137 //******************************************************************************
138 // Compute left-hand side of PDE
139 //! \author J. Bakosi
140 //******************************************************************************
141 {
142  tk::Timer t;
143 
144  const auto& x = m_coord[0];
145  const auto& y = m_coord[1];
146  const auto& z = m_coord[2];
147 
148  for (std::size_t e=0; e<m_inpoel.size()/4; ++e) {
149  const auto a = m_inpoel[e*4+0];
150  const auto b = m_inpoel[e*4+1];
151  const auto c = m_inpoel[e*4+2];
152  const auto d = m_inpoel[e*4+3];
153  std::array< tk::real, 3 > ba{{ x[b]-x[a], y[b]-y[a], z[b]-z[a] }},
154  ca{{ x[c]-x[a], y[c]-y[a], z[c]-z[a] }},
155  da{{ x[d]-x[a], y[d]-y[a], z[d]-z[a] }};
156  const auto J = tk::triple( ba, ca, da ) / 120.0;
157 
158  const auto A = m_gid[a];
159  const auto B = m_gid[b];
160  const auto C = m_gid[c];
161  const auto D = m_gid[d];
162  auto& lA = m_lhs[A];
163  lA[A] += 2.0*J;
164  lA[B] += J;
165  lA[C] += J;
166  lA[D] += J;
167  auto& lB = m_lhs[B];
168  lB[A] += J;
169  lB[B] += 2.0*J;
170  lB[C] += J;
171  lB[D] += J;
172  auto& lC = m_lhs[C];
173  lC[A] += J;
174  lC[B] += J;
175  lC[C] += 2.0*J;
176  lC[D] += J;
177  auto& lD = m_lhs[D];
178  lD[A] += J;
179  lD[B] += J;
180  lD[C] += J;
181  lD[D] += 2.0*J;
182  }
183 
184  m_timestamp.emplace_back( "Compute left-hand side matrix", t.dsec() );
185 
186  m_lsmproxy.ckLocalBranch()->charelhs( thisIndex, m_lhs );
187 }

190 Performer::rhs( tk::real mult, std::map< std::size_t, tk::real >& unk )
191 //******************************************************************************
192 // Compute right-hand side of PDE
193 //! \author J. Bakosi
194 //******************************************************************************
195 {
196  tk::Timer t;
197 
198  const auto& x = m_coord[0];
199  const auto& y = m_coord[1];
200  const auto& z = m_coord[2];
201 
202  const tk::real U0 = 0.5;
203  const tk::real LAMBDA = 5.0e-4;
204 
205  m_rhs.clear();
206 
207  for (std::size_t e=0; e<m_inpoel.size()/4; ++e) {
208  const auto a = m_inpoel[e*4+0];
209  const auto b = m_inpoel[e*4+1];
210  const auto c = m_inpoel[e*4+2];
211  const auto d = m_inpoel[e*4+3];
212 
213  // compute element Jacobi determinant
214  std::array< tk::real, 3 > ba{{ x[b]-x[a], y[b]-y[a], z[b]-z[a] }},
215  ca{{ x[c]-x[a], y[c]-y[a], z[c]-z[a] }},
216  da{{ x[d]-x[a], y[d]-y[a], z[d]-z[a] }};
217  const auto J = tk::triple( ba, ca, da );
218 
219  // construct tetrahedron element-level matrices
220 
221  // consistent mass
222  std::array< std::array< tk::real, 4 >, 4 > mass; // nnode*nnode [4][4]
223  // diagonal
224  mass[0][0] = mass[1][1] = mass[2][2] = mass[3][3] = J/60.0;
225  // off-diagonal
226  mass[0][1] = mass[0][2] = mass[0][3] =
227  mass[1][0] = mass[1][2] = mass[1][3] =
228  mass[2][0] = mass[2][1] = mass[2][3] =
229  mass[3][0] = mass[3][1] = mass[3][2] = J/120.0;
230 
231  // prescribed shear velocity
232  std::array< std::array< tk::real, 4 >, 3 > vel; // ndim*nnode [3][4]
233  vel[0][0] = U0 + LAMBDA * y[a]; vel[1][0] = 0.0; vel[2][0] = 0.0;
234  vel[0][1] = U0 + LAMBDA * y[b]; vel[1][1] = 0.0; vel[2][1] = 0.0;
235  vel[0][2] = U0 + LAMBDA * y[c]; vel[1][2] = 0.0; vel[2][2] = 0.0;
236  vel[0][3] = U0 + LAMBDA * y[d]; vel[1][3] = 0.0; vel[2][3] = 0.0;
237 
238  // shape function derivatives
239  std::array< std::array< tk::real, 3 >, 4 > grad; // nnode*ndim [4][3]
240  grad[1] = tk::crossdiv( ca, da, J );
241  grad[2] = tk::crossdiv( ba, da, J );
242  grad[3] = tk::crossdiv( ba, ca, J );
243  for (std::size_t i=0; i<3; ++i)
244  grad[0][i] = -grad[1][i]-grad[2][i]-grad[3][i];
245 
246  // add mass contribution to rhs
247  for (std::size_t i=0; i<4; ++i) {
248  const auto g = m_gid[ m_inpoel[e*4+i] ]; // global node id
249  const auto& u = unk[g]; // ref to unknown at g
250  auto& r = m_rhs[g]; // ref to rhs at g
251  for (std::size_t j=0; j<4; ++j) // add contribution to rhs
252  r += mass[i][j] * 1.5;//u;
253  }
254

283 void
284 Performer::initIds( const std::vector< std::size_t >& gelem )
285 //******************************************************************************
286 //! Initialize local->global, global->local node ids, element connectivity
287 //! \param[in] gelem Set of unique owned global element ids
288 //! \author J. Bakosi
289 //******************************************************************************
290 {
291  tk::Timer t;
292 
293  // Build unique global node ids of owned elements
294  for (auto e : gelem) {
295  m_gid.push_back( g_tetinpoel[e*4] );
296  m_gid.push_back( g_tetinpoel[e*4+1] );
297  m_gid.push_back( g_tetinpoel[e*4+2] );
298  m_gid.push_back( g_tetinpoel[e*4+3] );
299  }
300  tk::unique( m_gid );
301 
302  // Assign local node ids to global node ids
303  const auto lnode = assignLid( m_gid );
304 
305  // Generate element connectivity for owned elements using local point ids
306  for (auto e : gelem) {
307  m_inpoel.push_back( tk::lid( lnode, g_tetinpoel[e*4] ) );
308  m_inpoel.push_back( tk::lid( lnode, g_tetinpoel[e*4+1] ) );
309  m_inpoel.push_back( tk::lid( lnode, g_tetinpoel[e*4+2] ) );
310  m_inpoel.push_back( tk::lid( lnode, g_tetinpoel[e*4+3] ) );
311  }
312 
313  // Send off number of columns per row to linear system merger
314  m_lsmproxy.ckLocalBranch()->charerow( thisProxy, thisIndex, m_gid );
315 
316  m_timestamp.emplace_back( "Initialize mesh point ids, element connectivity",
317  t.dsec() );
318 }
