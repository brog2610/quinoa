00021 struct Toggle_common {
00022   enum class Enum { F1, F2, F3, F4 };
00023   //! Toggle only has a protected constructor, designed to be base class only
00024   struct toggle : tk::Toggle< Enum > {
00025      // The child constructor must initialize all state data Toggle holds: a
00026      // std::string, a std::map< Enum, std::string >, and its inverse, a
00027      // std::map< std::string, Enum >. The easiest is to use initializer lists
00028      // and initialize in place. The size of the two maps must be equal.
00029      toggle( std::string&& g,
00030              std::map< Enum, std::string >&& n,
00031              std::map< std::string, Enum >&& v ) :
00032       tk::Toggle< Enum >( std::move(g), std::move(n), std::move(v) ) {}
00033   };
00034   //! Example toggle
