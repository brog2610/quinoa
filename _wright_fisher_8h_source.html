00111     //! \brief Advance particles according to the Wright-Fisher SDE
00112     //! \param[inout] particles Array of particle properties
00113     //! \param[in] stream Thread (or more precisely stream) ID
00114     //! \param[in] dt Time step size
00115     //! \param[in] moments Map of statistical moments
00116     //! \author J. Bakosi
00117     void advance( tk::ParProps& particles,
00118                   int stream,
00119                   tk::real dt,
00120                   const std::map< tk::ctr::Product, tk::real >& moments )
00121     {
00122       // Compute sum of coefficients
00123       const auto omega = std::accumulate( begin(m_omega), end(m_omega), 0.0 );
00124       const auto npar = particles.npar();
00125 
00126       for (auto p=decltype(npar){0}; p<npar; ++p) {
00127         // Need to build the square-root of the Wright-Fisher diffusion matrix:
00128         // B_ij = y_i * ( delta_ij - y_j ). If the matrix is positive definite,
00129         // the Cholesky decomposition would work, however, B_ij is only positive
00130         // semi-definite, so Cholesky may fail and considering floating-point
00131         // errors it will definitely fail at some point.
00132         //
00133         // A stable square-root for B_ij is not yet implemented. Here is some
00134         // details and points for further development:
00135         //
00136         // An interesting fact from http://math.stackexchange.com/a/332465, on
00137         // gauging the eigen values of a matrix:
00138         //
00139         // "Choosing each diagonal entry to be greater than the sum of the
00140         // absolute values of the other entries in the same row will immediately
00141         // imply that all of the eigenvalues of A are positive, and therefore
00142         // that A is positive definite."
00143         //
00144         // The WF diffusion matrix seems to be positive semi-definite with at
00145         // least one very small eigenvalue O(10e-18). The row and column sums
00146         // are zero. It does not appear that the matrix has negative
00147         // eigenvalues.
00148         //
00149         // Using Cholesky decomposition to compute the square-root may fail.
00150         // Other methods, such as diagonalizing the matrix or the LDL
00151         // decomposition, may work, but have not tried. The former requires
00152         // finding the eigenvalues, with e.g., LAPACK's DSPEVD, thereby
00153         // diagonalizing the matrix, and taking the square-root of the diagonal
00154         // elements to find the square-root of the matrix. See
00155         // http://en.wikipedia.org/wiki/Square_root_of_a_matrix.  This may be
00156         // okay, but probably overkill, as we don't really need the eigenvalues,
00157         // only the square-root of the matrix.
00158         //
00159         // The latter (LDL) is similar to Cholesky, but does not involve square
00160         // root. See http://en.wikipedia.org/wiki/Cholesky_decomposition. See
00161         // also Golub van Loan: Sec.4.1.2 LDL. It appears that LDL would compare
00162         // in perfromance to Cholesky and would work even on large indefinite
00163         // matrices. LAPACK probably has LDL.
00164         //
00165         // I'm putting this aside for now. The next step should probably be
00166         // idintifying the LAPACK for LDL and use it to advance the WF system
00167         // using LD^{1/2}.
00168         //
00169         // See also:
00170         // http://en.wikipedia.org/wiki/Positive-definite_matrix
00171         // file:///opt/intel/composerxe/Documentation/en_US/mkl/mklman/index.htm
00172         // http://icl.cs.utk.edu/lapack-forum/archives/lapack/msg01497.html
00173         // http://yarchive.net/comp/sqrtm.html
00174 
00175         tk::real B[m_ncomp][m_ncomp];
00176         tk::real Bo[m_ncomp][m_ncomp];
00177         for (ncomp_t i=0; i<m_ncomp; ++i) {
00178           const tk::real& pari = particles( p, i, m_offset );
00179           for (ncomp_t j=0; j<m_ncomp; ++j) {
00180             const tk::real& parj = particles( p, j, m_offset );
00181             if (i == j) {
00182               B[i][i] = std::abs( pari * (1.0 - pari) );
00183               if (B[i][i] < 1.0e-10) B[i][i] = 1.0;
00184             } else {
00185               B[i][j] = -pari*parj;
00186             }
00187           }
00188         }
00189         std::memcpy( Bo, B, m_ncomp*m_ncomp*sizeof(tk::real) );
00190         // Compute diffusion matrix (lower triangle of Cholesky-decomposition)
00191         lapack_int n = static_cast< lapack_int >( m_ncomp );
00192         lapack_int info = LAPACKE_dpotrf( LAPACK_ROW_MAJOR, 'L', n, B[0], n );
00193         if (info != 0 ) {
00194           print_matrix( "=======\nOriginal Matrix", Bo[0], n, n );
00195           std::cout << "info: " << info << std::endl;
00196           print_matrix( "Result of Cholesky factorization", B[0], n, n );
00197           for (ncomp_t i=0; i<m_ncomp; ++i) {
00198             std::cout <<
00199               std::setprecision( std::numeric_limits< tk::real >::digits10 )
00200             << i << " par: " << particles( p, i, m_offset) << std::endl;
00201           }
00202         }
00203         //ErrChk( info == 0, "Wright-Fisher Cholesky decomposition unsuccessful, "
00204         //                   "info = " + std::to_string( info ) + ", particle: " +
00205         //                   std::to_string( p ) );
00206 
00207         // Advance the first m_ncomp (N-1) scalars
00208         if (info == 0) {
00209           ncomp_t i = 0;
00210           for (i=0; i<m_ncomp-1; ++i) {
00211             tk::real& par = particles( p, i, m_offset );
00212             // Advance first m_ncomp (K=N-1) scalars due to drift
00213             par += 0.5*(m_omega[i] - omega*par)*dt;
00214             // Advance first m_ncomp (K=N-1) particles with Cholesky-decomposed
00215             // lower triangle (diffusion matrix)
00216             for (ncomp_t j=0; j<m_ncomp-1; ++j)
00217               if (j<=i) {
00218                 tk::real dW;
00219                 m_rng.gaussian( stream, 1, &dW );
00220                 par += B[i][j] * sqrt(dt) * dW;
00221               }
00222           }
00223           // Compute the (N-1)th scalar from unit-sum
00224           tk::real& par = particles( p, i, m_offset );
00225           par = 1.0 - particles( p, 0, m_offset );
00226           for (i=1; i<m_ncomp-1; ++i) {
00227             par -= particles( p, i, m_offset );
00228           }
00229         }
00230       }
00231     }
