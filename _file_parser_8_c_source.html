00026 FileParser::FileParser( const std::string& filename ) : m_filename( filename )
00027 //******************************************************************************
00028 //  Constructor
00029 //! \param[in] filename File to be parsed by the parser
00030 //! \details This constructor does basic tests in an attempt to determine if the
00031 //!   file to be parsed exists and is in good shape and does associated error
00032 //!   handling. This file stream is local, only used for error checking, and
00033 //!   thus is not part of the object state here since the parser, inheriting
00034 //!   from FileParser, e.g., walker::InputDeckParser, parses by completely
00035 //!   outsourcing the parsing (to PEGTL), so there is no need to store the file
00036 //!   stream handle here.
00037 //! \author  J. Bakosi
00038 //******************************************************************************
00039 {
00040   // Make sure there is a filename
00041   Assert( !filename.empty(), "No filename specified" );
00042 
00043   // Local file stream handle
00044   std::ifstream q;
00045 
00046   // Check if file exists, throw exception if it does not
00047   q.open( filename, std::ifstream::in );
00048   ErrChk( q.good(), "Failed to open file: " + filename );
00049 
00050   // Attempt to read a character, throw if it fails
00051   // It is curious that on some systems opening a directory instead of a file
00052   // with the above ifstream::open() call does not set the failbit. Thus we get
00053   // here fine, so we try to read a character from it. If it is a directory or
00054   // an empty file the read will fail, so we throw. Read more at: http://
00055   // stackoverflow.com/questions/9591036/
00056   // ifstream-open-doesnt-set-error-bits-when-argument-is-a-directory.
00057   q.get();
00058   ErrChk( q.good(), "Failed to read from file: " + filename );
00059 
00060   // Close it
00061   q.close();
00062   ErrChk( !q.fail(), "Failed to close file: " + filename );
00063 }

00090     // decide if error or warning
00091     char underchar = ' ';
00092     if (e.find( "Error" ) != std::string::npos) { err = true; underchar = '^'; }

00096       auto sloc = e.find( "at " );
00097       if (sloc != std::string::npos) {  // if we have location info
00098         // skip "at "
00099         sloc += 3;
00100         // find a comma starting from after "at "
00101         auto eloc = e.find_first_of( ',', sloc );
00102         // extract line number of error from error message
00103         const std::size_t lnum = std::stoul( e.substr( sloc, eloc-sloc ) );
00104         // store number of digits in line number
00105         const auto dlnum = eloc - sloc;
00106         // find a dot starting from after "at "
00107         eloc = e.find_first_of( '.', sloc );
00108         // skip line number
00109         sloc = e.find_first_of( ',', sloc ) + 1;
00110         // extract column number of error from error message
00111         const decltype(sloc) cnum = std::stoul( e.substr( sloc, eloc-sloc ) )-1;
00112         // store erroneous line information in map
00113         auto& l = lines[ lnum ];
00114         // store number of digits in line number
00115         l.dlnum = dlnum;
00116         // get erroneous line from file and store
00117         l.parsed = id.line( lnum );
00118         // store message
00119         l.msg.push_back( e );

00121         if (l.underline.empty()) l.underline = std::string(l.parsed.size(),' ');
00122         // find beginning of erroneous argument, this can be found in either e
00123         // (the full error message which may contain the erroneous substring
00124         // between single quotes and can also contain white space), or in
00125         // l.parsed (the erroneouss line from the file), if we find a
00126         // singly-quoted substring in e, we find the location of that substring
00127         // in l.parsed, if we don't find a singly-quoted substring in e,
00128         // we reverse-search l.parsed until a white space, in which case the
00129         // error that will be underlined will be a single word
00130         sloc = e.find( '\'' );
00131         if (sloc == std::string::npos) {
00132           sloc = l.parsed.rfind( ' ', cnum-1 );
00133         } else {
00134           auto sloc2 = e.find( '\'', sloc+1 );
00135           sloc = l.parsed.find( e.substr( sloc+1, sloc2-sloc-1 ) ) - 1;
00136         }
00137         // special-handle the beginning of the line with no space in front of it
00138         if (sloc == std::string::npos) sloc = 0; else ++sloc;
00139         // underline error and warning differently
00140         for (auto i=sloc; i<cnum; ++i) l.underline[i] = underchar;
00141       }
00142     }
00143   }
