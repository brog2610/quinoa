00077 //! Charm++ message type for sending a vector of strings of strings separated by
00078 //! semi-colons. The size of the vector must be determined at compile-time and
00079 //! comes in via the template argument. This avoids using variable-size
00080 //! messages, which is more complicated, more error-prone, and most likely less
00081 //! amenable to optimizations. The constructor and the value accessor interface
00082 //! with std::vector< std::string >, but converted to POD so that the Charm++
00083 //! runtime system easily serialize the data and migrate it over the network.
00084 //! \author J. Bakosi
00085 template< std::size_t Size = 1 >
00086 struct VecStrsMsg : public CMessage_VecStrsMsg< Size > {
00087   using value_type = std::vector< std::vector< std::string > >;
00088   //! Constructor
00089   explicit VecStrsMsg( const std::vector< std::string >& v ) {
00090     Assert( Size <= v.size(), "Size of input vector to VecStrsMsg is larger "
00091                               "than specified in template argument" );
00092     for (std::size_t i=0; i<Size; ++i)
00093       strncpy( strings[i], v[i].c_str(), CSTYLE_STRLEN );
00094   }
00095   //! Value accessor
00096   value_type get() {
00097     value_type v( Size );
00098     for (std::size_t i=0; i<Size; ++i) {
00099       std::string s( strings[i] );
00100       boost::char_separator<char> sep(";");
00101       boost::tokenizer< boost::char_separator<char> > tokens( s, sep );
00102       for (const auto& t : tokens) v[i].push_back( t ); 
00103     }
00104     return v;
00105   }
00106   //! Data
00107   char strings[Size][CSTYLE_STRLEN];
