2 /*!
3  \file src/DiffEq/GeneralizedDirichlet.h
4  \author J. Bakosi
5  \date Mon 01 Jun 2015 02:35:05 PM MDT
6  \copyright 2012-2015, Jozsef Bakosi.
7  \brief Lochner's generalized Dirichlet SDE
8  \details This file implements the time integration of a system of stochastic
9  differential equations (SDEs) whose invariant is Lochner's [generalized
10  Dirichlet distribution](http://en.wikipedia.org/wiki/Generalized_Dirichlet_distribution).
11 
12  In a nutshell, the equation integrated governs a set of scalars, \f$0 \le
13  Y_i\f$, \f$i=1,\dots,K\f$, \f$\sum_{i=1}^KY_i\le1\f$, as
14  \f[ \begin{split}
15  \mathrm{d}Y_i(t) = \frac{\mathcal{U}_i}{2}\left\{ b_i\Big[S_i
16  \mathcal{Y}_K - (1-S_i)Y_i\Big] + Y_i\mathcal{Y}_K
17  \sum_{j=i}^{K-1}\frac{c_{ij}}{\mathcal{Y}_j}\right\}\mathrm{d}t +
18  \sqrt{\kappa_i Y_i \mathcal{Y}_K \mathcal{U}_i}\mathrm{d}W_i(t), \\
19  \qquad i=1,\dots,K, \end{split}
20  \f]
21  where \f$\mathrm{d}W_i(t)\f$ is an isotropic vector-valued [Wiener
22  process](http://en.wikipedia.org/wiki/Wiener_process) with independent
23  increments. The statistically stationary solution of the above coupled
24  system of nonlinear stochastic differential equations is the generalized
25  Dirichlet distribution,
26  \f[
27  \newcommand{\bv}[1]{{\mbox{$\mathbf{#1}$}}}
28  \mathscr{G}(\bv{Y},\bv{\alpha},\bv{\beta}) =
29  \prod_{i=1}^K\frac{\Gamma(\alpha_i+\beta_i)}{\Gamma(\alpha_i)
30  \Gamma(\beta_i)}Y_i^{\alpha_i-1} \mathcal{Y}_i^{\gamma_i} \qquad
31  \mathrm{with} \qquad \mathcal{Y}_i = 1-\sum_{k=1}^i Y_k,
32  \f]
33  provided the coefficients, \f$b_i\!>\!0\f$, \f$\kappa_i\!>\!0\f$,
34  \f$0\!<\!S_i\!<\!1\f$, and \f$c_{ij}\f$, with \f$c_{ij}\!=\!0\f$ for
35  \f$i\!>\!j\f$, \f$i,j\!=\!1,\dots,K\!-\!1\f$, satisfy
36  \f[ \begin{split}
37  \alpha_i & = \frac{b_i}{\kappa_i}S_i, \qquad i=1,\dots,K,\\
38  1-\gamma_i & = \frac{c_{1i}}{\kappa_1} = \dots = \frac{c_{ii}}{\kappa_i},
39  \qquad i=1,\dots,K-1,\\
40  1+\gamma_K & = \frac{b_1}{\kappa_1}(1-S_1) = \dots =
41  \frac{b_K}{\kappa_K}(1-S_K). \end{split}
42  \f]
43 
44  Here \f$\mathcal{U}_i = \prod_{j=1}^{K-i}\mathcal{Y}_{K-j}^{-1}\f$,
45  \f$\alpha_i>0\f$, and \f$\beta_i>0\f$ are parameters, while
46  \f$\gamma_i=\beta_i-\alpha_{i+1}-\beta_{i+1}\f$ for \f$i=1,\dots,K-1\f$, and
47  \f$\gamma_K=\beta_K-1\f$. \f$\Gamma(\cdot)\f$ denotes the [gamma
48  function](http://en.wikipedia.org/wiki/Gamma_function). To keep the
49  invariant distribution generalized Dirichlet, the above set of constraints
50  on the coefficients must be satisfied. For more details on the generalized
51  Dirichlet SDE, see http://dx.doi.org/10.1063/1.4822416.
52 */

111  //! \brief Advance particles according to the generalized Dirichlet SDE
112  //! \param[inout] particles Array of particle properties
113  //! \param[in] stream Thread (or more precisely stream) ID
114  //! \param[in] dt Time step size
115  //! \param[in] moments Map of statistical moments
116  //! \author J. Bakosi
117  void advance( tk::ParProps& particles,
118  int stream,
119  tk::real dt,
120  const std::map< tk::ctr::Product, tk::real >& moments )
121  {
122  const auto npar = particles.npar();
123  for (auto p=decltype(npar){0}; p<npar; ++p) {
124  // Y_i = 1 - sum_{k=1}^{i} y_k
125  tk::real Y[m_ncomp];
126  Y[0] = 1.0 - particles( p, 0, m_offset );
127  for (ncomp_t i=1; i<m_ncomp; ++i)
128  Y[i] = Y[i-1] - particles( p, i, m_offset );
129 
130  // U_i = prod_{j=1}^{K-i} 1/Y_{K-j}
131  tk::real U[m_ncomp];
132  U[m_ncomp-1] = 1.0;
133  for (long i=static_cast<long>(m_ncomp)-2; i>=0; --i) U[i] = U[i+1]/Y[i];
134 
135  // Generate Gaussian random numbers with zero mean and unit variance
136  tk::real dW[m_ncomp];
137  m_rng.gaussian( stream, m_ncomp, dW );
138 
139  // Advance first m_ncomp (K=N-1) scalars
140  ncomp_t k=0;
141  for (ncomp_t i=0; i<m_ncomp; ++i) {
142  tk::real& par = particles( p, i, m_offset );
143  tk::real d = m_k[i] * par * Y[m_ncomp-1] * U[i] * dt;
144  d = (d > 0.0 ? sqrt(d) : 0.0);
145  tk::real a=0.0;
146  for (ncomp_t j=i; j<m_ncomp-1; ++j) a += m_cij[k++]/Y[j];
147  par += U[i]/2.0*( m_b[i]*( m_S[i]*Y[m_ncomp-1] - (1.0-m_S[i])*par ) +
148  par*Y[m_ncomp-1]*a )*dt + d*dW[i];
149  }
150  }
151  }
