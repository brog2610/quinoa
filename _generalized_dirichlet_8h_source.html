00007   \brief     Lochner's generalized Dirichlet SDE
00008   \details   This file implements the time integration of a system of stochastic
00009     differential equations (SDEs) whose invariant is Lochner's [generalized
00010     Dirichlet distribution](http://en.wikipedia.org/wiki/Generalized_Dirichlet_distribution).
00011 
00012     In a nutshell, the equation integrated governs a set of scalars, \f$0 \le
00013     Y_i\f$, \f$i=1,\dots,K\f$, \f$\sum_{i=1}^KY_i\le1\f$, as
00014     \f[ \begin{split}
00015       \mathrm{d}Y_i(t) = \frac{\mathcal{U}_i}{2}\left\{ b_i\Big[S_i
00016       \mathcal{Y}_K - (1-S_i)Y_i\Big] + Y_i\mathcal{Y}_K
00017       \sum_{j=i}^{K-1}\frac{c_{ij}}{\mathcal{Y}_j}\right\}\mathrm{d}t +
00018       \sqrt{\kappa_i Y_i \mathcal{Y}_K \mathcal{U}_i}\mathrm{d}W_i(t), \\
00019       \qquad i=1,\dots,K, \end{split}
00020     \f]
00021     where \f$\mathrm{d}W_i(t)\f$ is an isotropic vector-valued [Wiener
00022     process](http://en.wikipedia.org/wiki/Wiener_process) with independent
00023     increments. The statistically stationary solution of the above coupled
00024     system of nonlinear stochastic differential equations is the generalized
00025     Dirichlet distribution,
00026     \f[
00027        \newcommand{\bv}[1]{{\mbox{$\mathbf{#1}$}}}
00028        \mathscr{G}(\bv{Y},\bv{\alpha},\bv{\beta}) =
00029        \prod_{i=1}^K\frac{\Gamma(\alpha_i+\beta_i)}{\Gamma(\alpha_i)
00030        \Gamma(\beta_i)}Y_i^{\alpha_i-1} \mathcal{Y}_i^{\gamma_i} \qquad
00031        \mathrm{with} \qquad \mathcal{Y}_i = 1-\sum_{k=1}^i Y_k,
00032     \f]
00033     provided the coefficients, \f$b_i\!>\!0\f$, \f$\kappa_i\!>\!0\f$,
00034     \f$0\!<\!S_i\!<\!1\f$, and \f$c_{ij}\f$, with \f$c_{ij}\!=\!0\f$ for
00035     \f$i\!>\!j\f$, \f$i,j\!=\!1,\dots,K\!-\!1\f$, satisfy
00036     \f[ \begin{split}
00037        \alpha_i & = \frac{b_i}{\kappa_i}S_i, \qquad i=1,\dots,K,\\
00038        1-\gamma_i & = \frac{c_{1i}}{\kappa_1} = \dots = \frac{c_{ii}}{\kappa_i},
00039        \qquad i=1,\dots,K-1,\\
00040        1+\gamma_K & = \frac{b_1}{\kappa_1}(1-S_1) = \dots =
00041        \frac{b_K}{\kappa_K}(1-S_K). \end{split}
00042     \f]
00043 
00044     Here \f$\mathcal{U}_i = \prod_{j=1}^{K-i}\mathcal{Y}_{K-j}^{-1}\f$,
00045     \f$\alpha_i>0\f$, and \f$\beta_i>0\f$ are parameters, while
00046     \f$\gamma_i=\beta_i-\alpha_{i+1}-\beta_{i+1}\f$ for \f$i=1,\dots,K-1\f$, and
00047     \f$\gamma_K=\beta_K-1\f$. \f$\Gamma(\cdot)\f$ denotes the [gamma
00048     function](http://en.wikipedia.org/wiki/Gamma_function). To keep the
00049     invariant distribution generalized Dirichlet, the above set of constraints
00050     on the coefficients must be satisfied. For more details on the generalized
00051     Dirichlet SDE, see http://dx.doi.org/10.1063/1.4822416.
00052 */

00115     //! \param[in] moments Map of statistical moments
00116     //! \author J. Bakosi
00117     void advance( tk::ParProps& particles,
00118                   int stream,
00119                   tk::real dt,
00120                   const std::map< tk::ctr::Product, tk::real >& moments )
00121     {
00122       const auto npar = particles.npar();
00123       for (auto p=decltype(npar){0}; p<npar; ++p) {
00124         // Y_i = 1 - sum_{k=1}^{i} y_k
00125         tk::real Y[m_ncomp];
00126         Y[0] = 1.0 - particles( p, 0, m_offset );
00127         for (ncomp_t i=1; i<m_ncomp; ++i)
00128           Y[i] = Y[i-1] - particles( p, i, m_offset );
00129 
00130         // U_i = prod_{j=1}^{K-i} 1/Y_{K-j}
00131         tk::real U[m_ncomp];
00132         U[m_ncomp-1] = 1.0;
00133         for (long i=static_cast<long>(m_ncomp)-2; i>=0; --i) U[i] = U[i+1]/Y[i];
00134 
00135         // Generate Gaussian random numbers with zero mean and unit variance
00136         tk::real dW[m_ncomp];
00137         m_rng.gaussian( stream, m_ncomp, dW );
00138 
00139         // Advance first m_ncomp (K=N-1) scalars
00140         ncomp_t k=0;
00141         for (ncomp_t i=0; i<m_ncomp; ++i) {
00142           tk::real& par = particles( p, i, m_offset );
00143           tk::real d = m_k[i] * par * Y[m_ncomp-1] * U[i] * dt;
00144           d = (d > 0.0 ? sqrt(d) : 0.0);
00145           tk::real a=0.0;
00146           for (ncomp_t j=i; j<m_ncomp-1; ++j) a += m_cij[k++]/Y[j];
00147           par += U[i]/2.0*( m_b[i]*( m_S[i]*Y[m_ncomp-1] - (1.0-m_S[i])*par ) +
00148                             par*Y[m_ncomp-1]*a )*dt + d*dW[i];
00149         }
00150       }
00151     }
