00104 //! \brief Charm++ main chare for the shock hydroddynamics executable, inciter.
00105 //! \details In inciter the Charm++ runtime system is initialized only after the
00106 //!   mesh has been read in, partitioned, and the necessary data structures,
00107 //!   e.g., communication maps, have been generated. This delayed initialization
00108 //!   of the Charm++ runtime system is required since the mesh partitioning is

00114   public:
00115     //! \brief Constructor
00116     //! \details The main chare constructor is the main entry point of the
00117     //!   Charm++ portion of inciter, called by the Charm++ runtime system. The
00118     //!   constructor does basic initialization steps, prints out some useful
00119     //!   information to screen (in verbose mode), and instantiates a driver.
00120     //!   Since Charm++ is fully asynchronous, the constructor usually spawns
00121     //!   asynchronous objects and immediately exits. Thus in the body of the
00122     //!   main chare constructor we fire up an 'execute' chare, which then calls
00123     //!   back to Main::execute(). Finishing the main chare constructor the
00124     //!   Charm++ runtime system then starts the network-migration of all
00125     //!   global-scope data (if any). The execute chare calling back to
00126     //!   Main::execute() signals the end of the migration of the global-scope
00127     //!   data. Then we are ready to execute the driver. Since inciter is
00128     //!   parallel and asynchronous, its driver fires up additional Charm++
00129     //!   chare objects which then call back to Main::finalize() at some point
00130     //!   in the future when all work has been finished. finalize() then exits
00131     //!   by calling Charm++'s CkExit(), shutting down the runtime system.
00132     //! \see http://charm.cs.illinois.edu/manuals/html/charm++/manual.html
00133     Main( CkArgMsg* msg )
00134     try :
00135       // Create pretty printer initializing output streams based on command line
00136       m_print( inciter::g_cmdline.get< tag::verbose >()
00137                  ? std::cout : std::clog ),
00138       // Create Inciter driver
00139       m_driver( inciter::InciterDriver( m_print ) ),
00140       // Start new timer measuring the total runtime
00141       m_timer(1),
00142       // Import, i.e., swallow, timers from the initial MPI portion
00143       m_timestamp( std::move(inciter::g_timestamp) )
00144     {
00145       const auto helpcmd = inciter::g_cmdline.get< tag::help >();
00146       const auto helpctr = inciter::g_cmdline.get< tag::helpctr >();
00147       const auto helpkw = inciter::g_cmdline.get< tag::helpkw >();
00148       // Exit if help was requested or exectuable was called without argument
00149       if (msg->argc == 1 || helpcmd || helpctr || !helpkw.keyword.empty()) {
00150         CkExit();
00151       } else {  // business as usual
00152         mainProxy = thisProxy;
00153         // Fire up an asynchronous execute object, which when created at some
00154         // future point in time will call back to this->execute(). This is
00155         // necessary so that this->execute() can access already migrated

00157         CProxy_execute::ckNew();
00158         // Start new timer measuring the migration of global-scope data
