104 //! \brief Charm++ main chare for the shock hydroddynamics executable, inciter.
105 //! \details In inciter the Charm++ runtime system is initialized only after the
106 //! mesh has been read in, partitioned, and the necessary data structures,
107 //! e.g., communication maps, have been generated. This delayed initialization
108 //! of the Charm++ runtime system is required since the mesh partitioning is
109 //! done by Zoltan, an MPI library. Note that this Charm++ main chare object
110 //! should not be in a namespace.
111 //! \author J. Bakosi
112 class Main : public CBase_Main {
113 
114  public:
115  //! \brief Constructor
116  //! \details The main chare constructor is the main entry point of the
117  //! Charm++ portion of inciter, called by the Charm++ runtime system. The
118  //! constructor does basic initialization steps, prints out some useful
119  //! information to screen (in verbose mode), and instantiates a driver.
120  //! Since Charm++ is fully asynchronous, the constructor usually spawns
121  //! asynchronous objects and immediately exits. Thus in the body of the
122  //! main chare constructor we fire up an 'execute' chare, which then calls
123  //! back to Main::execute(). Finishing the main chare constructor the
124  //! Charm++ runtime system then starts the network-migration of all
125  //! global-scope data (if any). The execute chare calling back to
126  //! Main::execute() signals the end of the migration of the global-scope
127  //! data. Then we are ready to execute the driver. Since inciter is
128  //! parallel and asynchronous, its driver fires up additional Charm++
129  //! chare objects which then call back to Main::finalize() at some point
130  //! in the future when all work has been finished. finalize() then exits
131  //! by calling Charm++'s CkExit(), shutting down the runtime system.
132  //! \see http://charm.cs.illinois.edu/manuals/html/charm++/manual.html
133  Main( CkArgMsg* msg )
134  try :
135  // Create pretty printer initializing output streams based on command line
136  m_print( inciter::g_cmdline.get< tag::verbose >()
137  ? std::cout : std::clog ),
138  // Create Inciter driver
139  m_driver( inciter::InciterDriver( m_print ) ),
140  // Start new timer measuring the total runtime
141  m_timer(1),
142  // Import, i.e., swallow, timers from the initial MPI portion
143  m_timestamp( std::move(inciter::g_timestamp) )
144  {
145  const auto helpcmd = inciter::g_cmdline.get< tag::help >();
146  const auto helpctr = inciter::g_cmdline.get< tag::helpctr >();
147  const auto helpkw = inciter::g_cmdline.get< tag::helpkw >();
148  // Exit if help was requested or exectuable was called without argument
149  if (msg->argc == 1 || helpcmd || helpctr || !helpkw.keyword.empty()) {
150  CkExit();
151  } else { // business as usual
152  mainProxy = thisProxy;
153  // Fire up an asynchronous execute object, which when created at some
154  // future point in time will call back to this->execute(). This is
155  // necessary so that this->execute() can access already migrated
156  // global-scope data.
157  CProxy_execute::ckNew();
158  // Start new timer measuring the migration of global-scope data
159  m_timer.emplace_back();
160  }
161  delete msg;
162  } catch (...) { tk::processExceptionCharm(); }
163 
164  //! Execute driver created and initialized by constructor
165  void execute() {
166  try {
167  m_timestamp.emplace_back( "Migrate Charm++ read-only global-scope data",
168  m_timer[1].hms());
169  m_driver.execute();
170  } catch (...) { tk::processExceptionCharm(); }
171  }
172 
173  //! Normal exit point
174  void finalize() {
175  try {
176  if (!m_timer.empty()) {
177  m_timestamp.emplace_back(
178  "Total Charm++ runtime . . . . . . . . . . . . . . . . . . . . . . "
179  ". . . . .",
180  m_timer[0].hms() );
181  m_print.time( "Timers (h:m:s)", m_timestamp );
182  m_print.perf( "Performance statistics", m_perf );
183  m_print.endpart();
184  }
185  } catch (...) { tk::processExceptionCharm(); }
186  // Tell the Charm++ runtime system to exit
187  CkExit();
188  }
189

191  void timestamp( std::string label, tk::real stamp ) {
192  try {
193  m_timestamp.emplace_back( label, tk::hms( stamp ) );
194  } catch (...) { tk::processExceptionCharm(); }
195  }
196  //! Add multiple time stamps contributing to final timers output
197  void timestamp( const std::vector< std::pair< std::string, tk::real > >& s )
198  { for (const auto& t : s) timestamp( t.first, t.second ); }
199

201  void perfstat( std::string label, tk::real value ) {

203  m_perf.emplace_back( label, value );
204  } catch (...) { tk::processExceptionCharm(); }
205  }
206  //! Add multiple performance statistics contributing to final perf output
207  void perfstat( const std::vector< std::pair< std::string, tk::real > >& p )
208  { for (const auto& s : p) perfstat( s.first, s.second ); }

229 //! \brief Inciter main()
230 //! \details Inciter does have a main() function so that Zoltan, an MPI library,
231 //! can partition the mesh. Then we initialize the Charm++ runtime system and
232 //! do a calculation. This is necessary, since MPI_Init() is a bit adamant
233 //! about capturing resources it wants and hence it has to be called before
234 //! Charm is initialized.
235 //! \author J. Bakosi
236 int main( int argc, char **argv ) {
237 
238  using namespace inciter;
239 
240  tk::Timer mpi; // start timing the MPI portion
241 
242  // Initialize MPI
243  MPI_Init( &argc, &argv );
244 
245  try {
246  // Parse command line
247  parseCmdLine( argc, argv, g_cmdline );
248 
249  // Instantiate inciter's pretty printer
250  InciterPrint
251  iprint( g_cmdline.get< tag::verbose >() ? std::cout : std::clog );
252 
253  // Parse input deck, echo info
254  init( g_cmdline, iprint, g_inputdeck, argc, argv );
255 
256  // Prepare computational mesh and fill some global-scope data (all given at
257  // the top of this file) so that Charm++ chares can access them
258  prepareMesh( g_cmdline, iprint, g_inputdeck, // <- const
259  g_timestamp, g_npoin, g_point, g_element, g_meshfilemap,
260  g_tetinpoel, g_esup, g_pcomm, g_ecomm );
261 
262  } catch (...) { tk::processExceptionMPI(); }
263 
264  g_timestamp.emplace_back(
265  "Total MPI runtime . . . . . . . . . . . . . . . . . . . . . . . . . . "
266  ". . .",
267  mpi.hms() );
268 
269  // Run Charm++ main chare using the partitioned graph
270  CharmLibInit( MPI_COMM_WORLD, argc, argv );
271  //MPI_Barrier( MPI_COMM_WORLD );
272  // ...
273  //MPI_Barrier( MPI_COMM_WORLD );
274  CharmLibExit();
275 
276  // Finalize MPI
277  MPI_Barrier( MPI_COMM_WORLD );
278  MPI_Finalize();
279 
280  return tk::ErrCode::SUCCESS;
281 }
