00060 UnsMeshMap::UnsMeshMap( std::size_t npoin,
00061                         const std::vector< std::vector< std::size_t > >& point )
00062 //******************************************************************************
00063 // Constructor
00064 //! \param[in] npoin Total number of points in mesh
00065 //! \param[in] point Global mesh point ids owned by each array element
00066 //! \author J. Bakosi
00067 //******************************************************************************
00068 {
00069   Assert( npoin > 0, "Need at least a single mesh point" );
00070   Assert( point.size() >= CkNumPes(),
00071           "UnsMeshMap only works with nchare >= numPEs" );
00072 
00073   // Vector of maps to associate the number of mesh points an array element
00074   // contributes to a pe to the array element
00075   std::vector< std::map< std::size_t, std::size_t > > owner( point.size() );
00076 
00077   // Compute number of mesh points per PE (chunksize)
00078   const auto numpes = static_cast< std::size_t >( CkNumPes() );
00079   const auto chunksize = npoin>numpes ? npoin/numpes : 1;
00080 
00081   // Count up number of mesh points contributing to a PE for each array element
00082   for (std::size_t e=0; e<point.size(); ++e)    // for all array elements
00083     for (auto p : point[e]) {           // for all points array element e owns
00084       auto pe = p / chunksize;          // PE array element e contributes to
00085       if (pe == CkNumPes()) --pe;
00086       ++owner[e][pe];  // count up number of points element e contributing to pe
00087     }
00088 
00089   // Find and store the PE with the largest number of points owned by an array
00090   // element. This is the key (PE) of the largest mapped_type of the element's
00091   // map, owner[e].
00092   using pr = decltype(owner)::value_type::value_type;
00093   m_pe.resize( owner.size() );
00094   for (std::size_t e=0; e<owner.size(); ++e) {
00095     const auto most = std::max_element( begin(owner[e]), end(owner[e]),
00096                                         []( const pr& p1, const pr& p2 )
00097                                           { return p1.second < p2.second; } );
00098     m_pe[e] = most->first;
00099   }
00100 
00101   // Check if all PEs have at least one array element to create, fix if not
00102   fixPEs();
00103 }

00106 UnsMeshMap::fixPEs()
00107 //******************************************************************************
00108 //  Check that all PEs create at least a single array element, fix if not
00109 //! \details This is required because if the array elements, placed using this
00110 //!   map object, send messages to some Charm++ chare group branches and the
00111 //!   group happens to use Charm++'s structured dagger, such as LinSysMerger,
00112 //!   memory problems will occur.
00113 //! \author J. Bakosi
00114 //******************************************************************************
00115 {
00116   // Build unique set of PEs the array elements are assigned to
00117   std::set< std::size_t > nkind;
00118   for (auto p : m_pe) nkind.insert( p );
00119 
00120 //   std::cout << CkMyPe() << ": ";
00121 //   for (auto p : m_pe) std::cout << p << " ";
00122 //   std::cout << std::endl;
00123 
00124   // If not all PEs have at least one array element to create, go through all
00125   // PEs that have no array elements assigned, and construct a map that
00126   // associates the number of PEs to array elements. Then as a replacement PE,
00127   // pick the PE that has the most array elements assigned. Then assign the
00128   // first of the elements assigned to the PE with the most work to the
00129   // replacement PE.
00130   if (nkind.size() != CkNumPes())
00131     for (std::size_t p=0; p<CkNumPes(); ++p)
00132       if (!nkind.count(p)) {    // if PE p has no array elements to create
00133         // Count up number elements for each PE
00134         std::map< std::size_t, std::size_t > npe;
00135         using pr = decltype(npe)::value_type;
00136         for (auto q : m_pe) ++npe[q];
00137         // Find the PE with the most array elements assigned
00138         const auto most = std::max_element( begin(npe), end(npe),
00139                                             []( const pr& p1, const pr& p2 )
00140                                             { return p1.second < p2.second; } );
00141         for (std::size_t q=0; q<m_pe.size(); ++q)
00142           if (m_pe[q] == most->first) { // first element of the most loaded PE
00143             //std::cout << CkMyPe() << ": replace " << m_pe[q] << " by " << p
00144             //          << " for e " << q << std::endl;
00145             m_pe[q] = p;        // replace PE of m_pe[q] of element q by p
00146             q = m_pe.size();    // quit (enough to replace one)
00147           }
00148       }
00149 
00150   nkind.clear();
00151   for (auto p : m_pe) nkind.insert( p );
00152   Assert( nkind.size() == CkNumPes(),
00153           "There are still PE(s) with no array elements assigned" );
00154 }
