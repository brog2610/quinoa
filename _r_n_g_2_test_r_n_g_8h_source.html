23  //! Test uniform distribution of a random number generator
24  //! \param[in] r RNG to test
25  //! \details In real code the member function used to generate random numbers
26  //! are called by different threads, but here we pass a different thread ID
27  //! (first argument to the member function called) will exercise multiple
28  //! streams in serial, i.e., emulate multi-threaded RNG on a single thread.
29  //! This is done here this way because the unit test harness deals out each
30  //! test to a PE, thus calling the generators from real threads would create
31  //! contention.
32  template< class rng >
33  static void test_uniform( const rng& r ) {
34  std::size_t num = 100000;
35  std::vector< double > numbers( num );
36  auto n = r.nthreads();
37  for (std::size_t i=0; i<n; ++i)
38  r.uniform( static_cast<int>(i), num/n, &numbers[i*num/n] );
39  for (auto n : numbers) ensure( "sample space incorrect", 0.0<n && n<1.0 );
40  test_stats( numbers, 0.5, 1.0/12.0, 0.0, -6.0/5.0 );
41  }

77  auto n = r.nthreads();
78  for (std::size_t i=0; i<n; ++i)

82  ensure( "sample space incorrect", -eps<n && n<1.0+eps );
83  // test first four moments
84  test_stats( numbers, a/(a+b), a*b/(a+b)/(a+b)/(a+b+1.0),

87  // test scaled sample space

130  // Test the first the four first moments of random numbers passed in
131  static void test_stats( const std::vector< double >& numbers,
132  double correct_mean,
133  double correct_variance,
134  double correct_skewness,
135  double correct_excess_kurtosis )
136  {
137  const double precision = 0.05;
138  std::array< double, 4 > s{{ 0.0, 0.0, 0.0, 0.0 }};
139  for (auto n : numbers) s[0] += n;
140  s[0] /= static_cast< double >( numbers.size() );
141  ensure_equals( "mean inaccurate", s[0], correct_mean, precision );
142  for (auto n : numbers) {
143  s[1] += (n-s[0])*(n-s[0]);
144  s[2] += (n-s[0])*(n-s[0])*(n-s[0]);
145  s[3] += (n-s[0])*(n-s[0])*(n-s[0])*(n-s[0]);
146  }
147  s[1] /= static_cast< double >( numbers.size() );
148  s[2] /= static_cast< double >( numbers.size() );
149  s[3] /= static_cast< double >( numbers.size() );
150  ensure_equals( "variance inaccurate", s[1], correct_variance, precision );
151  ensure_equals( "skewness inaccurate",
152  s[2]/std::pow(s[1],1.5), correct_skewness, precision );
153  ensure_equals( "excess kurtosis inaccurate",
154  s[3]/s[1]/s[1]-3.0, correct_excess_kurtosis, precision );
155  }
