00077 std::map< tk::ctr::RawRNGType, tk::RNG >
00078 RNGStack::selected( const std::vector< tk::ctr::RNGType >& sel ) const
00079 //******************************************************************************
00080 //  Instantiate selected RNGs from factory and place them in map
00081 //! \param[in] sel Vector of selected RNGs to instantiate (selected by the user)
00082 //! \return A std::map of keys and their associated instantiated RNG objects
00083 //! \author  J. Bakosi
00084 //******************************************************************************
00085 {
00086   using tk::ctr::RawRNGType;
00087   std::map< RawRNGType, tk::RNG > rng;
00088   for (const auto& s : sel) {
00089     const auto r = m_factory.find(s);
00090     if (r != end(m_factory))
00091       rng.emplace( static_cast< RawRNGType >( s ), r->second() );
00092     else Throw( "RNG not found in factory" );
00093   }
00094   return rng;
00095 }
