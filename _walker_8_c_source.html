00090 //! Pack/Unpack selected RNGs. This Pack/Unpack method (re-)creates the full RNG
00091 //! stack since it needs to (re-)bind function pointers on different processing
00092 //! elements. Therefore we circumvent Charm's usual pack/unpack for this type,
00093 //! and thus sizing does not make sense: sizing is a no-op. We could initialize
00094 //! the stack in RNGTestDriver's constructor and let this function re-create the
00095 //! stack only when unpacking, but that leads to repeating the same code twice:
00096 //! once in RNGTestDriver's constructor, once here. Another option is to use
00097 //! this pack/unpack routine to both initially create (when packing) and to
00098 //! re-create (when unpacking) the stack, which eliminates the need for
00099 //! pre-creating the object in RNGTestDriver's constructor and therefore
00100 //! eliminates the repeated code. This explains the guard for sizing: the code
00101 //! below is called for packing only (in serial) and packing and unpacking (in
00102 //! parallel).
00103 inline
00104 void operator|( PUP::er& p, std::map< tk::ctr::RawRNGType, tk::RNG >& rng ) {
00105   if (!p.isSizing()) {
00106     tk::RNGStack stack(
00107       #ifdef HAS_MKL
00108       g_inputdeck.get< tag::param, tag::rngmkl >(),
00109       #endif
00110       g_inputdeck.get< tag::param, tag::rngsse >() );
00111     rng = stack.selected( g_inputdeck.get< tag::selected, tag::rng >() );
00112   }
00113 }

00139   public:
00140     //! \brief Constructor
00141     //! \details The main chare constructor is the main entry point of the
00142     //!   program, called by the Charm++ runtime system. The constructor does
00143     //!   basic initialization steps, e.g., parser the command-line, prints out
00144     //!   some useful information to screen (in verbose mode), and instantiates
00145     //!   a driver. Since Charm++ is fully asynchronous, the constructor
00146     //!   usually spawns asynchronous objects and immediately exits. Thus in the
00147     //!   body of the main chare constructor we fire up an 'execute' chare,
00148     //!   which then calls back to Main::execute(). Finishing the main chare
00149     //!   constructor the Charm++ runtime system then starts the
00150     //!   network-migration of all global-scope data (if any). The execute chare
00151     //!   calling back to Main::execute() signals the end of the migration of
00152     //!   the global-scope data. Then we are ready to execute the driver. Since
00153     //!   the random walker is parallel and asynchronous, its driver fires up
00154     //!   additional Charm++ chare objects which then call back to
00155     //!   Main::finalize() at some point in the future when all work has been
00156     //!   finished. finalize() then exits by calling Charm++'s CkExit(),
00157     //!   shutting down the runtime system.
00158     //! \see http://charm.cs.illinois.edu/manuals/html/charm++/manual.html
00159     Main( CkArgMsg* msg )
00160     try :
00161       // Parse command line into m_cmdline using default simple pretty printer
00162       m_cmdParser( msg->argc, msg->argv, tk::Print(), m_cmdline ),
00163       // Create pretty printer initializing output streams based on command line
00164       m_print( m_cmdline.get< tag::verbose >() ? std::cout : std::clog ),
00165       // Create Walker driver
00166       m_driver( tk::Main< walker::WalkerDriver >
00167                         ( msg->argc, msg->argv,
00168                           m_cmdline,
00169                           tk::HeaderType::WALKER,
00170                           WALKER_EXECUTABLE,
00171                           m_print ) ),
00172       m_timer(1)        // Start new timer measuring the total runtime
00173     {
00174       delete msg;
00175       mainProxy = thisProxy;
00176       // Fire up an asynchronous execute object, which when created at some
00177       // future point in time will call back to this->execute(). This is
00178       // necessary so that this->execute() can access already migrated
00179       // global-scope data.
00180       CProxy_execute::ckNew();
00181       // Start new timer measuring the migration of global-scope data
00182       m_timer.emplace_back();
00183     } catch (...) { tk::processExceptionCharm(); }
