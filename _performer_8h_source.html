00149     //! Compute initial conditions for dispersion in simple shear flow

00151     ansol_shear( std::size_t i, tk::real t ) const {
00152       const tk::real X0 = 0.5;           // x position of source
00153       const tk::real Y0 = 0.5;           // y position of source
00154       const tk::real U0 = 0.5;           // velocity in x direction
00155       const tk::real t0 = 2.4;           // initial time
00156       const tk::real LAMBDA = 5.0e-4;    // amount of shear, (lambda = du/dy)
00157       const tk::real D = 10.0;           // scalar diffusivity
00158       const tk::real MASS = 4 * M_PI * t0 *
00159                             std::sqrt( 1.0 + LAMBDA * LAMBDA * t0 * t0 / 12.0 );
00160       const auto& x = m_coord[0];
00161       const auto& y = m_coord[1];
00162       tk::real a = x[i] - X0 - U0*t - 0.5*LAMBDA*y[i]*t;
00163       tk::real b = 1.0 + LAMBDA*LAMBDA*t*t/12.0;
00164       return MASS * exp( -a*a/(4*D*t*b) - y[i]*y[i]/(4*D*t) )
00165                   / ( 4.0 * M_PI * t * std::sqrt(b) );
00166     };
