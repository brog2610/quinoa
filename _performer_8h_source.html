135  //! Compute initial conditions for dispersion in simple shear flow
136  tk::real ansol_shear( std::size_t i, tk::real t ) const {
137  const tk::real X0 = 7200.0; // x position of source
138  const tk::real t0 = g_inputdeck.get< tag::discr, tag::t0 >(); // initial t
139  const tk::real U0 = 0.5; // velocity in x direction
140  const tk::real LAMBDA = 5.0e-4; // amount of shear, (lambda = du/dy)
141  const tk::real D = 10.0; // scalar diffusivity
142  const tk::real M = 4.0*M_PI*t0*std::sqrt( 1.0 + LAMBDA*LAMBDA*t0*t0/12.0 );
143  const auto& x = m_coord[0];
144  const auto& y = m_coord[1];
145  tk::real a = x[i] - X0 - U0*t - 0.5*LAMBDA*y[i]*t;
146  tk::real b = 1.0 + LAMBDA*LAMBDA*t*t/12.0;
147  return M * exp( -a*a/(4.0*M_PI*D*t*b) - y[i]*y[i]/(4.0*D*t) )
148  / ( 4.0*M_PI*t*std::sqrt(b) );
149  };
150 
151  //! Compute initial conditions representing a joint Gaussian
152  tk::real ansol_gauss( std::size_t i ) const {
153  const tk::real X0 = 0.5, Y0 = 0.5, Z0 = 0.5;
154  const tk::real v1 = 0.1, v2 = 0.05, v3 = 0.05;
155  const auto& x = m_coord[0];
156  const auto& y = m_coord[1];
157  const auto& z = m_coord[2];
158  const tk::real rx = x[i]-X0;
159  const tk::real ry = y[i]-Y0;
160  const tk::real rz = z[i]-Z0;
161  return std::exp( -0.5 * (rx*rx/v1 + ry*ry/v2 + rz*rz/v3) ) /
162  2.0 / M_PI / v1 / v2 / v3;
163  }
