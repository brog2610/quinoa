149  //! Compute initial conditions for dispersion in simple shear flow
150  tk::real
151  ansol_shear( std::size_t i, tk::real t ) const {
152  const tk::real X0 = 0.5; // x position of source
153  const tk::real Y0 = 0.5; // y position of source
154  const tk::real U0 = 0.5; // velocity in x direction
155  const tk::real t0 = 2.4; // initial time
156  const tk::real LAMBDA = 5.0e-4; // amount of shear, (lambda = du/dy)
157  const tk::real D = 10.0; // scalar diffusivity
158  const tk::real MASS = 4 * M_PI * t0 *
159  std::sqrt( 1.0 + LAMBDA * LAMBDA * t0 * t0 / 12.0 );
160  const auto& x = m_coord[0];
161  const auto& y = m_coord[1];
162  tk::real a = x[i] - X0 - U0*t - 0.5*LAMBDA*y[i]*t;
163  tk::real b = 1.0 + LAMBDA*LAMBDA*t*t/12.0;
164  return MASS * exp( -a*a/(4*D*t*b) - y[i]*y[i]/(4*D*t) )
165  / ( 4.0 * M_PI * t * std::sqrt(b) );
166  };

168  //! Compute initial conditions representing a joint Gaussian
169  tk::real
170  ansol_gauss( std::size_t i ) const {
171  const tk::real X0 = 0.5, Y0 = 0.5, Z0 = 0.5;
172  const tk::real v1 = 0.1, v2 = 0.05, v3 = 0.05;
173  const auto& x = m_coord[0];
174  const auto& y = m_coord[1];
175  const auto& z = m_coord[2];
176  const tk::real rx = x[i]-X0;
177  const tk::real ry = y[i]-Y0;
178  const tk::real rz = z[i]-Z0;
179  return std::exp( -0.5 * (rx*rx/v1 + ry*ry/v2 + rz*rz/v3) ) /
180  2.0 / M_PI / v1 / v2 / v3;
181  }
