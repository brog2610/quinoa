00020 //! Driver. The class below uses runtime polymorphism without client-side
00021 //! inheritance: inheritance is confined to the internals of the class below,
00022 //! inivisble to client-code. The class exclusively deals with ownership
00023 //! enabling client-side value semantics. Credit goes to Sean Parent at Adobe:
00024 //! https://github.com/sean-parent/sean-parent.github.com/wiki/

00030     template< typename T >
00031     explicit Driver( T x ) : self( make_unique< Model<T> >( std::move(x) ) ) {}

00033     //! Constructor taking a function pointer to a constructor of an object
00034     //! modeling Concept (see below). Passing std::function allows late
00035     //! execution of the constructor, i.e., as late as inside this class'
00036     //! constructor, and thus usage from a factory.
00037     template< typename T >
00038     explicit Driver( std::function<T()> x ) :
00039       self( make_unique< Model<T> >( std::move(x()) ) ) {}
00040 
00041     //! Public interface to execute

00049     //! Move assignment

00052     Driver( Driver&& ) noexcept = default;

00054   private:
00055     //! Concept is a pure virtual base class specifying the requirements of
00056     //! polymorphic objects deriving from it
00057     struct Concept {
00058       virtual ~Concept() = default;
00059       virtual Concept* copy() const = 0;
00060       virtual void execute() const = 0;
00061     };
