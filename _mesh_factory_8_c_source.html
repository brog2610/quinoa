00038   std::string s( Reader( filename ).firstline().substr(0,3) );
00039

00041     return MeshReader::GMSH;
00042   } else if ( s == "CDF" || s == "HDF" ) {
00043     return MeshReader::EXODUSII;

00045     try {

00063 //! \param[in] filename Filename to pick its type based on extension given
00064 //! \return enum specifying the mesh writer type

00071 
00072   if ( ext == "msh" ) {
00073     return MeshWriter::GMSH;
00074   } else if ( ext == "exo" || ext == "h5" ) {
00075     return MeshWriter::EXODUSII;

00077     return MeshWriter::NETGEN;
00078   } else {
00079     Throw( "Output mesh file type could not be determined from extension of "
00080            "filename '" + filename + "'; valid extensions are: "
