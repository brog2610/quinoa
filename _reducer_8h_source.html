57 //! \brief Deserialize and merge map of Key and vector of Ts from Charm's
58 //! CkReductionMsg

60 //! \return Merged map

63 std::map< Key, std::vector< T > >
64 merge( CkReductionMsg* msg )
65 {
66  using Map = std::map< Key, std::vector< T > >;

68  // Create PUP deserializer based on message passed in
