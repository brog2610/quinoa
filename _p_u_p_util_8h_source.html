00043 //! Pack/Unpack enum class. In Charm++ usually both the pup() overload and an
00044 //! overload for operator| are defined for all serializable types. However, we
00045 //! cannot define operator| for enum class as it would conflict with Charm++'s
00046 //! catch-all, template< class T > inline void operator|(PUP::er &p,T &t) {...}.
00047 //! \param[in] p Charm++'s pack/unpack object
00048 //! \param[in] e Enum class to pack/unpack
00049 //! \author J. Bakosi
00050 template< typename E,
00051           typename std::enable_if< std::is_enum< E >::value, int >::type = 0 >
00052 inline void pup( PUP::er& p, E& e ) {
00053   auto v = static_cast< typename std::underlying_type< E >::type >( e );
00054   p | v;
00055   e = static_cast< E >( v );
00056 }

00064 //! PUP_tuple_impl: specialization for empty std::tuple

00067 pup_tuple_impl( PUP::er&, std::tuple< Tp... >& ) {}
00068 //! PUP_tuple_impl: specialization for non-empty std::tuple

00071 pup_tuple_impl( PUP::er& p, std::tuple< Tp... >& t ) {
