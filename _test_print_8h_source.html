00100   auto p = prq.save< tk::QUIET >();  // save quiet stream pointer
00101   // Instantiate new printer using two new stringstreams. Note, if we use the

00107   std::stringstream v, q;

00110   pr.reset< tk::QUIET >( p );
00111   // Write to new quiet stream, now should have "blah blah"
