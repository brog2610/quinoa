40 average( const std::map< std::string, std::vector< T > >& mapvec,

42 //******************************************************************************
43 // Compute average of values of vector in std::map::mapped_type
44 //! \param[in] mapvec Map of vectors associated to labels
45 //! \param[in] addendum Optional string to add to the label
46 //! \return Vector of pairs of label and average
47 //! \author J. Bakosi

50  std::vector< std::pair< std::string, T > > s;

64 //******************************************************************************
65 // Compute variance of values of vector in std::map::mapped_type
66 //! \param[in] mapvec Map of vectors associated to labels
67 //! \param[in] avg Vector of labels and averages (labels unused)
68 //! \param[in] addendum Optional string to add to the label
69 //! \return Vector of pairs of label and variance
70 //! \author J. Bakosi
71 //******************************************************************************
72 {
73  Assert( mapvec.size() == avg.size(),
74  "Map and vector must be equal size for variance calculation" );
75 
76  std::vector< std::pair< std::string, T > > s;
77  std::size_t i = 0;
78  for (const auto& t : mapvec) {
79  T sum = 0.0;
80  for (const auto& v : t.second) sum += (v-avg[i].second)*(v-avg[i].second);
81  s.emplace_back( t.first + addendum, sum/static_cast<T>(t.second.size()) );
82  ++i;
83  }
